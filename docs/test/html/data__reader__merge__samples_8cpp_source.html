87 size_t data_reader_merge_samples::compute_num_samples_psum() {
88  size_t global_num_samples = 0;
89  // Prepend a 0 to make things easier.
90  m_num_samples_psum.push_back(0);
91  for (auto&& reader : m_data_readers) {
92  m_num_samples_psum.push_back(reader->get_num_data());
93  global_num_samples += reader->get_num_data();
94  }
95  std::partial_sum(m_num_samples_psum.begin(), m_num_samples_psum.end(),
96  m_num_samples_psum.begin());
97  return global_num_samples;
98 }

100 void data_reader_merge_samples::sanity_check_for_consistency(
101  int num_labels, int data_size, int label_size, const std::vector<int> &data_dims) {
102  for (auto&& reader : m_data_readers) {
103  if (num_labels != reader->get_num_labels()) {
104  throw lbann_exception(
105  "data_reader_merge_samples: data readers do not have the same number of labels");
106  }
107  if (data_size != reader->get_linearized_data_size()) {
108  throw lbann_exception(
109  "data_reader_merge_samples: data readers do not have the same data size");
110  }
111  if (label_size != reader->get_linearized_label_size()) {
112  throw lbann_exception(
113  "data_reader_merge_samples: data readers do not have the same label size");
114  }
115  if (data_dims != reader->get_data_dims()) {
116  throw lbann_exception(
117  "data_reader_merge_samples: data readers do not have the same data dims");
118  }
119  }

131 void data_reader_merge_samples::load() {
132  if (options::get()->has_bool("use_data_store") && options::get()->get_bool("use_data_store")) {
133  data_reader_merge_samples::load_using_data_store();
134  return;
135  }
136 
137  // Load each subsidiary data reader.
138  for (auto&& reader : m_data_readers) {
139  reader->set_comm(m_comm);
140  reader->load();
141  }
142  // Compute the total number of samples and do some sanity checks.
143  int num_labels = m_data_readers[0]->get_num_labels();
144  int data_size = m_data_readers[0]->get_linearized_data_size();
145  int label_size = m_data_readers[0]->get_linearized_label_size();
146  const std::vector<int> data_dims = m_data_readers[0]->get_data_dims();
147  sanity_check_for_consistency(num_labels, data_size, label_size, data_dims);
148 
149  size_t global_num_samples = compute_num_samples_psum();
150  setup_indices(global_num_samples);
151 }

153 bool data_reader_merge_samples::fetch_datum(Mat& X, int data_id, int mb_idx,
154  int tid) {
155  // Find the right data reader to delegate to.
156  for (size_t i = 0; i < m_data_readers.size(); ++i) {
157  if (data_id < m_num_samples_psum[i + 1]) {
158  data_id -= m_num_samples_psum[i];
159  return m_data_readers[i]->fetch_datum(X, data_id, mb_idx, tid);
160  }
161  }
162  throw lbann_exception(
163  "data_reader_merge_samples: do not have data ID " +
164  std::to_string(data_id));
165 }
