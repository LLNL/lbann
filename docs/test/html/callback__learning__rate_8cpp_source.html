43 void lbann_callback_learning_rate::setup(model *m) {
44 
45  // Add all weights if list of weights is not initialized
46  std::vector<weights *> weights_list(m_weights.begin(), m_weights.end());
47  if (weights_list.empty()) {
48  weights_list = m->get_weights();
49  }
50 
51  // Remove weights that are not being optimized
52  m_weights.clear();
53  for (weights *w : weights_list) {
54  optimizer *opt = w->get_optimizer();
55  if (opt != nullptr) {
56  m_weights.insert(w);
57  // Initialize the global learning rate, exactly once.
58  if (m_cur_global_lr == 0.0f) {
59  m_cur_global_lr = opt->get_learning_rate();
60  }
61  }
62  }
63 
64 }

66 void lbann_callback_learning_rate::on_epoch_end(model *m) {
67  const float new_lr = global_schedule(m);
68  const float old_global_lr = m_cur_global_lr;
69  m_cur_global_lr = new_lr;
70  lbann_comm *comm = m->get_comm();
71  if (comm->am_model_master() && new_lr != old_global_lr) {
72  std::cout << "Model " << comm->get_model_rank() << ": "
73  << "changing global learning rate to " << new_lr
74  << " at epoch " << m->get_cur_epoch() << std::endl;
75  }
76  for (weights *w : m_weights) {
77  optimizer *opt = w->get_optimizer();
78  const float old_lr = opt->get_learning_rate();
79  if (old_lr != new_lr) {
80  opt->set_learning_rate(new_lr);
81  }
82  }
83 }

85 void lbann_callback_learning_rate::on_backward_prop_end(model *m) {
86  for (weights *w : m_weights) {
87  optimizer& opt = *w->get_optimizer();
88  const float old_lr = opt.get_learning_rate();
89  const float new_lr = optimizer_schedule(m, opt);
90  if (old_lr != new_lr) {
91  opt.set_learning_rate(new_lr);
92  }
93  }

121 float lbann_callback_adaptive_learning_rate::global_schedule(model *m) {
122  // Determine behavior the first time this is called in an epoch
123  if (m_cur_epoch != m->get_cur_epoch()) {
124  m_cur_epoch = m->get_cur_epoch();
125  const execution_mode mode = m->get_execution_mode();
126  const EvalType score = m->get_objective_function()->get_mean_value(mode);
127  if (score < m_last_score) {
128  // Reset wait counter if score has decreased
129  m_last_score = score;
130  m_wait = 0;
131  m_adjust_learning_rate = false;
132  } else if (m_wait >= m_patience) {
133  // Adjust learning rate if patience has been exceeded
134  m_last_score = score;
135  m_wait = 0;
136  m_adjust_learning_rate = true;
137  } else {
138  // Otherwise increment wait counter
139  m_wait++;
140  m_adjust_learning_rate = false;
141  }
142  }
143 
144  // Adjust learning rate if needed
145  if (m_adjust_learning_rate) {
146  return m_cur_global_lr * m_amt;
147  } else {
148  return m_cur_global_lr;
149  }
150 }

164 float lbann_callback_drop_fixed_learning_rate::global_schedule(model* m) {
165  // Delete last drop epoch if we have already passed it
166  while (!m_drop_epochs.empty()
167  && m->get_cur_epoch() > m_drop_epochs.back()) {
168  m_drop_epochs.pop_back();
169  }
170 
171  // Adjust learning rate if at a drop epoch
172  if (!m_drop_epochs.empty() && m->get_cur_epoch() == m_drop_epochs.back()) {
173  return m_cur_global_lr * m_amt;
174  } else {
175  return m_cur_global_lr;
176  }
177 }

195 void lbann_callback_linear_growth_learning_rate::setup(model *m) {
196  lbann_callback_learning_rate::setup(m);
197  // Compute the learning rate increase.
198  if (!m_weights.empty()) {
199  // Assumes all optimizers have the same initial learning rate.
200  m_base_lr = m_cur_global_lr;
201  m_inc = (m_target - m_base_lr) / m_num_epochs;
202  }
203 }
204 
205 float lbann_callback_linear_growth_learning_rate::global_schedule(model *m) {
206  if (m->get_cur_epoch() < m_delay) {
207  return m_cur_global_lr;
208  } else if (m->get_cur_epoch() <= m_num_epochs + m_delay) {
209  int num_left = m_num_epochs + m_delay - m->get_cur_epoch();
210  return m_base_lr + m_inc*(m_num_epochs - num_left);
211  } else {
212  return m_cur_global_lr;
213  }
214 }

257 float lbann_callback_poly_learning_rate::optimizer_schedule(model *m, optimizer &opt) {
258  const uint64_t cur_iter = static_cast<uint64_t>(m->get_cur_step());
259  if (m_max_iter > cur_iter) {
260  m_lr = static_cast<float>(std::pow(static_cast<double>(m_max_iter - cur_iter)/m_max_iter, m_p));
261  }
262  const float scale = m_lr / m_last_epoch_lr;
263  return m_cur_global_lr * scale;
264 }

275 float lbann_callback_optimizerwise_adaptive_learning_rate::optimizer_schedule(
276  model *m, optimizer &opt) {
277  DataType param_norm = El::Nrm2(opt.get_weights().get_values());
278  DataType param_grad_norm = El::Nrm2(opt.get_gradient());
279  if (param_norm > DataType(0) && param_grad_norm > DataType(0)) {
280  // TODO: Should incorporate weight decay, etc. here.
281  return m_cur_global_lr * m_scale * param_norm / param_grad_norm;
282  } else {
283  return opt.get_learning_rate();
284  }
