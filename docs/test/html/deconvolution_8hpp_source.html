122  void setup_dims() override {
123 
124  // Initialize previous neuron tensor dimensions
125  base_convolution_layer::setup_dims();
126 
127  // Initialize deconvolution kernel dimensions
128  // Note that unlike the convolutional kernel, the previous layer's
129  // number of channels is now the leading position -- keep in mind
130  // that deconvolution is the transpose of a convolution
131  this->m_kernel_dims.insert(this->m_kernel_dims.begin(),
132  this->m_prev_neuron_dims[0]);
133 
134  // Check if previous neuron tensor dimensions are valid
135  #ifdef LBANN_DEBUG
136  if(this->m_num_neuron_dims != (int) this->m_kernel_dims.size() - 1) {
137  throw lbann_exception("deconvolution_layer: previous neuron tensor dimensions are unexpected");
138  }
139  #endif
140 
141  // Initialize neuron tensor dimensions
142  this->m_neuron_dims[0] = this->m_kernel_dims[1];
143  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
144  this->m_neuron_dims[i+1]
145  = ((this->m_prev_neuron_dims[i+1]-1) * this->m_strides[i]
146  + this->m_kernel_dims[i+2] - 2*this->m_pads[i]);
147  }
148  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),
149  this->m_neuron_dims.end(),
150  1,
151  std::multiplies<int>());
152 
153  // Get size of convolutional kernel
154  this->m_kernel_size = std::accumulate(m_kernel_dims.begin(),
155  m_kernel_dims.end(),
156  1,
157  std::multiplies<int>());
158 
159  }
