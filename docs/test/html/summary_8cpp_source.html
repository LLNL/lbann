52 void lbann_summary::reduce_mean(const std::string tag,
53  const AbsDistMat& mat,
54  int step) {
55  // Local sum
56  DataType sum = 0.0;
57 
58  // Check distributed matrix format
59  El::DistData mat_format(mat);
60  if(mat_format.colDist == El::STAR && mat_format.rowDist == El::STAR) {
61  // Compute local sum on master process if matrix is Star,Star
62  if(m_comm->am_model_master()) {
63  sum = local_sum(mat.LockedMatrix());
64  }
65  } else {
66  // Compute local sum on all processes if matrix is in MC,MR;
67  // Star,VC; or similar format
68  // TODO: implement for matrices in Circ,Circ; MC,Star; or similar
69  // formats
70  sum = local_sum(mat.LockedMatrix());
71  }
72 
73  // Add local sum to list of pending means
74  m_pending_means.emplace_back(tag, step, sum, 0.0f, mat.Height() * mat.Width());
75 }

131 void lbann_summary::reduce_histogram(const std::string tag,
132  const AbsDistMat& mat,
133  int step) {
134  DataType mat_local_min = local_min(mat.LockedMatrix());
135  DataType mat_local_max = local_max(mat.LockedMatrix());
136  // Local sum and squared sum
137  DataType sum = 0.0f;
138  DataType sqsum = 0.0f;
139  // Check distributed matrix format
140  El::DistData mat_format(mat);
141  if(mat_format.colDist == El::STAR && mat_format.rowDist == El::STAR) {
142  // Compute local sums on master process if matrix is Star,Star
143  if(m_comm->am_model_master()) {
144  local_sum_sqsum(mat.LockedMatrix(), sum, sqsum);
145  }
146  } else {
147  // Compute local sums on all processes if matrix is in MC,MR;
148  // Star,VC; or similar format
149  // TODO: implement for matrices in Circ,Circ; MC,Star; or similar
150  // formats
151  local_sum_sqsum(mat.LockedMatrix(), sum, sqsum);
152  }
153  // Compute local buckets.
154  std::vector<float> buckets(m_histogram_buckets.size(), 0.0f);
155  const int height = mat.LocalHeight();
156  const int width = mat.LocalWidth();
157  const int ldim = mat.LDim();
158  const DataType *__restrict__ mat_buf = mat.LockedMatrix().LockedBuffer();
159  for (int row = 0; row < height; ++row) {
160  for (int col = 0; col < width; ++col) {
161  // Note: This could be optimized; upper_bound takes O(logn) time.
162  int bucket = std::upper_bound(
163  m_histogram_buckets.begin(), m_histogram_buckets.end(),
164  mat_buf[row + col * ldim]) - m_histogram_buckets.begin();
165  buckets[bucket] += 1.0f;
166  }
167  }
168  // Add to list of pending histograms.
169  m_pending_histograms.emplace_back(
170  tag, step, buckets, mat_local_min, mat_local_max, mat.Height() * mat.Width(),
171  sum, sqsum);
172  // TODO: Support histograms on multiple models.
173 }

195 void lbann_summary::flush_means() {
196  if (m_pending_means.empty()) {
197  return;
198  }
199  std::vector<DataType> local_sums;
200  for (const auto& op : m_pending_means) {
201  local_sums.push_back(op.local);
202  }
203  if (m_comm->am_model_master()) {
204  std::vector<DataType> global_sums(local_sums.size());
205  m_comm->model_reduce(local_sums.data(), local_sums.size(),
206  global_sums.data());
207  // Compute the means in-place.
208  for (unsigned i = 0; i < global_sums.size(); ++i) {
209  global_sums[i] /= m_pending_means[i].num;
210  }
211  gather_scalar_summary(m_pending_means, global_sums);
212  } else {
213  m_comm->model_reduce(local_sums.data(), local_sums.size(),
214  m_comm->get_model_master());
215  }
216  m_pending_means.clear();
217 }

259 void lbann_summary::flush_stdevs() {
260  if (m_pending_stdevs.empty()) {
261  return;
262  }
263  std::vector<DataType> local_sums;
264  std::vector<DataType> local_sqsums;
265  for (const auto& op : m_pending_stdevs) {
266  local_sums.push_back(op.local);
267  local_sqsums.push_back(op.local2);
268  }
269  if (m_comm->am_model_master()) {
270  // Compute the model sample standard deviation as:
271  // sqrt[1/(n-1) (sqsum - (1/n)*sum^2)]
272  // The n-1 is to use an unbiased variance estimate.
273  // This unrolls the usual formulation of standard deviation some, to avoid
274  // global operations when pushing the operation.
275  std::vector<DataType> global_sums(local_sums.size());
276  std::vector<DataType> global_sqsums(local_sqsums.size());
277  m_comm->model_reduce(local_sums.data(), local_sums.size(),
278  global_sums.data());
279  m_comm->model_reduce(local_sqsums.data(), local_sqsums.size(),
280  global_sqsums.data());
281  // Re-use the global_sums vector for the standard deviation.
282  for (unsigned i = 0; i < global_sums.size(); ++i) {
283  global_sums[i] = std::sqrt(
284  (global_sqsums[i] -
285  global_sums[i] * global_sums[i] / m_pending_stdevs[i].num) /
286  (m_pending_stdevs[i].num - 1));
287  }
288  gather_scalar_summary(m_pending_stdevs, global_sums);
289  } else {
290  m_comm->model_reduce(local_sums.data(), local_sums.size(),
291  m_comm->get_model_master());
292  m_comm->model_reduce(local_sqsums.data(), local_sqsums.size(),
293  m_comm->get_model_master());
294  }
295  m_pending_stdevs.clear();
296 }

332 void lbann_summary::flush_histograms() {
333  if (m_pending_histograms.empty()) {
334  return;
335  }
336  std::vector<DataType> local_mins;
337  std::vector<DataType> local_maxes;
338  std::vector<DataType> local_sums;
339  std::vector<DataType> local_sqsums;
340  std::vector<float> buckets;
341  for (const auto& op : m_pending_histograms) {
342  local_mins.push_back(op.min);
343  local_maxes.push_back(op.max);
344  local_sums.push_back(op.sum);
345  local_sqsums.push_back(op.sqsum);
346  buckets.insert(buckets.end(), op.buckets.begin(), op.buckets.end());
347  }
348  if (m_comm->am_model_master()) {
349  std::vector<DataType> model_mins(local_mins.size());
350  std::vector<DataType> model_maxes(local_maxes.size());
351  std::vector<DataType> model_sums(local_sums.size());
352  std::vector<DataType> model_sqsums(local_sqsums.size());
353  std::vector<float> model_buckets(buckets.size());
354  m_comm->model_reduce(local_mins.data(), local_mins.size(),
355  model_mins.data(), El::mpi::MIN);
356  m_comm->model_reduce(local_maxes.data(), local_maxes.size(),
357  model_maxes.data(), El::mpi::MAX);
358  m_comm->model_reduce(local_sums.data(), model_sums.size(),
359  model_sums.data());
360  m_comm->model_reduce(local_sqsums.data(), local_sqsums.size(),
361  model_sqsums.data());
362  m_comm->model_reduce(buckets.data(), buckets.size(),
363  model_buckets.data());
364  // Gather to the world master for writing out.
365  if (m_comm->am_world_master()) {
366  std::vector<DataType> global_mins(
367  m_comm->get_num_models() * model_mins.size());
368  std::vector<DataType> global_maxes(
369  m_comm->get_num_models() * model_maxes.size());
370  std::vector<DataType> global_sums(
371  m_comm->get_num_models() * model_sums.size());
372  std::vector<DataType> global_sqsums(
373  m_comm->get_num_models() * model_sqsums.size());
374  std::vector<float> global_buckets(
375  m_comm->get_num_models() * model_buckets.size());
376  m_comm->intermodel_gather(model_mins.data(), model_mins.size(),
377  global_mins.data());
378  m_comm->intermodel_gather(model_maxes.data(), model_maxes.size(),
379  global_maxes.data());
380  m_comm->intermodel_gather(model_sums.data(), model_sums.size(),
381  global_sums.data());
382  m_comm->intermodel_gather(model_sqsums.data(), model_sqsums.size(),
383  global_sqsums.data());
384  m_comm->intermodel_gather(model_buckets.data(), model_buckets.size(),
385  global_buckets.data());
386  for (unsigned i = 0; i < global_mins.size(); ++i) {
387  int model = i / m_pending_histograms.size();
388  unsigned ops_pos = i % m_pending_histograms.size();
389  std::vector<float> tmp_buckets(
390  global_buckets.begin() + i*m_histogram_buckets.size(),
391  global_buckets.begin() + (i+1)*m_histogram_buckets.size());
392  m_sw->add_histogram(prepend_model(m_pending_histograms[ops_pos].tag, model),
393  tmp_buckets, global_mins[i], global_maxes[i],
394  m_pending_histograms[ops_pos].num,
395  global_sums[i], global_sqsums[i],
396  m_pending_histograms[ops_pos].step);
397  }
398  } else {
399  m_comm->intermodel_gather(model_mins.data(), model_mins.size(),
400  m_comm->get_intermodel_master());
401  m_comm->intermodel_gather(model_maxes.data(), model_maxes.size(),
402  m_comm->get_intermodel_master());
403  m_comm->intermodel_gather(model_sums.data(), model_sums.size(),
404  m_comm->get_intermodel_master());
405  m_comm->intermodel_gather(model_sqsums.data(), model_sqsums.size(),
406  m_comm->get_intermodel_master());
407  m_comm->intermodel_gather(model_buckets.data(), model_buckets.size(),
408  m_comm->get_intermodel_master());
409  }
410  } else {
411  m_comm->model_reduce(local_mins.data(), local_mins.size(),
412  m_comm->get_model_master(), El::mpi::MIN);
413  m_comm->model_reduce(local_maxes.data(), local_maxes.size(),
414  m_comm->get_model_master(), El::mpi::MAX);
415  m_comm->model_reduce(local_sums.data(), local_sums.size(),
416  m_comm->get_model_master());
417  m_comm->model_reduce(local_sqsums.data(), local_sqsums.size(),
418  m_comm->get_model_master());
419  m_comm->model_reduce(buckets.data(), buckets.size(),
420  m_comm->get_model_master());
421  }
422  m_pending_histograms.clear();
423 }

425 DataType lbann_summary::local_sum(const Mat& mat) const {
426  // Note there are more numerically stable ways to compute a sum.
427  const El::Int height = mat.Height();
428  const El::Int width = mat.Width();
429  const El::Int ldim = mat.LDim();
430  const DataType * __restrict__ mat_buf = mat.LockedBuffer();
431  auto sum = DataType(0);
432  if (ldim == height) {
433  const El::Int size = height*width;
434 #pragma omp parallel for reduction(+:sum)
435  for (El::Int i = 0; i < size; ++i) {
436  sum += mat_buf[i];
437  }
438  } else {
439 #pragma omp parallel for reduction(+:sum) collapse(2)
440  for (El::Int row = 0; row < height; ++row) {
441  for (El::Int col = 0; col < width; ++col) {
442  sum += mat_buf[row + col * ldim];
443  }
444  }
445  }
446  return sum;
447 }

449 void lbann_summary::local_sum_sqsum(
450  const Mat& mat, DataType& sum, DataType& sqsum) const {
451  // Note there are more numerically stable ways to compute a sum.
452  const El::Int height = mat.Height();
453  const El::Int width = mat.Width();
454  const El::Int ldim = mat.LDim();
455  const DataType * __restrict__ mat_buf = mat.LockedBuffer();
456  sum = DataType(0);
457  sqsum = DataType(0);
458  if (ldim == height) {
459  const El::Int size = height*width;
460 #pragma omp parallel for reduction(+:sum,sqsum)
461  for (El::Int i = 0; i < size; ++i) {
462  const DataType val = mat_buf[i];
463  sum += val;
464  sqsum += val*val;
465  }
466  } else {
467 #pragma omp parallel for reduction(+:sum,sqsum) collapse(2)
468  for (El::Int row = 0; row < height; ++row) {
469  for (El::Int col = 0; col < width; ++col) {
470  const DataType val = mat_buf[row + col*ldim];
471  sum += val;
472  sqsum += val * val;
473  }
474  }
475  }
476 }

478 DataType lbann_summary::local_min(const Mat& mat) const {
479  const El::Int height = mat.Height();
480  const El::Int width = mat.Width();
481  const El::Int ldim = mat.LDim();
482  const DataType * __restrict__ mat_buf = mat.LockedBuffer();
483  auto min = std::numeric_limits<DataType>::max();
484  if (ldim == height) {
485  const El::Int size = height*width;
486 #pragma omp parallel for reduction(min:min)
487  for (El::Int i = 0; i < size; ++i) {
488  min = std::min(min, mat_buf[i]);
489  }
490  } else {
491 #pragma omp parallel for reduction(min:min) collapse(2)
492  for (El::Int row = 0; row < height; ++row) {
493  for (El::Int col = 0; col < width; ++col) {
494  min = std::min(min, mat_buf[row + col*ldim]);
495  }
496  }
497  }
498  return min;
499 }

524 DataType lbann_summary::local_2norm(const Mat& mat) const {
525  // Note there are more numerically stable ways to compute this.
526  const El::Int height = mat.Height();
527  const El::Int width = mat.Width();
528  const El::Int ldim = mat.LDim();
529  const DataType * __restrict__ mat_buf = mat.LockedBuffer();
530  auto norm = DataType(0);
531  if (ldim == height) {
532  const El::Int size = height*width;
533 #pragma omp parallel for reduction(+:norm)
534  for (El::Int i = 0; i < size; ++i) {
535  norm += mat_buf[i] * mat_buf[i];
536  }
537  } else {
538 #pragma omp parallel for reduction(+:norm) collapse(2)
539  for (El::Int row = 0; row < height; ++row) {
540  for (El::Int col = 0; col < width; ++col) {
541  norm += mat_buf[row + col * ldim] * mat_buf[row + col * ldim];
542  }
543  }
544  }
545  return std::sqrt(norm);
546 }

553 void lbann_summary::gather_scalar_summary(
554  const std::vector<pending_op>& ops, std::vector<DataType>& scalars) {
555  if (m_comm->am_world_master()) {
556  std::vector<DataType> data(m_comm->get_num_models() * scalars.size());
557  m_comm->intermodel_gather(scalars.data(), scalars.size(), data.data());
558  for (unsigned i = 0; i < data.size(); ++i) {
559  int model = i / ops.size();
560  unsigned ops_pos = i % ops.size();
561  m_sw->add_scalar(prepend_model(ops[ops_pos].tag, model),
562  data[i], ops[ops_pos].step);
