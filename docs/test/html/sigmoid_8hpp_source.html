104  DataType activation(DataType x) const override {
105  #ifdef LBANN_ENABLE_SIGMOID_CUTOFF
106  // Ensure -m_cutoff <= x <= m_cutoff
107  x = std::min(std::max(x, -m_cutoff), m_cutoff);
108  #endif // LBANN_ENABLE_SIGMOID_CUTOFF
109  return 1 / (DataType(1) + std::exp(-x));
110  }
111 
112  DataType activation_derivative(DataType x) const override {
113  #ifdef LBANN_ENABLE_SIGMOID_CUTOFF
114  if (x < -m_cutoff || x > m_cutoff) { return DataType(0); }
115  #endif // LBANN_ENABLE_SIGMOID_CUTOFF
116  const auto sigx = activation(x);
117  return sigx * (DataType(1) - sigx);
118  }
