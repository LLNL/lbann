38 std::vector<int> get_tokens(std::string str, const std::vector<char> delims) {
39  std::vector<int> tokens;
40  size_t pos;
41 
42  for (const auto d : delims) {
43  pos = str.find_first_of(d);
44  if (pos == std::string::npos) {
45  // std::cerr << "Not able to split " << str << " by " << d << std::endl;
46  return std::vector<int>();
47  }
48  tokens.push_back(atoi(str.substr(0, pos).c_str()));
49  str = str.substr(pos+1, str.size());
50  }
51 
52  return tokens;
53 }

55 std::vector<std::string> get_tokens(const std::string str, const std::string delims) {
56  std::vector<std::string> parsed;
57  size_t pos_start = 0u;
58  size_t pos_end = 0u;
59 
60  while ((pos_end != std::string::npos) && (pos_start != std::string::npos)) {
61  pos_start = str.find_first_not_of(delims, pos_end);
62  if (pos_start != std::string::npos) {
63  pos_end = str.find_first_of(delims, pos_start);
64  parsed.push_back(str.substr(pos_start, (pos_end-pos_start)));
65  }
66  }
67  return parsed;
68 }
