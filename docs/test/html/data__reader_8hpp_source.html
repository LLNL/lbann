536  persist_value= persist_type::train;
537  }
538 
539  snprintf(fieldname, sizeof(fieldname), "%s__mini_batch_size", name);
540  p.write_uint64(persist_value, fieldname, (uint64_t) m_mini_batch_size);
541 
542  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
543  p.write_uint64(persist_value, fieldname, (uint64_t) m_current_mini_batch_idx);
544 
545  int size = m_shuffled_indices.size();
546  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
547  p.write_uint64(persist_value, fieldname, (uint64_t) size);
548 
549  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
550  p.write_uint64(persist_value, fieldname, (uint64_t) m_current_pos);
551 
552  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
