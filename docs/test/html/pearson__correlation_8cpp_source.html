103 void pearson_correlation_metric::setup(model& m) {
104  metric::setup(m);
105  const El::DistData dist_data(get_target_layer().get_prediction());
106  if (dist_data.colDist == El::MC
107  && dist_data.rowDist == El::MR) {
108  m_prediction_means = new StarMRMat(*dist_data.grid, dist_data.root);
109  m_prediction_stdevs = new StarMRMat(*dist_data.grid, dist_data.root);
110  m_ground_truth_means = new StarMRMat(*dist_data.grid, dist_data.root);
111  m_ground_truth_stdevs = new StarMRMat(*dist_data.grid, dist_data.root);
112  m_covariances = new StarMRMat(*dist_data.grid, dist_data.root);
113  } else if (dist_data.colDist == El::STAR
114  && dist_data.rowDist == El::VC) {
115  m_prediction_means = new StarVCMat(*dist_data.grid, dist_data.root);
116  m_prediction_stdevs = new StarVCMat(*dist_data.grid, dist_data.root);
117  m_ground_truth_means = new StarVCMat(*dist_data.grid, dist_data.root);
118  m_ground_truth_stdevs = new StarVCMat(*dist_data.grid, dist_data.root);
119  m_covariances = new StarVCMat(*dist_data.grid, dist_data.root);
120  } else {
121  std::stringstream err;
122  err << __FILE__ << " " << __LINE__ << " :: "
123  << "prediction matrix has invalid distribution "
124  << "(colDist=" << dist_data.colDist << ","
125  << "rowDist=" << dist_data.rowDist << ")";
126  throw lbann_exception(err.str());
127  }
128  m_prediction_means->Resize(1, m.get_max_mini_batch_size());
129  m_prediction_stdevs->Resize(1, m.get_max_mini_batch_size());
130  m_ground_truth_means->Resize(1, m.get_max_mini_batch_size());
131  m_ground_truth_stdevs->Resize(1, m.get_max_mini_batch_size());
132  m_covariances->Resize(1, m.get_max_mini_batch_size());
133 }

135 EvalType pearson_correlation_metric::evaluate_compute(const AbsDistMat& prediction,
136  const AbsDistMat& ground_truth) {
137 
138  // Compute means, standard deviations, and covariances
139  columnwise_mean_and_stdev(prediction, *m_prediction_means, *m_prediction_stdevs);
140  columnwise_mean_and_stdev(ground_truth, *m_ground_truth_means, *m_ground_truth_stdevs);
141  columnwise_covariance(prediction, ground_truth,
142  *m_prediction_means, *m_ground_truth_means,
143  *m_covariances);
144 
145  // Compute Pearson correlation of each column
146  // Note: Pearson(x,y) = cov(x,y) / ( stdev(x) * stdev(y) )
147  EvalType local_sum = 0.0;
148  for (int col = 0; col < m_covariances->LocalWidth(); ++col) {
149  const EvalType pred_stdev = m_prediction_stdevs->GetLocal(0, col);
150  const EvalType true_stdev = m_ground_truth_stdevs->GetLocal(0, col);
151  const EvalType cov = m_covariances->GetLocal(0, col);
155  if(true_stdev == 0
156  && m_ground_truth_stdevs->GlobalCol(col) == (m_ground_truth_stdevs->Width()-1)) {
157  continue;
158  }
159  local_sum += cov / (pred_stdev * true_stdev);
160  }
161  return get_comm().allreduce(local_sum, m_covariances->DistComm());
162 
163 }
