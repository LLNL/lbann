31 void mean_squared_error_loss::start_evaluate_compute(
32  const AbsDistMat& predictions,
33  const AbsDistMat& ground_truth) {
34  // Local matrices
35  const Mat& predictions_local = predictions.LockedMatrix();
36  const Mat& ground_truth_local = ground_truth.LockedMatrix();
37 
38  // Matrix parameters
39  const El::Int height = predictions.Height();
40  const El::Int width = predictions.Width();
41  const El::Int local_height = predictions_local.Height();
42  const El::Int local_width = predictions_local.Width();
43 
44  // Compute sum of squared errors
45  EvalType sum = EvalType(0);
46  #pragma omp parallel for reduction(+:sum) collapse(2)
47  for(El::Int col = 0; col < local_width; ++col) {
48  for(El::Int row = 0; row < local_height; ++row) {
49  const EvalType true_val = ground_truth_local(row, col);
50  const EvalType pred_val = predictions_local(row, col);
51  const EvalType error = true_val - pred_val;
52  sum += error * error;
53  }
54  }
55  m_sum = sum / (height * width); // Can't reduce on class members.
56  get_comm().nb_allreduce(&m_sum, 1, predictions.DistComm(), m_allreduce_req);
57 }

59 EvalType mean_squared_error_loss::finish_evaluate_compute(
60  const AbsDistMat& predictions,
61  const AbsDistMat& ground_truth) {
62  get_comm().wait(m_allreduce_req);
63  return m_sum;
64 }

66 void mean_squared_error_loss::differentiate_compute(const AbsDistMat& predictions,

70  // Local matrices

72  const Mat& ground_truth_local = ground_truth.LockedMatrix();
73  Mat& gradient_local = gradient.Matrix();
74 
75  // Matrix parameters
76  const int height = gradient.Height();
77  const El::Int local_height = gradient_local.Height();
78  const El::Int local_width = gradient_local.Width();
79 
80  // Compute gradient
81  const DataType scale = DataType(2) / height;
82  #pragma omp parallel for collapse(2)
83  for (El::Int col = 0; col < local_width; ++col) {
84  for (El::Int row = 0; row < local_height; ++row) {
85  const DataType true_val = ground_truth_local(row, col);
86  const DataType pred_val = predictions_local(row, col);
87  gradient_local(row, col) = (pred_val - true_val) * scale;
88  }
89  }
