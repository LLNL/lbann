36 model* instantiate_model(lbann_comm* comm,
37  objective_function* obj,
38  const lbann_data::Optimizer& proto_opt,
39  const lbann_data::Model& proto_model) {
40  std::stringstream err;
41 
42  // Default optimizer
43  auto&& opt = construct_optimizer(comm, proto_opt);
44 
45  // Construct model
46  const auto& type = proto_model.name();
47  const auto& mini_batch_size = proto_model.mini_batch_size();
48  if (type == "sequential_model" || type == "") {
49  return new sequential_model(comm, mini_batch_size, obj, opt);
50  }
51  if (type == "directed_acyclic_graph_model") {
52  return new directed_acyclic_graph_model(comm, mini_batch_size, obj, opt);
53  }
54  if (type == "recurrent_model") {
55  const auto& params = proto_model.recurrent();
56  return new recurrent_model(comm,
57  mini_batch_size,
58  obj,
59  opt,
60  params.unroll_depth());
61  }
62  if (type == "siamese_model") {
63  const auto& params = proto_model.siamese();
64  return new siamese_model(comm,
65  mini_batch_size,
66  obj,
67  opt,
68  params.num_heads());
69  }
70 
71  // Throw error if model type is not supported
72  err << "unknown model type (" << type << ")";
73  LBANN_ERROR(err.str());
74  return nullptr;
75 
76 }

78 void assign_layers_to_objective_function(std::vector<Layer*>& layer_list,
79  objective_function* obj,
80  const lbann_data::ObjectiveFunction& proto_obj) {
81  std::stringstream err;
82 
83  // Construct map from layer names to layers
84  std::unordered_map<std::string, Layer*> names_to_layers;
85  for (auto&& l : layer_list) {
86  const auto& name = l->get_name();
87  if (names_to_layers.count(name) > 0) {
88  err << "layer name \"" << name << "\" is not unique";
89  LBANN_ERROR(err.str());
90  }
91  names_to_layers[name] = l;
92  }
93 
94  // Assign evaluation layers to layer terms in objective function
95  auto&& obj_terms = obj->get_terms();
96  int num_layer_terms = 0;
97  for (size_t i = 0; i < obj_terms.size(); ++i) {
98  auto&& term = dynamic_cast<layer_term*>(obj_terms[i]);
99  if (term != nullptr) {
100  ++num_layer_terms;
101  if (num_layer_terms > proto_obj.layer_term_size()) { continue; }
102  const auto& params = proto_obj.layer_term(num_layer_terms-1);
103  auto&& eval = names_to_layers[params.layer()];
104  term->set_evaluation_layer(eval);
105  }
106  }
107 
108  // Check that layer terms in objective function match prototext
109  if (num_layer_terms != proto_obj.layer_term_size()) {
110  err << "number of layer terms in objective function does not match prototext "
111  << "(expected " << proto_obj.layer_term_size() << ", "
112  << "found " << num_layer_terms << ")";
113  LBANN_ERROR(err.str());
114  }
115 
116 }

119 void assign_weights_to_layers(std::vector<Layer*>& layer_list,
120  std::vector<weights*>& weights_list,
121  const lbann_data::Model& proto_model) {
122  std::stringstream err;
123 
124  // Construct map from weights names to weights
125  std::unordered_map<std::string, weights*> names_to_weights;
126  for (auto&& w : weights_list) {
127  const auto& name = w->get_name();
128  if (names_to_weights.count(name) > 0) {
129  err << "weights name \"" << name << "\" is not unique";
130  LBANN_ERROR(err.str());
131  }
132  names_to_weights[name] = w;
133  }
134 
135  // Find weights assigned to each layer
136  for (int i=0; i<proto_model.layer_size(); ++i) {
137  const auto& proto_layer = proto_model.layer(i);
138  auto& layer_weights = layer_list[i]->get_weights();
139  for (auto&& name : parse_list<std::string>(proto_layer.weights())) {
140  auto&& w = names_to_weights[name];
141  if (w == nullptr) {
142  err << "could not find weights named \"" << name << "\", "
143  << "which are expected by layer " << layer_list[i]->get_name();
144  LBANN_ERROR(err.str());
145  }
146  layer_weights.push_back(w);
147  }
148  }

154 model* construct_model(lbann_comm* comm,
155  cudnn::cudnn_manager* cudnn,
156  std::map<execution_mode, generic_data_reader*>& data_readers,
157  const lbann_data::Optimizer& proto_opt,
158  const lbann_data::Model& proto_model) {
159 
160  // Add layer graph
161  auto&& layer_list = construct_layer_graph(comm,
162  data_readers,
163  cudnn,
164  proto_model);
165 
166  // Construct objective function
167  const auto& proto_obj = proto_model.objective_function();
168  auto&& obj = construct_objective_function(proto_obj);
169  assign_layers_to_objective_function(layer_list, obj, proto_obj);
170 
171  // Instantiate model
172  auto&& m = instantiate_model(comm, obj, proto_opt, proto_model);
173  for (auto&& l : layer_list) { m->add_layer(l); }
174 
175  // Add weights and assign to layers
176  for (int i=0; i<proto_model.weights_size(); i++) {
177  m->add_weights(construct_weights(comm,
178  cudnn,
179  proto_opt,
180  proto_model.weights(i)));
181  }
182  auto weights_list = m->get_weights();
183  assign_weights_to_layers(layer_list, weights_list, proto_model);
184 
185  // Add metrics
186  for (int i=0; i<proto_model.metric_size(); ++i) {
187  m->add_metric(construct_metric(comm, proto_model.metric(i)));
188  }
189 
190  // Add callbacks
191  auto&& summarizer = construct_summarizer(comm, proto_model);
192  for (int i=0; i<proto_model.callback_size(); i++) {
193  m->add_callback(construct_callback(comm,
194  proto_model.callback(i),
195  data_readers,
196  layer_list,
197  weights_list,
198  summarizer));
199  }
200 
201  return m;
202 
203 }
