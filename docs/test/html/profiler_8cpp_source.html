59 static void prof_region_begin(const char *s, int c) {
60  synchronize_all_devices();
61  // Doesn't work with gcc 4.9
62  // nvtxEventAttributes_t ev = {0};
63  nvtxEventAttributes_t ev;
64  memset(&ev, 0, sizeof(nvtxEventAttributes_t));
65  ev.version = NVTX_VERSION;
66  ev.size = NVTX_EVENT_ATTRIB_STRUCT_SIZE;
67  ev.colorType = NVTX_COLOR_ARGB;
68  ev.color = c;
69  ev.messageType = NVTX_MESSAGE_TYPE_ASCII;
70  ev.message.ascii = s;
71  nvtxRangePushEx(&ev);
72 }
73 static void prof_region_end(const char *s) {
74  synchronize_all_devices();

76 }

121 int lbann_callback_profiler::get_color(Layer *l) {
122  const std::string &lname = l->get_type();
123  int idx = 4;
124  if (lname == "fully_connected") {
125  idx = 5;
126  } else if (lname == "convolution") {
127  idx = 6;
128  } else if (lname == "pooling_layer") {
129  idx = 7;
130  } else if (lname == "input_layer_distributed_minibatch_parallel_io") {
131  idx = 8;
132  }
133  return colors[idx % num_colors];
134 }
