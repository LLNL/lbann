189  void setup_dims() override {
190 
191  // Initialize previous neuron tensor dimensions
192  transform_layer::setup_dims();
193 
194  // Initialize neuron tensor dimensions
195  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
196  const int effective_dim = (this->m_prev_neuron_dims[i+1]
197  + 2 * m_pads[i] - m_pool_dims[i] + 1);
198  this->m_neuron_dims[i+1] = ((effective_dim + m_strides[i] - 1)

260  void fp_compute_cudnn() {
261  #ifndef LBANN_HAS_CUDNN
262  throw lbann_exception("pooling_layer: cuDNN not detected");
263  #else
264 
265  // Useful constants
266  const DataType one = 1;
267  const DataType zero = 0;
268 
269  // Perform pooling with each GPU
270  const int num_gpus = this->m_cudnn->get_num_gpus();
271  for(int i=0; i<num_gpus; ++i) {
272  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
273  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
274  this->m_cudnn->get_stream(i)));
275  CHECK_CUDNN(cudnnPoolingForward(this->m_cudnn->get_handle(i),
276  m_pooling_cudnn_desc,
277  &one,
278  this->m_prev_activations_cudnn_desc,
279  this->m_prev_activations_d[0].get_locked_data(i),
280  &zero,
281  this->m_activations_cudnn_desc,

289  void bp_compute_cudnn() {
290  #ifndef LBANN_HAS_CUDNN
291  throw lbann_exception("pooling_layer: cuDNN not detected");
292  #else
293 
294  // Useful constants
295  const DataType one = DataType(1);
296 
297  // Get number of GPUs
298  const int num_gpus = this->m_cudnn->get_num_gpus();
299 
300  // Perform back propagation on each GPU
301  for(int i=0; i<num_gpus; ++i) {

323  void fp_compute_im2col() {
324 
325  // Throw exception if pooling mode is not max or average pooling
326  if(m_pool_mode != pool_mode::max
327  && m_pool_mode != pool_mode::average) {
328  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
329  }
330 
331  // Local matrices
332  const auto& local_input = get_local_prev_activations();
333  auto& local_output = get_local_activations();
334 
335  // Pool parameters
336  const int local_width = local_input.Width();
337  const int num_channels = this->m_prev_neuron_dims[0];
338  const int num_per_output_channel = this->m_num_neurons / num_channels;
339 
340  // Initialize max pool indices if needed
341  if(m_pool_mode == pool_mode::max) {
342  m_max_pool_indices.assign(this->m_num_neurons * local_width, 0);
343  }
344 
345  // Initialize matrices
346  Mat im2col_mat(m_pool_size * num_channels, num_per_output_channel);
347  Mat input_mat;
348 
349  // Iterate through data samples
350  for(int sample = 0; sample < local_width; ++sample) {
351 
352  // Construct im2col matrix from input
353  El::LockedView(input_mat, local_input,
354  El::ALL, El::IR(sample));
355  im2col(input_mat,
356  im2col_mat,
357  num_channels,
358  this->m_num_prev_neuron_dims - 1,
359  &this->m_prev_neuron_dims[1],
360  m_pads.data(),
361  m_pool_dims.data(),
362  m_strides.data());
363 
364  if(m_pool_mode == pool_mode::max) {
365  // Apply max pooling
366  DataType *output_buffer = local_output.Buffer(0, sample);
367  int *indices_buffer = &m_max_pool_indices[sample * this->m_num_neurons];
368  #pragma omp parallel for
369  for(int channel = 0; channel < num_channels; ++channel) {
370  for(int j = 0; j < num_per_output_channel; ++j) {
371  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
372  DataType max_entry = im2col_buffer[0];
373  int max_index = 0;
374  for(int i = 1; i < m_pool_size; ++i) {
375  const DataType current_entry = im2col_buffer[i];
376  if(current_entry > max_entry) {
377  max_entry = current_entry;
378  max_index = i;
379  }
380  }
381  const int output_index = j + channel * num_per_output_channel;
382  output_buffer[output_index] = max_entry;
383  indices_buffer[output_index] = max_index;
384  }
385  }
386  }
387 
388  if(m_pool_mode == pool_mode::average) {
389  // Apply average pooling
390  DataType *output_buffer = local_output.Buffer(0, sample);
391  #pragma omp parallel for
392  for(int channel = 0; channel < num_channels; ++channel) {
393  for(int j = 0; j < num_per_output_channel; ++j) {
394  const DataType *im2col_buffer
395  = im2col_mat.LockedBuffer(channel*m_pool_size, j);
396  DataType output_entry = 0;
397  for(int i = 0; i < m_pool_size; ++i) {
398  output_entry += im2col_buffer[i];
399  }
400  output_entry /= m_pool_size;
401  const int output_index = j + channel * num_per_output_channel;
402  output_buffer[output_index] = output_entry;
403  }
404  }
405  }
406 
407  }
408 
409  }

412  void bp_compute_im2col() {
413 
414  // Throw exception if pooling mode is not max or average pooling
415  if(m_pool_mode != pool_mode::max
416  && m_pool_mode != pool_mode::average) {
417  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
418  }
419 
420  // Local matrices
421  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
422  auto& local_gradient_wrt_input = get_local_error_signals();
423 
424  // Pool parameters
425  const int input_size = local_gradient_wrt_input.Height();
426  const int local_width = local_gradient_wrt_output.Width();
427  const int num_channels = this->m_prev_neuron_dims[0];
428  const int num_per_input_channel = this->m_num_neurons / num_channels;
429 
430  // Initialize matrices
431  Mat im2col_mat(m_pool_size * num_channels, num_per_input_channel);
432  Mat gradient_wrt_input_col(input_size, 1);
433 
434  // Iterate through data samples
435  for(int sample = 0; sample < local_width; ++sample) {
436 
437  // Compute gradient w.r.t. im2col matrix for max pooling
438  if(m_pool_mode == pool_mode::max) {
439 
440  // Clear im2col matrix
441  El::Zero(im2col_mat);
442 
443  // Copy previous error signal to im2col matrix entries
444  // corresponding to max
445  const DataType *gradient_wrt_output_buffer
446  = local_gradient_wrt_output.LockedBuffer(0, sample);
447  const int *indices_buffer
448  = &m_max_pool_indices[sample * this->m_num_neurons];
449  #pragma omp parallel for
450  for(int channel = 0; channel < num_channels; ++channel) {
451  for(int j = 0; j < num_per_input_channel; ++j) {
452  const int input_index = j + channel * num_per_input_channel;
453  const int max_index = indices_buffer[input_index];
454  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
455  im2col_buffer[max_index]
456  = gradient_wrt_output_buffer[input_index];
457  }
458  }
459 
460  }
461 
462  // Compute gradient w.r.t. im2col matrix for average pooling
463  if(m_pool_mode == pool_mode::average) {
464  const DataType *gradient_wrt_output_buffer
465  = local_gradient_wrt_output.LockedBuffer(0, sample);
466  #pragma omp parallel for
467  for(int channel = 0; channel < num_channels; ++channel) {
468  for(int j = 0; j < num_per_input_channel; ++j) {
469  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
470  const int input_index = j + channel * num_per_input_channel;
471  const DataType output_entry
472  = gradient_wrt_output_buffer[input_index] / m_pool_size;
473  for(int i = 0; i < m_pool_size; ++i) {
474  im2col_buffer[i] = output_entry;
475  }
476  }
477  }
478 
479  }
480 
481  // Compute error signal (i.e. gradient w.r.t. input)
482  col2im(im2col_mat,
483  gradient_wrt_input_col,
484  num_channels,
485  this->m_num_prev_neuron_dims - 1,
486  &this->m_prev_neuron_dims[1],
487  m_pads.data(),
488  m_pool_dims.data(),
489  m_strides.data());
490  local_gradient_wrt_input(El::ALL, El::IR(sample)) += gradient_wrt_input_col;
491 
492  }
493 
494  }
495
