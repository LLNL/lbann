215 std::string Layer::get_topo_description() const {
216  std::stringstream ss;
217  const size_t num_children = get_num_children();
218  for (size_t i = 0; i < num_children; ++i) {
219  const auto& dims = get_neuron_dims(i);
220  if (i > 0) { ss << ", "; }
221  ss << "activations";
222  if (num_children > 1) { ss << "[" << i << "]"; }
223  ss << " = [";
224  switch (dims.size()) {
225  case 0:
226  ss << "0"; break;
227  case 2:
228  ss << dims[0] << "c x "
229  << dims[1] << "w";
230  break;
231  case 3:
232  ss << dims[0] << "c x "
233  << dims[1] << "w x "
234  << dims[2] << "h";
235  break;
236  default:
237  ss << dims[0];
238  for (size_t j = 1; j < dims.size(); ++j) {
239  ss << " x " << dims[j];
240  }
241  }
242  ss << ", " << get_activations(i).Width() << "s]";
243  }
244  return ss.str();

247 void Layer::forward_prop() {
248  const auto fp_start = get_time();
249 
250  // Setup matrix data, e.g. input matrices
251  fp_setup_data(m_model->get_current_mini_batch_size());
252 
253  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
254  // Synchronize GPUs and check for errors
255  if (m_using_gpus) { this->m_cudnn->check_error(); }
256  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
257 
258  // Apply layer's compute function
259  const auto fp_compute_start = get_time();
260  fp_compute();
261  m_fp_compute_time += get_time() - fp_compute_start;
262 
263  // Add this layer as a gradient source for weight optimizers
264  for (auto&& w : m_weights) {
265  optimizer* opt = w->get_optimizer();
266  if (opt != nullptr) { opt->add_gradient_source(this); }
267  }
268 
269  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
270  // Synchronize GPUs and check for errors
271  if (m_using_gpus) { this->m_cudnn->check_error(); }
272  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
273 
274  m_fp_time += get_time() - fp_start;
275 }

277 void Layer::back_prop() {
278  const auto bp_start = get_time();
279 
280  // Setup matrix data, e.g. input matrices
281  bp_setup_data(m_model->get_current_mini_batch_size());
282 
283  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
284  // Synchronize GPUs and check for errors
285  if (m_using_gpus) { this->m_cudnn->check_error(); }
286  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
287 
288  // Backprop the compute function.
289  const auto bp_compute_start = get_time();
290  bp_compute();
291  m_bp_compute_time += get_time() - bp_compute_start;
292 
293  // Remove this layer as a gradient source for weight optimizers
294  for (auto&& w : m_weights) {
295  auto&& opt = w->get_optimizer();
296  if (opt != nullptr) { opt->remove_gradient_source(this); }
297  }
298 
299  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
300  // Synchronize GPUs and check for errors
301  if (m_using_gpus) { this->m_cudnn->check_error(); }
302  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
303 
304  m_bp_time += get_time() - bp_start;
305 }

324 void Layer::summarize_stats(lbann_summary& summarizer, int step) {
325  std::string prefix = m_name + "/";
326  summarizer.reduce_scalar(prefix + "fp_time", m_fp_time, step);
327  summarizer.reduce_scalar(prefix + "bp_time", m_bp_time, step);
328  summarizer.reduce_scalar(prefix + "update_time", m_update_time, step);
329  reset_counters();
330  // Combine the optimizer step time from all the weights.
331  double step_time = 0.0;
332  for (weights *w : get_weights()) {
333  optimizer *opt = w->get_optimizer();
334  if (opt) {
335  step_time += opt->get_step_time();
336  opt->reset_counters();
337  }
338  }
339  summarizer.reduce_scalar(prefix + "opt_time", step_time, step);
340 }

342 void Layer::summarize_matrices(lbann_summary& summarizer, int step) {
343 
344  // Summarize activation matrices
345  const int num_children = get_num_children();
346  for (int i = 0; i < num_children; ++i) {
347  std::string prefix = m_name + "/activations";
348  if (num_children > 1) { prefix += std::to_string(i); }
349  summarizer.reduce_mean(prefix + "/mean", *m_activations[i], step);
350  summarizer.reduce_min(prefix + "/min", *m_activations[i], step);
351  summarizer.reduce_max(prefix + "/max", *m_activations[i], step);
352  summarizer.reduce_stdev(prefix + "/stdev", *m_activations[i], step);
353  summarizer.reduce_2norm(prefix + "/2norm2", *m_activations[i], step);
354  }
355 
356  // Summarize error signal matrices
357  const int num_parents = get_num_parents();
358  for (int i = 0; i < num_parents; ++i) {
359  std::string prefix = m_name + "/error_signals";
360  if (num_parents > 1) { prefix += std::to_string(i); }
361  summarizer.reduce_mean(prefix + "/mean", *m_error_signals[i], step);
362  summarizer.reduce_min(prefix + "/min", *m_error_signals[i], step);
363  summarizer.reduce_max(prefix + "/max", *m_error_signals[i], step);
364  summarizer.reduce_stdev(prefix + "/stdev", *m_error_signals[i], step);
365  summarizer.reduce_2norm(prefix + "/2norm2", *m_error_signals[i], step);
366  }

458 void Layer::clear_error_signals(int mini_batch_size) {
459  // Note: matrices are cleared (without deallocating memory) in case
460  // they are matrix views.
461  for (int i = 0; i < get_num_children(); ++i) {
462  get_prev_error_signals(i).Empty(false);
463  get_prev_error_signals(i).Resize(get_num_neurons(i), mini_batch_size);
464  }
465  for (int i = 0; i < get_num_parents(); ++i) {
466  get_error_signals(i).Empty(false);
467  get_error_signals(i).Resize(get_num_prev_neurons(i), mini_batch_size);
468  if (m_using_gpus) {
469 #ifdef LBANN_HAS_CUDNN
470  m_error_signals_d[i].zero();
471 #endif // LBANN_HAS_CUDNN
472  } else {
473  El::Zero(get_error_signals(i));
474  }
475  }

511 void Layer::setup_pointers() {
512 
513  // Check if the number of parents/children are valid
514  if(m_expected_num_parent_layers >= 0
515  && get_num_parents() != m_expected_num_parent_layers) {
516  std::stringstream err;
517  err << __FILE__ << " " << __LINE__ << " :: "
518  << "layer " << m_name << " has an invalid number of parent layers "
519  << "(expected " << m_expected_num_parent_layers << ", "
520  << "but found " << m_parent_layers.size() << ")";
521  throw lbann_exception(err.str());
522  }
523  if(m_expected_num_child_layers >= 0
524  && get_num_children() != m_expected_num_child_layers) {
525  std::stringstream err;
