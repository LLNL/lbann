101 void data_reader_jag::set_linearized_image_size() {
102  if (!m_image_loaded) {
103  m_linearized_image_size = 0u;
104  m_image_width = 0u;
105  m_image_height = 0u;
106  } else {
107  m_linearized_image_size
108  = std::accumulate(m_images.shape.begin()+1, m_images.shape.end(),
109  1u, std::multiplies<size_t>());
110  if (m_linearized_image_size != static_cast<size_t>(m_image_width*m_image_height)) {

112  m_image_height = 1;
113  m_image_width = static_cast<int>(m_linearized_image_size);

115  std::stringstream err;

191 void data_reader_jag::load() {
192  const std::string data_dir = add_delimiter(get_file_dir());
193  const std::string namestr = get_data_filename();
194  std::vector<std::string> file_names = get_tokens(namestr);
195  if (file_names.size() != 3u) {
196  throw lbann_exception(
197  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
198  " :: unexpected number of files " + std::to_string(file_names.size()));
199  }
200 
201  for(auto& str : file_names) {
202  str = data_dir + str;
203  }
204  load(file_names[0], file_names[1], file_names[2], m_first_n);
205 
206  size_t num_samples = get_num_samples();
207 
208  if (m_first_n > 0) {
209  num_samples = (static_cast<size_t>(m_first_n) <= num_samples)?
210  static_cast<size_t>(m_first_n) : num_samples;
211 
212  m_first_n = num_samples;
213  set_use_percent(1.0);
214  set_absolute_sample_count(0u);
215  }
216 
217  // reset indices
218  m_shuffled_indices.clear();
219 
220  m_shuffled_indices.resize(num_samples);
221  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
222 
223  select_subset_of_data();
224 }

227 void data_reader_jag::load(const std::string image_file,
228  const std::string scalar_file,
229  const std::string input_file,
230  const size_t first_n) {
231  if ((m_model_mode == Inverse || m_model_mode == AutoI) &&
232  !image_file.empty() && !check_if_file_exists(image_file)) {
233  throw lbann_exception("data_reader_jag: failed to load " + image_file);
234  }
235  if ((m_model_mode == AutoS) &&
236  !scalar_file.empty() && !check_if_file_exists(scalar_file)) {
237  throw lbann_exception("data_reader_jag: failed to load " + scalar_file);
238  }
239  if ((m_model_mode == Inverse) &&
240  !input_file.empty() && !check_if_file_exists(input_file)) {
241  throw lbann_exception("data_reader_jag: failed to load " + input_file);
242  }
243 
244 
245  m_num_samples = 0u;
246 
247  // read in only those that will be used
248  if ((m_model_mode == Inverse || m_model_mode == AutoI) && !image_file.empty()) {
249  m_images = cnpy::npy_load(image_file);
250  if (first_n > 0u) { // to use only first_n samples
251  cnpy_utils::shrink_to_fit(m_images, first_n);
252  }
253  m_image_loaded = true;
254  set_linearized_image_size();
255  }
256  if ((m_model_mode == AutoS) && !scalar_file.empty()) {
257  m_scalars = cnpy::npy_load(scalar_file);
258  if (first_n > 0u) { // to use only first_n samples
259  cnpy_utils::shrink_to_fit(m_scalars, first_n);
260  }
261  m_scalar_loaded = true;
262  set_linearized_scalar_size();
263  }
264  if ((m_model_mode == Inverse) && !input_file.empty()) {
265  m_inputs = cnpy::npy_load(input_file);
266  if (first_n > 0u) { // to use only first_n samples
267  cnpy_utils::shrink_to_fit(m_inputs, first_n);
268  }
269  m_input_loaded = true;
270  set_linearized_input_size();
271  }
272 
273  size_t num_samples = 0u;
274  bool ok = check_data(num_samples);
275 
276  if (!ok) {
277  get_description();
278  throw lbann_exception("data_reader_jag: loaded data format not consistent");
279  }
280  m_num_samples = num_samples;
281  if (m_image_loaded) {
282  m_img_min = get_image_min();
283  m_img_max = get_image_max();
284  if (m_img_min == m_img_max) {
285  throw lbann_exception("data_reader_jag: no_variation in data");
286  }
287  normalize_image();
288  }
289 }

292 bool data_reader_jag::check_data(size_t& num_samples) const {
293  bool ok = true;
294  num_samples = 0u;
295  if (ok && m_image_loaded) {
296  ok = (m_linearized_image_size > 0u) && (m_images.word_size == sizeof(data_t));
297  if (!ok) {
298  std::cerr << "m_images.shape.size() = " << m_images.shape.size() << std::endl
299  << "m_linearized_image_size = " << m_linearized_image_size << std::endl
300  << "m_images.word_size = " << m_images.word_size << std::endl
301  << "sizeof(data_t) = " << sizeof(data_t) << std::endl;
302  return false;
303  }
304  num_samples = m_images.shape[0];
305  }
306  if (ok && m_scalar_loaded) {
307  ok = (m_linearized_scalar_size > 0u) && (m_scalars.word_size == sizeof(scalar_t));
308  if (!ok) {
309  std::cerr << "m_scalars.shape.size() = " << m_scalars.shape.size() << std::endl
310  << "m_linearized_scalar_size = " << m_linearized_scalar_size << std::endl
311  << "m_scalars.word_size = " << m_scalars.word_size << std::endl
312  << "sizeof(scalar_t) = " << sizeof(scalar_t) << std::endl;
313  return false;
314  }
315  if (num_samples > 0u) {
316  ok = ok && (num_samples == m_scalars.shape[0]);
317  } else {
318  num_samples = m_scalars.shape[0];
319  }
320  }
321  if (ok && m_input_loaded) {
322  ok = (m_linearized_input_size > 0u) && (m_inputs.word_size == sizeof(input_t));
323  if (!ok) {
324  std::cerr << "m_inputs.shape.size() = " << m_inputs.shape.size() << std::endl
325  << "m_linearized_input_size = " << m_linearized_input_size << std::endl
326  << "m_inputs.word_size = " << m_inputs.word_size << std::endl
327  << "sizeof(input_t) = " << sizeof(input_t) << std::endl;
328  return false;
329  }
330  if (num_samples > 0u) {
331  ok = ok && (num_samples == m_inputs.shape[0]);
332  } else {
333  num_samples = m_inputs.shape[0];
334  }
335  }
336  if (!ok) {
337  num_samples = 0u;
338  } else {
339  switch (m_model_mode) {
340  case Inverse:
341  ok = m_image_loaded && m_input_loaded;
342  break;
343  case AutoI:
344  ok = m_image_loaded;
345  break;
346  case AutoS:
347  ok = m_scalar_loaded;
348  break;
349  default:
350  throw lbann_exception("data_reader_jag::check_data() : unknown mode");
351  }
352  }
353 
354  return ok;
355 }

377 void data_reader_jag::normalize_image() {
378  if (!m_image_loaded) {
379  return;
380  }
381  using depth_t = cv_image_type<data_t>;
382  const int type_code = depth_t::T(1u);
383 
384  if (m_image_normalization == 1) {
385  data_t* const ptr = get_image_ptr(0);
386  // Present the entire image data as a single image
387  // and normalize it once and for all
388  cv::Mat img(m_num_samples, m_linearized_image_size,
389  type_code, reinterpret_cast<void*>(ptr));
390  cv::normalize(img, img, 0.0, 1.0, cv::NORM_MINMAX);
391  } else if (m_image_normalization == 2) {
392  // normalize each image independently
393  for (size_t i=0u; i < m_num_samples; ++i) {
394  data_t* const ptr = get_image_ptr(i);
395  cv::Mat img(1, m_linearized_image_size,
396  type_code, reinterpret_cast<void*>(ptr));
397  cv::normalize(img, img, 0.0, 1.0, cv::NORM_MINMAX);
398  }
399  } else if (m_image_normalization != 0) {
400  throw lbann_exception("data_reader_jag: invalid normalization mode " +
401  std::to_string(m_image_normalization));
402  }
403 }

409 cv::Mat data_reader_jag::get_image(const size_t i) const {
410  using InputBuf_T = cv_image_type<data_t>;
411 
412  data_t* const ptr = get_image_ptr(i);
413  if (ptr == nullptr) {
414  return cv::Mat();
415  }
416  // Construct a zero copying view to data
417  const cv::Mat img_org(m_linearized_image_size, 1, InputBuf_T::T(1u),
418  reinterpret_cast<void*>(ptr));
419 
420  cv::Mat img;
421  if (std::is_same<data_t, DataType>::value) {
422  img = img_org.clone();
423  } else {
424  img_org.convertTo(img, cv_image_type<DataType>::T(1u));
425  }
426  return img.reshape(0, m_image_height);
427 }
