43 int assign_partners(lbann_comm* comm) {
44  if (comm->am_world_master()) {
45  // Generate partner assignments on master process
46  std::cout << "LTFB tournament partners:";
47  const auto& num_models = comm->get_num_models();
48  const auto& procs_per_model = comm->get_procs_per_model();
49  std::vector<int> models(num_models);
50  std::iota(models.begin(), models.end(), 0);
51  std::shuffle(models.begin(), models.end(), get_fast_generator());
52  std::vector<int> partners(num_models * procs_per_model);
53  for (int i = 0; i < num_models; i += 2) {
54  const auto& model1 = models[i];
55  const auto& model2 = (i+1 < num_models) ? models[i+1] : model1;
56  std::cout << (i > 0 ? "," : "") << " {" << model1;
57  if (model1 != model2) { std::cout << "," << model2; }
58  std::cout << "}";
59  std::fill_n(partners.begin() + model1 * procs_per_model,
60  procs_per_model, model2);
61  std::fill_n(partners.begin() + model2 * procs_per_model,
62  procs_per_model, model1);
63  }
64  std::cout << std::endl;
65  return comm->scatter(partners.data(), comm->get_world_comm());
66  } else {
67  return comm->scatter<int>(0, comm->get_world_comm());
68  }
69 }

75 void exchange_weights(lbann_comm* comm,
76  std::vector<weights*>& model_weights,
77  std::vector<weights*>& local_weights,
78  int partner) {
79  for (size_t i = 0; i < model_weights.size(); ++i) {
80  *local_weights[i] = *model_weights[i];
81  const auto& local_matrix = local_weights[i]->get_values();
82  auto&& remote_matrix = local_matrix.Copy();
83  const auto& size = local_matrix.LocalHeight() * local_matrix.LocalWidth();
84  if (size > 0) {
85  comm->sendrecv(local_matrix.LockedBuffer(), size, partner,
86  remote_matrix->Buffer(), size, partner);
87  model_weights[i]->set_values(*remote_matrix);
88  }
89  delete remote_matrix;
90  }
91 }

94 EvalType evaluate(model *m) {
95  const auto& mode = m->get_execution_mode();
96  m->evaluate(execution_mode::validation);
97  m->set_execution_mode(mode);
98  for (const auto& met : m->get_metrics()) {
99  if (dynamic_cast<categorical_accuracy_metric*>(met) != nullptr) {
100  return met->get_mean_value(execution_mode::validation);
101  }
102  }
103  return EvalType(0);
104 }

156 void lbann_callback_ltfb::on_batch_begin(model *m) {
157 
158  // Check whether to start LTFB round
159  const auto& mode = m->get_execution_mode();
160  const auto& step = m->get_cur_step();
161  if (mode != execution_mode::training) { return; }
162  if (step % m_round_size != 0 || step == 0) { return; }
163  if (m_comm->am_world_master()) {
164  std::cout << "---- LTFB round (step " << step << ") ----" << std::endl;
165  }
166 
167  // Determine partner model for tournament
168  const auto& local_model = m_comm->get_model_rank();
169  const auto& remote_model = assign_partners(m_comm);
170  if (remote_model == local_model) { return; }
171 
172  // Evaluate local model on tournament data
173  if (m_comm->am_world_master()) {
174  std::cout << "LTFB: evaluating local model..." << std::endl;
175  }
176  const auto& local_score = evaluate(m);
177 
178  // Evaluate remote model on tournament data
179  // Note: Weights from remote model are copied into local model
180  if (m_comm->am_world_master()) {
181  std::cout << "LTFB: evaluating remote model..." << std::endl;
182  }
183  auto model_weights = m->get_weights();
184  exchange_weights(m_comm, model_weights, m_local_weights, remote_model);
185  const auto& remote_score = evaluate(m);
186 
187  // Restore local weights if they achieve a better score
188  if (remote_score <= local_score) {
189  for (size_t i = 0; i < model_weights.size(); ++i) {
190  *model_weights[i] = *m_local_weights[i];
191  }
192  } else {
193  if (m_comm->am_model_master()) {
194  std::cout << "LTFB: replacing model " << local_model << " "
195  << "(" << local_score << "% categorical accuracy) "
196  << "with model " << remote_model << " "
197  << "(" << remote_score << "% categorical accuracy)"
198  << std::endl;
199  }
200  }
201 
202 }
