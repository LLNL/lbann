39  int num_dims,

51  void setup_dims() override {
52  // Store neuron tensor dimensions
53  const int num_neuron_dims = this->m_num_neuron_dims;
54  const std::vector<int> neuron_dims = this->m_neuron_dims;
55 
56  // Initialize previous neuron tensor dimensions
57  transform_layer::setup_dims();
58 
59  // Initialize neuron tensor dimensions
60  this->m_num_neuron_dims = num_neuron_dims;
61  this->m_neuron_dims = neuron_dims;
62 
63  // Determine any unspecified dimensions
64  int unspecified_dim = -1;
65  for(int dim = 0; dim < this->m_num_neuron_dims; ++dim) {
66  if(this->m_neuron_dims[dim] <= 0) {
67  unspecified_dim = dim;
68  this->m_neuron_dims[dim] = 1;
69  }
70  }
71  if(unspecified_dim >= 0) {
72  const int specified_size = std::accumulate(this->m_neuron_dims.begin(),
73  this->m_neuron_dims.end(),
74  1,
75  std::multiplies<int>());
76  this->m_neuron_dims[unspecified_dim] = this->m_num_neurons / specified_size;
77  }
78 
79  // Check that reshape is valid
80  if(this->m_num_neurons != std::accumulate(this->m_neuron_dims.begin(),
81  this->m_neuron_dims.end(),
82  1,
83  std::multiplies<int>())) {
84  std::string num_neurons = " {";
85  for(int n: m_neuron_dims) num_neurons += " " + std::to_string(n);
86  num_neurons += " }";
87  throw lbann_exception("reshape_layer: invalid neuron dimensions, " +
88  std::to_string(m_num_neurons) + " != " + num_neurons);
89  }
90 
91  }
