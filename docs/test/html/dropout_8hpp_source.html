182  void fp_compute_cpu() {
183 
184  // Matrices
185  const auto& input = get_prev_activations();
186  auto& output = get_activations();
187 
188  // Do nothing if dropout is disabled
189  const auto& mode = this->m_model->get_execution_mode();
190  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
191  El::LockedView(output, input);
192  return;
193  }
194 
195  // Construct mask matrix
196  const DataType scale = 1 / m_keep_prob;
197  const auto& height = input.Height();
198  const auto& width = input.Width();
199  m_mask->Resize(height, width);
200 #ifdef LBANN_SEQUENTIAL_CONSISTENCY
201  bernoulli_fill_procdet(*m_mask, height, width, DataType(m_keep_prob));
202  *m_mask *= scale;
203 #else
204  El::EntrywiseMap(*m_mask,
205  (std::function<DataType(const DataType&)>)
206  ([this,scale](const DataType& z)->DataType {
207  auto& gen = get_fast_generator();
208  std::bernoulli_distribution dist(m_keep_prob);
209  return dist(gen) ? scale : DataType(0);
210  }));
211 #endif // LBANN_SEQUENTIAL_CONSISTENCY
212 
213  // Apply mask matrix to get activations
214  El::Hadamard(input, *m_mask, output);
215 
216  }

219  void bp_compute_cpu() {
220  const auto& gradient_wrt_output = get_prev_error_signals();
221  auto& gradient_wrt_input = get_error_signals();
222  const auto& mode = this->m_model->get_execution_mode();
223  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
224  El::Axpy(DataType(1), gradient_wrt_output, gradient_wrt_input);
225  } else {
226  El::Hadamard(gradient_wrt_output, *m_mask, *m_mask);
227  El::Axpy(DataType(1), *m_mask, gradient_wrt_input);
228  }
229  }

231  void fp_compute_gpu() {
232  #ifndef LBANN_HAS_CUDNN
233  LBANN_ERROR("cuDNN not detected");
234  #else
235  const auto& input_d = this->m_prev_activations_d[0];
236  auto& output_d = this->m_activations_d[0];
237  const auto& num_gpus = this->m_cudnn->get_num_gpus();
238 
239  // Do nothing if dropout is disabled
240  const auto& mode = this->m_model->get_execution_mode();
241  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
242  output_d.locked_view(input_d);
243  return;
244  }
245 
246  // Apply dropout on each GPU
247  for (int i = 0; i < num_gpus; ++i) {
248  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
249  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
250  this->m_cudnn->get_stream(i)));
251  CHECK_CUDNN(cudnnDropoutForward(this->m_cudnn->get_handle(i),
252  m_dropout_cudnn_desc[i],
253  this->m_prev_activations_cudnn_desc,
254  input_d.get_locked_data(i),
255  this->m_activations_cudnn_desc,
256  output_d.get_data(i),
257  m_reserve_space_d.get_data(i),
258  m_reserve_space_d.get_height() * sizeof(DataType)));
259  }
260 
261  #endif // LBANN_HAS_CUDNN
262  }

264  void bp_compute_gpu() {
265  #ifndef LBANN_HAS_CUDNN
266  LBANN_ERROR("cuDNN not detected");
267  #else
268  const auto& gradient_wrt_output_d = this->m_prev_error_signals_d[0];
269  auto& gradient_wrt_input_d = this->m_error_signals_d[0];
270  const auto& num_gpus = this->m_cudnn->get_num_gpus();
271 
272  // Add to error signal if dropout is disabled
273  const auto& mode = this->m_model->get_execution_mode();
274  if (mode != execution_mode::training || m_keep_prob < EvalType(0)) {
275  for (int i=0; i<num_gpus; ++i) {
276  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
277  cublas::geam(this->m_cudnn->get_cublas_handle(i),
278  CUBLAS_OP_N, CUBLAS_OP_N,
279  gradient_wrt_input_d.get_height(),
280  this->m_mini_batch_size_per_gpu,
281  DataType(1),
282  gradient_wrt_output_d.get_locked_data(i),
283  gradient_wrt_output_d.get_leading_dim(),
284  DataType(1),
285  gradient_wrt_input_d.get_locked_data(i),
286  gradient_wrt_input_d.get_leading_dim(),
287  gradient_wrt_input_d.get_data(i),
288  gradient_wrt_input_d.get_leading_dim());
289  }
290  return;
291  }
292 
293  // Apply dropout backprop on each GPU
295  for (int i = 0; i < num_gpus; ++i) {
296  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
297  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
298  this->m_cudnn->get_stream(i)));
299  CHECK_CUDNN(cudnnDropoutBackward(this->m_cudnn->get_handle(i),
300  m_dropout_cudnn_desc[i],
301  this->m_prev_error_signals_cudnn_desc,
302  gradient_wrt_output_d.get_locked_data(i),
303  this->m_error_signals_cudnn_desc,
304  gradient_wrt_input_d.get_data(i),
305  m_reserve_space_d.get_data(i),
306  m_reserve_space_d.get_height() * sizeof(DataType)));
307  }
308 
309  #endif // LBANN_HAS_CUDNN
310  }
