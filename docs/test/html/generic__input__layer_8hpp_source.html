170  void fp_setup_data(int mini_batch_size) override {
171 
172  // Determine model mini-batch size and effective mini-batch size
173  // Note: If inter-model communication is activated, the effective
174  // mini-batch is equal to the global mini-batch size.
176  mini_batch_size = get_current_mini_batch_size();
177  int effective_mini_batch_size = mini_batch_size;
178  for (auto&& cb : this->m_model->get_callbacks()) {
179  if (dynamic_cast<lbann_callback_imcomm*>(cb) != nullptr) {
180  effective_mini_batch_size = get_current_global_mini_batch_size();
181  break;
182  }
183  }
184 
185  // Set mini-batch size in model
186  this->m_model->set_current_mini_batch_size(mini_batch_size);
187  this->m_model->set_effective_mini_batch_size(effective_mini_batch_size);
188 
189  // Initialize matrices
190  io_layer::fp_setup_data(mini_batch_size);
191 
192  // Once the current mini-batch size is defined, set the standard view for activations only
193  io_buffer->set_local_matrix_bypass(&get_local_activations());
194  io_buffer->set_std_matrix_view(mini_batch_size);
195 
196  }

198  void fp_compute() override {
199  execution_mode mode = this->m_model->get_execution_mode();
200  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(get_data_reader(), mode);
201 
202  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
203  // Use the predetermined size of the mini-batch to set the current
204  // batch size for the neural network
205  num_samples_in_batch = get_current_mini_batch_size();
206 
207  update_num_samples_processed(num_samples_in_batch);
208  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
209  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
211  update_num_samples_processed(num_samples_in_batch);
212  }
213 
214  int expected_num_samples_in_batch = this->m_model->get_current_mini_batch_size();
215 
218  num_samples_in_batch = Layer::m_comm->model_broadcast(((distributed_io_buffer*) io_buffer)->current_root_rank(mode), num_samples_in_batch);
219  this->m_model->set_current_mini_batch_size(num_samples_in_batch
220  + get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()));
221 
222  io_buffer->distribute_from_local_matrix(get_activations(), get_data_reader(), mode);
223 
224  if(num_samples_in_batch !=
225  (expected_num_samples_in_batch - get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()))) {
226  std::stringstream err;
227  err << __FILE__ << " " << __LINE__ << " :: "
228  << "I/O layers number of samples processed ("<< num_samples_in_batch
229  <<") does not match the mini-batch size ("
230  << (expected_num_samples_in_batch -
231  get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()))
232  << ")";
233  throw lbann_exception(err.str());
234  }
235  }else {
236  std::stringstream err;
237  err << __FILE__ << " " << __LINE__ << " :: "
238  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
239  throw lbann_exception(err.str());
240  }
241  }

256  generic_data_reader *get_data_reader(const execution_mode mode) const {
257  generic_data_reader *data_reader = nullptr;
258
