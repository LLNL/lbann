35 void lbann_callback_checksmall::on_forward_prop_end(model *m, Layer *l) {
36  if (dynamic_cast<generic_target_layer*>(l) != nullptr) {
37  return;
38  }
39  const AbsDistMat& acts = l->get_activations();
40  if (!is_good(acts)) {
41  std::stringstream ss;
42  ss << name() << ": "
43  << "[" << std::to_string(m->get_comm()->get_rank_in_world()) << "]: "
44  << "error in activations of " << l->get_name() << " "
45  << "(step=" << std::to_string(m->get_cur_step()) << ")";

47  }
48 }

50 void lbann_callback_checksmall::on_backward_prop_end(model *m) {
51  for (weights *w : m->get_weights()) {
52  optimizer *opt = w->get_optimizer();

54  std::stringstream ss;

60  }
61  }

64 void lbann_callback_checksmall::on_batch_end(model *m) {
65  for (weights *w : m->get_weights()) {

67  std::stringstream ss;

70  << "error in weights of " << w->get_name() << " "

77 bool lbann_callback_checksmall::is_good(const AbsDistMat& m) {
78  const Mat& local_mat = m.LockedMatrix();
79  const El::Int height = local_mat.Height();
80  const El::Int width = local_mat.Width();
81  for (El::Int col = 0; col < width; ++col) {
82  for (El::Int row = 0; row < height; ++row) {
83  const DataType val = std::abs(local_mat(row, col));
84  if (val > 0 && val <= m_threshold) {
85  std::cout << "Found small value " << val << " "
86  << "at (" << row << "," << col << ")!" << std::endl;
87  return false;
88  }
89  }
90  }
91  return true;
92 }
