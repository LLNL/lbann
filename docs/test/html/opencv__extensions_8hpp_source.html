89 template<typename T>
90 struct depth_normalization {
91  static double factor() {
92  if (!std::is_integral<T>::value) {
93  return 1.0;
94  } else {
95  return 1.0/std::numeric_limits<T>::max();
96  }
97  }
98  static double inverse_factor() {
99  if (!std::is_integral<T>::value) {
100  return 1.0;
101  } else {
102  return std::numeric_limits<T>::max();
103  }
104  }
105 };
