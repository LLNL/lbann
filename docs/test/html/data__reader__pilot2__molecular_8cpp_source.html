39 void pilot2_molecular_reader::load() {
40  // support for data store functionality: when not using data store, all procs
41  // load the data; when using data store, only one does so
42  bool is_mine = true;
43  int rank = m_comm->get_rank_in_model();
44  // note: when support for merge_samples is in place, the condition
45  // "get_role() == "test" will go away. For now we need it, else
46  // merge_samples will break
47  options *opts = options::get();
48  if (opts->has_bool("use_data_store") && opts->get_bool("use_data_store") && get_role() == "test") {
49  if (rank != get_compound_rank()) {
50  is_mine = false;
51  }
52  }
53 
54  if (is_mine) {
55  std::string infile = get_file_dir() + get_data_filename();
56  // Ensure the file exists.
57  std::ifstream ifs(infile);
58  if (!ifs) {
59  throw lbann_exception(
60  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
61  " pilot2_molecular::load() - can't open file : " + infile);
62  }
63  ifs.close();
64 
65  // Load the dictionary.
66  cnpy::npz_t dict = cnpy::npz_load(infile);
67  // Verify we have features and neighbors.
68  if (dict.count("features") != 1) {
69  throw lbann_exception(
70  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
71  " pilot2_molecular::load() - no features");
72  }
73  if (dict.count("neighbors") != 1) {
74  throw lbann_exception(
75  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
76  " pilot2_molecular::load() - no neighbors");
77  }
78  m_features = dict["features"];
79  m_neighbors = dict["neighbors"];
80 
81  // Ensure we understand the word size.
82  if (!(m_features.word_size == 4 || m_features.word_size == 8)) {
83  throw lbann_exception(
84  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
85  " pilot2_molecular::load() - feature word size " +
86  std::to_string(m_features.word_size) + " not supported");
87  }
88  if (!(m_neighbors.word_size == 4 || m_neighbors.word_size == 8)) {
89  throw lbann_exception(
90  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
91  " pilot2_molecular::load() - neighbor word size " +
92  std::to_string(m_neighbors.word_size) + " not supported");
93  }
94  // Fortran data order not supported.
95  if (m_features.fortran_order) {
96  throw lbann_exception(
97  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
98  " pilot2_molecular::load() - feature fortran order not supported");
99  }
100  if (m_neighbors.fortran_order) {
101  throw lbann_exception(
102  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
103  " pilot2_molecular::load() - neighbor fortran order not supported");
104  }
105 
106  // Assume we collapse samples from every frame into one set.
107  m_num_samples = m_features.shape[0] * m_features.shape[1];
108  m_num_samples_per_frame = m_features.shape[1];
109  // The first two dimensions are the frame and the sample, so skip.
110  m_num_features = std::accumulate(
111  m_features.shape.begin() + 2, m_features.shape.end(), (unsigned) 1,
112  std::multiplies<unsigned>());
113 
114  m_word_size = m_neighbors.word_size;
115 
116  m_shape.resize(3);
117  m_shape[0] = m_num_neighbors + 1;
118  m_shape[1] = m_features.shape[2];
119  m_shape[2] = m_features.shape[3];
120  }
121 
122  // when using data store, need to bcast some variable to all procs
123  if (options::get()->get_bool("use_data_store")) {
124  std::vector<int> tmp(8);
125  if (rank == get_compound_rank()) {
126  //@todo: fix if we have floats!
127  m_neighbors_data_size = m_neighbors.data_holder->size() / 8;
128 
129  tmp[0] = m_num_samples;
130  tmp[1] = m_num_samples_per_frame;
131  tmp[2] = m_num_features;
132  tmp[3] = m_num_neighbors + 1;
133  tmp[4] = m_features.shape[2];
134  tmp[5] = m_features.shape[3];
135  tmp[6] = m_word_size;
136  tmp[7] = m_neighbors_data_size;
137  }
138  MPI_Bcast(tmp.data(), 8, MPI_INT, get_compound_rank(), m_comm->get_model_comm().comm);
139  m_num_samples = tmp[0];
140  m_num_samples_per_frame = tmp[1];
141  m_num_features = tmp[2];
142  m_shape.resize(3);
143  m_shape[0] = tmp[3];
144  m_shape[1] = tmp[4];
145  m_shape[2] = tmp[5];
146  m_word_size = tmp[6];
147  m_neighbors_data_size = tmp[7];
148  }
149 
150  // Reset indices.
151  m_shuffled_indices.clear();
152  m_shuffled_indices.resize(m_num_samples);
153  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
154  select_subset_of_data();
155 }

157 bool pilot2_molecular_reader::fetch_datum(
158  Mat& X, int data_id, int mb_idx, int tid) {
159 
160  if (m_data_store != nullptr) {
161  std::vector<double> *buf;
162  size_t jj = 0;
163  m_data_store->get_data_buf(data_id, tid, buf);
164  for (int idx = 0; idx < m_num_neighbors+1; idx++) {
165  for (int i = 0; i < m_num_features; ++i) {
166  X(m_num_features * idx + i, mb_idx) = (*buf)[jj++];
167  //note: scale_data was already computed by the data_store
168  }
169  }
170  return true;
171  }
172 
173  const int frame = get_frame(data_id);
174  // Fetch the actual molecule.
175  fetch_molecule(X, data_id, 0, mb_idx);
176  // Fetch the neighbors - note that the offset is 2x the max
177  // neighborhood size to accommodate the top and bottom of the
178  // bilayer
179  const int neighbor_frame_offset =
180  frame * m_num_samples_per_frame * (2 * m_max_neighborhood);
181  const int intra_frame_data_id = data_id - frame * m_num_samples_per_frame;
182  if (m_neighbors.word_size == 4) {
183  float *neighbor_data = m_neighbors.data<float>() +
184  neighbor_frame_offset + intra_frame_data_id * (2 * m_max_neighborhood);
185  // Start at 1 to skip self.
186  for (int i = 1; i < m_num_neighbors + 1; ++i) {
187  int neighbor_id = neighbor_data[i];
188  if (neighbor_id != -1) {
189  fetch_molecule(X, neighbor_id + frame * m_num_samples_per_frame, i,
190  mb_idx);
191  }
192  }
193  } else if (m_neighbors.word_size == 8) {
194  double *neighbor_data = m_neighbors.data<double>() +
195  neighbor_frame_offset + intra_frame_data_id * (2 * m_max_neighborhood);
196  // Start at 1 to skip self.
197  for (int i = 1; i < m_num_neighbors + 1; ++i) {
198  int neighbor_id = neighbor_data[i];
199  if (neighbor_id != -1) {
200  fetch_molecule(X, neighbor_id + frame * m_num_samples_per_frame, i,
201  mb_idx);
202  }
203  }
204  }
205  return true;
206 }

208 void pilot2_molecular_reader::fetch_molecule(Mat& X, int data_id, int idx,
209  int mb_idx) {
210  const int frame = get_frame(data_id);
211  // Compute the offset in features for this frame.
212  const int frame_offset = frame * m_num_features * m_num_samples_per_frame;
213  const int intra_frame_data_id = data_id - frame * m_num_samples_per_frame;
214  if (m_features.word_size == 4) {
215  float *data = m_features.data<float>() + frame_offset +
216  intra_frame_data_id * m_num_features;
217  for (int i = 0; i < m_num_features; ++i) {
218  X(m_num_features * idx + i, mb_idx) = scale_data<float>(i, data[i]);
219  }
220  } else if (m_features.word_size == 8) {
221  double *data = m_features.data<double>() + frame_offset +
222  intra_frame_data_id * m_num_features;
223  for (int i = 0; i < m_num_features; ++i) {
224  X(m_num_features * idx + i, mb_idx) = scale_data<double>(i, data[i]);
225  }
226  }
227 }
