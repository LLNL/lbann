47 void unroll_input_layer(int unroll_depth,
48  std::vector<Layer*>& layers,
49  std::unordered_map<const Layer*,Layer*>& prev_step_layer,
50  std::unordered_map<const Layer*,Layer*>& next_step_layer) {
51  std::stringstream err;
52 
53  // We expect first layer to be an input layer
54  auto&& input = dynamic_cast<generic_input_layer*>(layers.front());
55  if (input == nullptr) {
56  err << __FILE__ << " " << __LINE__ << " :: "
57  << "expected the first layer to be an input layer";
58  throw lbann_exception(err.str());
59  }
60 
61  // Determine slice points
62  const auto& input_dims = input->get_neuron_dims();
63  const auto& slice_size = input_dims.front() / unroll_depth;
64  std::vector<int> slice_points;
65  for (int step = 0; step <= unroll_depth; ++step) {
66  slice_points.push_back(slice_size * step);
67  }
68 
69  // Construct slice and split layer
70  Layer* slice = nullptr;
71  Layer* split = nullptr;
72  auto&& comm = input->get_comm();
73  cudnn::cudnn_manager* cudnn = nullptr;
74  switch (input->get_data_layout()) {
75  case data_layout::DATA_PARALLEL:
76  slice = new slice_layer<data_layout::DATA_PARALLEL>(comm, 0, slice_points, cudnn);
77  split = new split_layer<data_layout::DATA_PARALLEL>(comm, cudnn);
78  break;
79  case data_layout::MODEL_PARALLEL:
80  slice = new slice_layer<data_layout::MODEL_PARALLEL>(comm, 0, slice_points, cudnn);
81  split = new split_layer<data_layout::MODEL_PARALLEL>(comm, cudnn);
82  break;
83  default:
84  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
85  throw lbann_exception(err.str());
86  }
87  slice->set_name(input->get_name() + "_slice");
88  split->set_name(input->get_name() + "_split");
89  layers.insert(layers.begin() + 1, slice);
90  layers.insert(layers.begin() + 2, split);
91 
92  // Setup relationships between split layer and child layers
93  for (auto&& child : input->get_child_layers()) {
94  split->add_child_layer(child);
95  auto& child_parents = const_cast<Layer*>(child)->get_parent_layers();
96  std::replace(child_parents.begin(), child_parents.end(),
97  static_cast<Layer*>(input), split);
98  }
99  input->clear_child_layers();
100 
101  // Setup relationship between input layer, slice layer, and split layer
102  input->add_child_layer(slice);
103  slice->add_parent_layer(input);
104  slice->add_child_layer(split);
105  split->add_parent_layer(slice);
106 
107  // Input layer and slice layer are not unrolled any further
108  prev_step_layer[input] = input;
109  prev_step_layer[slice] = slice;
110  next_step_layer[input] = input;
111  next_step_layer[slice] = slice;
112 
113 }

119 void unroll_target_layer(int unroll_depth,
120  std::vector<Layer*>& layers,
121  std::unordered_map<const Layer*,Layer*>& prev_step_layer,
122  std::unordered_map<const Layer*,Layer*>& next_step_layer) {
123  std::stringstream err;
124 
125  // We expect last layer to be a target layer
126  auto&& target = dynamic_cast<generic_target_layer*>(layers.back());
127  if (target == nullptr) {
128  err << __FILE__ << " " << __LINE__ << " :: "
129  << "expected the last layer to be a target layer";
130  throw lbann_exception(err.str());
131  }
132 
133  // Construct concatenation layer
134  Layer* concat = nullptr;
135  auto&& comm = target->get_comm();
136  switch (target->get_data_layout()) {
137  case data_layout::DATA_PARALLEL:
138  concat = new concatenation_layer<data_layout::DATA_PARALLEL>(comm, 0, nullptr);
139  break;
140  case data_layout::MODEL_PARALLEL:
141  concat = new concatenation_layer<data_layout::MODEL_PARALLEL>(comm, 0, nullptr);
142  break;
143  default:
144  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
145  throw lbann_exception(err.str());
146  }
147  concat->set_name(target->get_name() + "_concat");
148  layers.insert(layers.end() - 1, concat);
149 
150  // Setup relationships between concatenation layer and parent layers
151  for (auto&& parent : target->get_parent_layers()) {
152  concat->add_parent_layer(parent);
153  auto& parent_children = const_cast<Layer*>(parent)->get_child_layers();
154  std::replace(parent_children.begin(), parent_children.end(),
155  static_cast<Layer*>(target), concat);
156  }
157  target->clear_parent_layers();
158 
159  // Setup relationship between target layer and concatenation layer
160  concat->add_child_layer(target);
161  target->add_parent_layer(concat);
162 
163  // Target layer and concatenation layer are not unrolled any further
164  prev_step_layer[target] = target;
165  prev_step_layer[concat] = concat;
166  next_step_layer[target] = target;
167  next_step_layer[concat] = concat;
168 
169 }

175 void add_unrolled_layers(int unroll_depth,
176  std::vector<Layer*>& layers,
177  std::unordered_map<const Layer*,Layer*>& prev_step_layer,
178  std::unordered_map<const Layer*,Layer*>& next_step_layer) {
179 
180  // Unroll network to desired depth
181  std::vector<Layer*> previous_step(layers.begin() + 2, layers.end() - 2);
182  const int num_step_layers = previous_step.size();
183  for (int step = 1; step < unroll_depth; ++step) {
184 
185  // Construct current step by copying layers from previous step
186  std::vector<Layer*> current_step;
187  for (const auto& previous_layer : previous_step) {
188  auto&& current_layer = previous_layer->copy();
189  current_step.push_back(current_layer);
190  prev_step_layer[current_layer] = previous_layer;
191  next_step_layer[previous_layer] = current_layer;
192  }
193 
194  // Fix pointers within current step
195  for (const auto& current_layer : current_step) {
196  auto layer_pointers = current_layer->get_layer_pointers();
197  for (auto& layer_pointer : layer_pointers) {
198  layer_pointer = next_step_layer[layer_pointer];
199  }
200  current_layer->set_layer_pointers(layer_pointers);
201  }
202 
203  // Add current step layers to model
204  layers.insert(layers.end() - 2, current_step.begin(), current_step.end());
205  previous_step = current_step;
206 
207  }
208 
209  // Rename layers
210  for (int step = 0; step < unroll_depth; ++step) {
211  const std::string name_suffix = "_step" + std::to_string(step);
212  const int step_start = 2 + step * num_step_layers;
213  const int step_end = 2 + (step + 1) * num_step_layers;
214  for (int i = step_start; i < step_end; ++i) {
215  layers[i]->set_name(layers[i]->get_name() + name_suffix);
216  }
217  }
218 
219 }

227 void add_placeholder_layers(std::vector<Layer*>& layers,
228  std::unordered_map<const Layer*,Layer*>& prev_step_layer,
229  std::unordered_map<const Layer*,Layer*>& next_step_layer) {
230 
231  std::unordered_map<const Layer*,bool> is_visited;
232  std::vector<Layer*> input_placeholders, output_placeholders;
233  for (const auto& l : layers) {
234 
235  // Create constant layers as input placeholders
236  for (auto&& parent : l->get_parent_layers()) {
237  if (!is_visited[parent] && prev_step_layer[parent] == nullptr) {
238  if (parent->get_num_neurons() <= 0) {
239  std::stringstream err;
240  err << __FILE__ << " " << __LINE__ << " :: "
241  << l->get_name() << " has ambiguous neuron "
242  << "dimensions since it depends on "
243  << parent->get_name() << ", which does not have "
244  << "specified neuron dimensions. Consider inserting a "
245  << "reshape layer after " << parent->get_name() << " "
246  << "to explicitly specify neuron dimensions.";
247  throw lbann_exception(err.str());
248  }
249  Layer* placeholder = nullptr;
250  auto&& comm = parent->get_comm();
251  switch (parent->get_data_layout()) {
252  case data_layout::DATA_PARALLEL:
253  placeholder = new constant_layer<data_layout::DATA_PARALLEL>(
254  comm,
255  DataType(0),
256  parent->get_neuron_dims()
257  );
258  break;
259  case data_layout::MODEL_PARALLEL:
260  placeholder = new constant_layer<data_layout::MODEL_PARALLEL>(
261  comm,
262  DataType(0),
263  parent->get_neuron_dims()
264  );
265  break;
266  default:
267  std::stringstream err;
268  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
269  throw lbann_exception(err.str());
270  }
271  placeholder->set_name(parent->get_name() + "_input_placeholder");
272  placeholder->add_child_layer(l);
273  input_placeholders.push_back(placeholder);
274  prev_step_layer[parent] = placeholder;
275  next_step_layer[placeholder] = const_cast<Layer*>(parent);
276  }
277  }
278 
279  // Visit layer
280  is_visited[l] = true;
281 
282  // Create dummy layers as output placeholders
283  for (auto&& child : l->get_child_layers()) {
284  if (is_visited[child] && next_step_layer[child] == nullptr) {
285  Layer* placeholder = nullptr;
286  auto&& comm = child->get_comm();
287  switch (child->get_data_layout()) {
288  case data_layout::DATA_PARALLEL:
289  placeholder = new dummy_layer<data_layout::DATA_PARALLEL>(comm);
290  break;
291  case data_layout::MODEL_PARALLEL:
292  placeholder = new dummy_layer<data_layout::MODEL_PARALLEL>(comm);
293  break;
294  default:
295  std::stringstream err;
296  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
297  throw lbann_exception(err.str());
298  }
299  placeholder->set_name(child->get_name() + "_output_placeholder");
300  placeholder->add_parent_layer(l);
301  output_placeholders.push_back(placeholder);
302  next_step_layer[child] = placeholder;
303  prev_step_layer[placeholder] = const_cast<Layer*>(child);
304  }
305  }
306 
307  }
308 
309  // Add placeholder layers to model
310  layers.insert(layers.begin() + 2,
311  input_placeholders.begin(),
312  input_placeholders.end());
313  layers.insert(layers.end() - 2,
314  output_placeholders.begin(),
315  output_placeholders.end());
316 
317 }

326 void setup_unrolled_layer_pointers(std::vector<Layer*>& layers,
327  const std::unordered_map<const Layer*,Layer*>& prev_step_layer,
328  const std::unordered_map<const Layer*,Layer*>& next_step_layer) {
329 
330  std::unordered_map<const Layer*,bool> is_visited;
331  for (auto&& l : layers) {
332  for (auto& parent : l->get_parent_layers()) {
333  if (!is_visited[parent]) {
334  parent = prev_step_layer.at(parent);
335  }
336  }
337  is_visited[l] = true;
338  for (auto& child : l->get_child_layers()) {
339  if (is_visited[child]) {
340  child = next_step_layer.at(child);
341  }
342  }
343  }

361 void recurrent_model::setup_layer_topology() {
362 
363  // Make sure parent/child relationships are reciprocated
364  for (const auto& l : m_layers) {
365  for (const auto& parent : l->get_parent_layers()) {
366  const_cast<Layer*>(parent)->add_child_layer(l);
367  }
368  for (const auto& child : l->get_child_layers()) {
369  const_cast<Layer*>(child)->add_parent_layer(l);
370  }
371  }
372 
373  // Unroll layers
374  if (m_layers.size() < 2) {
375  std::stringstream err;
376  err << __FILE__ << " " << __LINE__ << " :: "
377  << "expected the first layer to be an input layer "
378  << "and the last to be a target layer";
379  throw lbann_exception(err.str());
380 
381  }
382  m_previous_step_layer.clear();
383  m_next_step_layer.clear();
384  unroll_input_layer(m_unroll_depth, m_layers, m_previous_step_layer, m_next_step_layer);
385  unroll_target_layer(m_unroll_depth, m_layers, m_previous_step_layer, m_next_step_layer);
386  add_unrolled_layers(m_unroll_depth, m_layers, m_previous_step_layer, m_next_step_layer);
387  add_placeholder_layers(m_layers, m_previous_step_layer, m_next_step_layer);
388  setup_unrolled_layer_pointers(m_layers, m_previous_step_layer, m_next_step_layer);
389 
390  // Make sure unrolled topology is a DAG
391  directed_acyclic_graph_model::setup_layer_topology();
392 
393 }

395 void recurrent_model::setup_layers() {
396  for (size_t i=0; i<m_layers.size(); ++i) {
397  auto&& l = m_layers[i];
398 
399  // Set slice points for the inserted slice layer
400  if (i == 1) {
401  std::vector<int> slice_points;
402  const auto& input_dims = m_layers[0]->get_neuron_dims();
403  const auto& slice_size = input_dims[0] / m_unroll_depth;
404  for (int step = 0; step <= m_unroll_depth; ++step) {
405  slice_points.push_back(slice_size * step);
406  }
407  auto&& slice_dp = dynamic_cast<slice_layer<data_layout::DATA_PARALLEL>*>(l);
408  auto&& slice_mp = dynamic_cast<slice_layer<data_layout::MODEL_PARALLEL>*>(l);
409  if (slice_dp != nullptr) {
410  slice_dp->get_slice_points() = slice_points;
411  }
412  if (slice_mp != nullptr) {
413  slice_mp->get_slice_points() = slice_points;
414  }
415  if (slice_dp == nullptr && slice_mp == nullptr) {
416  std::stringstream err;
417  err << __FILE__ << " " << __LINE__ << " :: "
418  << "expected the second layer to be a slice layer, "
419  << "but " << l->get_name() << " is " << l->get_type();
420  throw lbann_exception(err.str());
421  }
422  }
423 
424  // Corresponding weights in different steps share weights
425  auto&& prev_step_layer = m_previous_step_layer[l];
426  if (prev_step_layer != nullptr) {
427  auto&& w = prev_step_layer->get_weights();
428  if (!w.empty()) {
429  l->set_weights(w);
430  }
431  }
432 
433  // Setup layer
434  l->set_model(this);
435  l->setup();
436  l->check_setup();
437  if (m_comm->am_world_master()) {
438  std::cout << print_layer_description(l) << std::endl;
439  }
440 
441  }
