82 void print_stack_trace() {
83  if (to_file.is_open()) {
84  to_file << "\n**************************************************************************\n";
85  }

87  #define MAX_STACK_FRAMES 64
88  static void *stack_traces[MAX_STACK_FRAMES];
89  int trace_size = backtrace(stack_traces, MAX_STACK_FRAMES);
90  char **messages = backtrace_symbols(stack_traces, trace_size);
91  Dl_info info;
92  for (int i=0; i<trace_size; i++) {
93  std::cerr << "rank: " << my_lbann_tracing_id << " :: ";
94  dladdr(stack_traces[i], &info);
95  if (info.dli_sname != NULL) {
96  char *demangled_name = abi::__cxa_demangle(info.dli_sname, nullptr, nullptr, nullptr);
97  if (demangled_name != NULL) {
98  std::string test(demangled_name);
99  /*if (test.find("__libc_start_main") != std::string::npos) {
100  i = 100000;
101  break;
102  }
103  */
104  if (test.find("lbann::stack_trace::print_stack_trace") == std::string::npos && test.find("lbann::stack_trace::print_lbann_exception_stack_trace") == std::string::npos) {
105  std::cerr << demangled_name << std::endl;
106  if (to_file.is_open()) {
107  to_file << " >>>> " << demangled_name << std::endl;
108  }
109  }
110  free(demangled_name);
111  } else {
112  std::cerr << "demangling failed for: " << info.dli_sname << std::endl;
113  if (to_file.is_open()) {

115  }
116  }
117  } else {

119  if (to_file.is_open()) {
120  to_file << " >>>> dli_sname == NULL for: " << stack_traces[i] << " backtrace message was: " << messages[i] << std::endl;
121  }
122  }
123  }
