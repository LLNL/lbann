76 void adagrad::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
77 
78  // Get local matrix data
79  const int local_height = values.LocalHeight();
80  const int local_width = values.LocalWidth();
81  DataType* __restrict__ values_buffer = values.Buffer();
82  const int values_ldim = values.LDim();
83  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
84  const int gradient_ldim = gradient.LDim();
85  DataType* __restrict__ cache_buffer = m_cache->Buffer();
86  const int cache_ldim = m_cache->LDim();
87 
88  // Check if matrix data is contiguous
89  if (values_ldim != local_height
90  || gradient_ldim != local_height
91  || cache_ldim != local_height) {
92  // Update with non-contiguous data
93  #pragma omp parallel for collapse(2)
94  for (int j=0; j<local_width; ++j) {
95  for (int i=0; i<local_height; ++i) {
96  DataType& x = values_buffer[i+j*values_ldim];
97  const DataType g = gradient_buffer[i+j*gradient_ldim];
98  DataType& c = cache_buffer[i+j*cache_ldim];
99  c += g * g;
100  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
101  }
102  }
103  } else {
104  // Update with contiguous data
105  #pragma omp parallel for
106  for (int i=0; i<local_height*local_width; ++i) {
107  DataType& x = values_buffer[i];
108  const DataType g = gradient_buffer[i];
109  DataType& c = cache_buffer[i];
110  c += g * g;
111  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
112  }
113  }
114 }
