38 template <typename T>
39 std::unordered_set<T*> select_from_list(std::string names,
40  std::vector<T*> list) {
41  std::unordered_set<T*> selected;
42  for (const auto& name : parse_list<std::string>(names)) {
43  for (auto&& t : list) {
44  if (name == t->get_name()) {
45  selected.insert(t);
46  }
47  }
48  }
49  return selected;
50 }

54 lbann_callback* construct_callback(lbann_comm* comm,
55  const lbann_data::Callback& proto_cb,
56  std::map<execution_mode, generic_data_reader*>& data_readers,
57  std::vector<Layer*> layer_list,
58  std::vector<weights*> weights_list,
59  lbann_summary* summarizer) {
60  std::stringstream err;
61 
63  // Display information
65  if (proto_cb.has_print()) {
66  const auto& interval = proto_cb.print().interval();
67  return new lbann_callback_print(interval);
68  }
69  if (proto_cb.has_timer()) {
70  return new lbann_callback_timer(summarizer);
71  }
72  if (proto_cb.has_disp_io_stats()) {
73  const auto& params = proto_cb.disp_io_stats();
74  auto&& selected_layers = select_from_list<Layer>(params.layers(),
75  layer_list);
76  return new lbann_callback_io(selected_layers);
77  }
78  if (proto_cb.has_save_images()) {
79  const auto& params = proto_cb.save_images();
80  auto&& reader = data_readers[execution_mode::training] ;
81  return new lbann_callback_save_images(reader,
82  params.image_dir(),
83  params.extension());
84  }
85 
87  // Inter-model communication
89  if (proto_cb.has_ltfb()) {
90  return new lbann_callback_ltfb(proto_cb.ltfb().round_size(),
91  summarizer);
92  }
94  if (proto_cb.has_imcomm()) {
95  const auto& params = proto_cb.imcomm();
96  const auto& type_str = params.intermodel_comm_method();
97  lbann_callback_imcomm::comm_type type = lbann_callback_imcomm::comm_type::NONE;
98  if (type_str == "none") {
99  type = lbann_callback_imcomm::comm_type::NONE;
100  } else if (type_str == "normal") {
101  type = lbann_callback_imcomm::comm_type::NORMAL;
102  } else if (type_str == "onebit_quantization") {
103  type = lbann_callback_imcomm::comm_type::ONEBIT_QUANTIZATION;
104  } else if (type_str == "thresh_quantization") {
105  type = lbann_callback_imcomm::comm_type::THRESH_QUANTIZATION;
106  } else if (type_str == "adaptive_quantization") {
107  type = lbann_callback_imcomm::comm_type::ADAPTIVE_QUANTIZATION;
108  } else {
109  err << "invalid inter-model communication type (" << type_str << ")";
110  LBANN_ERROR(err.str());
111  }
112  std::unordered_set<weights*> selected_weights;
113  return new lbann_callback_imcomm(type, selected_weights, summarizer);
114  }
115 
117  // Learning rate schedules
119  if (proto_cb.has_step_learning_rate()) {
120  const auto& params = proto_cb.step_learning_rate();
121  auto&& selected_weights = select_from_list<weights>(params.weights(),
122  weights_list);
123  return new lbann_callback_step_learning_rate(params.step(),
124  params.amt(),
125  selected_weights);
126  }
127  if (proto_cb.has_adaptive_learning_rate()) {
128  const auto& params = proto_cb.adaptive_learning_rate();
129  auto&& selected_weights = select_from_list<weights>(params.weights(),
130  weights_list);
131  return new lbann_callback_adaptive_learning_rate(params.patience(),
132  params.amt(),
133  selected_weights);
134  }
135  if (proto_cb.has_drop_fixed_learning_rate()) {
136  const auto& params = proto_cb.drop_fixed_learning_rate();
137  std::vector<int64_t> drop_epochs;
138  for (int i = 0; i < params.drop_epoch_size(); ++i) {
139  drop_epochs.push_back(params.drop_epoch(i));
140  }
141  auto&& selected_weights = select_from_list<weights>(params.weights(),
142  weights_list);
143  return new lbann_callback_drop_fixed_learning_rate(drop_epochs,
144  params.amt(),
145  selected_weights);
146  }
147  if (proto_cb.has_linear_growth_learning_rate()) {
148  const auto& params = proto_cb.linear_growth_learning_rate();
149  auto&& selected_weights = select_from_list<weights>(params.weights(),
150  weights_list);
151  return new lbann_callback_linear_growth_learning_rate(params.target(),
152  params.num_epochs(),
153  params.delay(),
154  selected_weights);
155  }
156  if (proto_cb.has_optimizerwise_adaptive_learning_rate()) {
157  const auto& params = proto_cb.optimizerwise_adaptive_learning_rate();
158  auto&& selected_weights = select_from_list<weights>(params.weights(),
159  weights_list);
160  return new lbann_callback_optimizerwise_adaptive_learning_rate(params.scale(),
161  selected_weights);
162  }
163  if (proto_cb.has_poly_learning_rate()) {
164  const auto& params = proto_cb.poly_learning_rate();
165  auto&& selected_weights = select_from_list<weights>(params.weights(),
166  weights_list);
167  return new lbann_callback_poly_learning_rate(params.power(),
168  params.num_epochs(),
169  params.max_iter(),
170  selected_weights);
171  }
172 
174  // Mini-batch schedules
176  if (proto_cb.has_step_minibatch()) {
177  const auto& params = proto_cb.step_minibatch();
178  return new lbann_callback_step_minibatch(params.starting_mbsize(),
179  params.step(),
180  params.ramp_time());
181  }
182  if (proto_cb.has_minibatch_schedule()) {
183  const auto& params = proto_cb.minibatch_schedule();
184  std::vector<lbann_callback_minibatch_schedule::minibatch_step> steps;
185  for (int i = 0; i < params.step_size(); ++i) {
186  const auto& proto_step = params.step(i);
187  steps.emplace_back(proto_step.epoch(),
188  proto_step.mbsize(),
189  proto_step.lr(),
190  proto_step.ramp_time());
191  }
192  return new lbann_callback_minibatch_schedule(params.starting_mbsize(),
193  steps);
194  }
195 
197  // Checkpointing and exporting
199  if (proto_cb.has_checkpoint()) {
200  const auto& params = proto_cb.checkpoint();
201  return new lbann_callback_checkpoint(params.checkpoint_dir(),
202  params.checkpoint_epochs(),
203  params.checkpoint_steps(),
204  params.checkpoint_secs(),
205  params.checkpoint_per_rank());
206  }
207  if (proto_cb.has_save_model()) {
208  const auto& params = proto_cb.save_model();
209  return new lbann_callback_save_model(params.dir(),
210  params.extension());
211  }
212 
214  // Profiling
216  if (proto_cb.has_summary()) {
217  const auto& params = proto_cb.summary();
218  return new lbann_callback_summary(summarizer,
219  params.batch_interval(),
220  params.mat_interval());
221  }
222  if (proto_cb.has_profiler()) {
223  return new lbann_callback_profiler();
224  }
225 
227  // Debugging
229  if (proto_cb.has_debug()) {
230  const auto& phase = proto_cb.debug().phase();
231  if (phase == "train" || phase == "training") {
232  return new lbann_callback_debug(execution_mode::training, summarizer);
233  } else if (phase == "validate" || phase == "validation") {
234  return new lbann_callback_debug(execution_mode::validation, summarizer);
235  } else if (phase == "test" || phase == "testing") {
236  return new lbann_callback_debug(execution_mode::testing, summarizer);
237  } else {
238  return new lbann_callback_debug();
239  }
240  }
241  if (proto_cb.has_debug_io()) {
242  const auto& params = proto_cb.debug_io();
243  const auto& phase = params.phase();
244  const auto& lvl = params.lvl();
245  if (phase == "train" || phase == "training") {
246  return new lbann_callback_debug_io(execution_mode::training, lvl);
247  } else if (phase == "validate" || phase == "validation") {
248  return new lbann_callback_debug_io(execution_mode::validation, lvl);
249  } else if (phase == "test" || phase == "testing") {
250  return new lbann_callback_debug_io(execution_mode::testing, lvl);
251  } else {
252  return new lbann_callback_debug_io();
253  }
254  }
255  if (proto_cb.has_dump_weights()) {
256  const auto& params = proto_cb.dump_weights();
257  return new lbann_callback_dump_weights(params.basename(),
258  params.interval());
259  }
260  if (proto_cb.has_dump_activations()) {
261  const auto& params = proto_cb.dump_activations();
262  const auto& layer_names = parse_list<>(params.layer_names());
263  return new lbann_callback_dump_activations(params.basename(),
264  params.interval(),
265  layer_names);
266  }
267  if (proto_cb.has_dump_gradients()) {
268  const auto& params = proto_cb.dump_gradients();
269  return new lbann_callback_dump_gradients(params.basename(),
270  params.interval());
271  }
272  if (proto_cb.has_dump_mb_indices()) {
273  const auto& params = proto_cb.dump_mb_indices();
274  return new lbann_callback_dump_minibatch_sample_indices(params.basename(),
275  params.interval());
276  }
277  if (proto_cb.has_check_dataset()) {
278  return new lbann_callback_check_dataset();
279  }
280  if (proto_cb.has_check_small()) {
281  return new lbann_callback_checksmall();
282  }
283  if (proto_cb.has_check_nan()) {
284  return new lbann_callback_checknan();
285  }
286  if (proto_cb.has_hang()) {
287  const auto& rank_to_hang = proto_cb.hang().rank();
288  if (comm->am_world_master()) {
289  if (rank_to_hang == -1) {
290  std::cout << "*** HANGING EVERY RANK IN HANG CALLBACK ***"
291  << std::endl;
292  } else {
293  std::cout << "*** HANGING RANK " << rank_to_hang
294  << " IN HANG CALLBACK ***" << std::endl;
295  }
296  }
297  return new lbann_callback_hang(rank_to_hang);
298  }
299 
301  // Gradient checking
303  if (proto_cb.has_gradient_check()) {
304  const auto& params = proto_cb.gradient_check();
305  return new lbann_callback_gradient_check(params.step_size(),
306  params.verbose(),
307  params.fail_on_error());
308  }
309 
310  return nullptr;
311 
312 }

314 lbann_summary* construct_summarizer(lbann_comm* comm,
315  const lbann_data::Model& m) {
316  lbann_summary *summary = nullptr;
317  bool master = comm->am_world_master();
318  int size = m.callback_size();
319  for (int j=0; j<size; j++) {
320  const lbann_data::Callback& callback = m.callback(j);
321  if (callback.has_summary()) {
322  const lbann_data::CallbackSummary& c = callback.summary();
323  if (master) {
324  std::cout << "constructing summarizer with dir: " << c.dir() << std::endl;
325  }
326 
327  //check to see if directory exists
328  struct stat sb;
329  if (! ( stat(c.dir().c_str(), &sb) == 0 && S_ISDIR(sb.st_mode) )) {
330  if (master) {
331  throw lbann_exception(
332  std::string {} + __FILE__ + " " + std::to_string(__LINE__) + " :: " +
333  "summary directory " + c.dir() + " does not exist");
334  }
335  }
336  summary = new lbann_summary(c.dir(), comm);
337  }
338  }
339  return summary;
340 }
