46 lbann_comm* initialize(int& argc, char**& argv, int seed) {
47  // Initialize Elemental.
48  El::Initialize(argc, argv);
49  // Create a new comm object.
50  // Initial creation with every process in one model.
51  auto* comm = new lbann_comm(0);
52 #if defined(LBANN_TOPO_AWARE)
53  // Determine the number of NUMA nodes present.
54  hwloc_topology_t topo;
55  hwloc_topology_init(&topo);
56  hwloc_topology_load(topo);
57  int numa_depth = hwloc_get_type_depth(topo, HWLOC_OBJ_NUMANODE);
58  if (numa_depth == HWLOC_TYPE_DEPTH_UNKNOWN) {
59  std::cout << comm->get_rank_in_world() <<
60  ": cannot determine hwloc NUMA-node depth" << std::endl;
61  }
62  int num_numa_nodes = hwloc_get_nbobjs_by_depth(topo, numa_depth);
63  // Warn if there are more NUMA nodes than processes per node.
64  // It's probably fine if there are more processes than NUMA nodes for now.
65  // We can adjust that later when we better understand the threaded perf.
66  int ppn = comm->get_procs_per_node();
67  if (num_numa_nodes > ppn) {
68  if (comm->get_rank_in_node() == 0) {
69  std::cout << comm->get_rank_in_world() <<
70  ": WARNING: node has " << num_numa_nodes <<
71  " NUMA nodes but you have " << ppn << " processes per node" <<
72  std::endl;
73  }
74  }
75  hwloc_topology_destroy(topo);
76 #endif
77  // Initialize local random number generators.
78  init_random(seed);
79  init_data_seq_random(seed);
80 
81  //initialization for stack tracing when a signal is raised
82  //or an lbann_exception thrown.
83  stack_trace::set_lbann_stack_trace_world_rank(comm->get_rank_in_world());
84 
85  return comm;
86 }
