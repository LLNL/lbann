37 void lbann_callback_dump_minibatch_sample_indices::dump_to_file(model *m, Layer *l, int64_t step) {
38  // Print minibatch sample indices of input layers
39  auto *input = dynamic_cast<generic_input_layer*>(l);
40  if (input != nullptr) {
41  El::Matrix<El::Int>* indices = l->get_sample_indices_per_mb();
42  if (indices == nullptr
43  || indices->Height() == 0
44  || indices->Width() == 0) {
45  return;
46  }
47  const std::string file
48  = (m_basename
49  + _to_string(m->get_execution_mode())
50  + "-model" + std::to_string(m->get_comm()->get_model_rank())
51  + "-rank" + std::to_string(m->get_comm()->get_rank_in_model())
52  + "-epoch" + std::to_string(m->get_cur_epoch())
53  + "-step" + std::to_string(m->get_cur_step())
54  + "-" + l->get_name()
55  + "-MB_Sample_Indices");
56  El::Write(*indices, file, El::ASCII);
57  }
58 }

64 void lbann_callback_dump_minibatch_sample_indices::on_evaluate_forward_prop_end(model *m, Layer *l) {
65  switch(m->get_execution_mode()) {
66  case execution_mode::validation:
67  dump_to_file(m, l, m->get_cur_validation_step());
68  break;
69  case execution_mode::testing:
70  dump_to_file(m, l, m->get_cur_testing_step());
71  break;
72  default:
73  throw lbann_exception("lbann_callback_dump_minibatch_sample_indices: invalid execution phase");
74  }
75 }
