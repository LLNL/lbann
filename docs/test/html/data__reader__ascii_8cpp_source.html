36 bool ascii_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
37 
38  // Get text sequence from file
39  const int pos = data_id - m_sequence_length;
40  const int num_chars = (std::min(pos + m_sequence_length, m_file_size)
41  - std::max(pos, 0));
42  std::vector<char> sequence(m_sequence_length, 0);
43  if (num_chars > 0) {
44  std::ifstream fs(get_file_dir() + get_data_filename(),
45  std::fstream::in);
46  fs.seekg(std::max(pos, 0));
47  fs.read(&sequence[std::max(-pos, 0)], num_chars);
48  fs.close();
49  }
50 
51  // Convert text sequence to binary vector
52  for (int i = 0; i < m_sequence_length; ++i) {
53  auto current_char = (int) sequence[i];
54  if (current_char < 0 || current_char >= 128) {
55  current_char = 0;
56  }
57  X(128 * i + current_char, mb_idx) = DataType(1);
58  }
59 
60  return true;
61 }
