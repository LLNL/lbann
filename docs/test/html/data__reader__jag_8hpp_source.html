184 template<typename T>
185 inline void set_minibatch_item(Mat& M, const int mb_idx, const T* const ptr, const size_t count) {
186  if ((count > 0u) && (ptr == nullptr)) {
187  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
188  " :: attempt to dereference a nullptr ");
189  }
190  for (size_t i = 0u; i < count; ++i) {
191  M.Set(static_cast<El::Int>(i), static_cast<El::Int>(mb_idx), static_cast<DataType>(ptr[i]));
192  }
193 }
