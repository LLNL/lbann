156 void data_store_csv::exchange_data() {
157  double tm1 = get_time();
158  std::stringstream err;
159 
160  //get indices I need for the next epoch, and start receives
161  std::unordered_set<int> indices;
162  get_indices(indices, m_rank);
163  std::vector<MPI_Request> recv_req(indices.size());
164  std::vector<MPI_Status> recv_status(indices.size());
165  m_my_minibatch_data.clear();
166  size_t jj = 0;
167  for (auto t : indices) {
168  m_my_minibatch_data[t].resize(m_vector_size);
169  int owner = get_index_owner(t);
170  if (owner >= m_np or owner < 0) {
171  err << __FILE__ << " " << __LINE__ << " :: "
172  << " ERROR: bad rank for owner in MPI_Irecv; owner: " << owner << " index: " << t << " jj: " << jj+1 << " of " << indices.size();
173  throw lbann_exception(err.str());
174  }
175  MPI_Irecv(
176  m_my_minibatch_data[t].data(), m_vector_size*sizeof(DataType), MPI_BYTE,
177  owner, t, m_mpi_comm, &(recv_req[jj++]));
178  }
179 
180  //start sends to all processors
181  std::vector<std::vector<MPI_Request>> send_req(m_np);
182  std::vector<std::vector<MPI_Status>> send_status(m_np);
183  for (int p=0; p<m_np; p++) {
184  get_my_indices(indices, p);
185  send_req[p].resize(indices.size());
186  send_status[p].resize(indices.size());
187  jj = 0;
188  for (auto t : indices) {
189  if (m_data.find(t) == m_data.end()) {
190  err << __FILE__ << " " << __LINE__ << " :: "
191  << " m_data.find(" << t << ") failed.";
192  throw lbann_exception(err.str());
193  }
194  if (m_data[t].size() != (size_t)m_vector_size) {
195  err << __FILE__ << " " << __LINE__ << " :: "
196  << " m_data[" << t << "].size = " << m_data[t].size()
197  << " should be: " << m_vector_size << "; " << jj+1
198  << " of " << indices.size()
199  << " m_reader->get_role: " << m_reader->get_role();
200  throw lbann_exception(err.str());
201  }
202  MPI_Isend(
203  m_data[t].data(), m_vector_size*sizeof(DataType), MPI_BYTE,
204  p, t, m_mpi_comm, &(send_req[p][jj++]));
205  }
206  }
207 
208  //wait for sends to finish
209  if (m_master) {
210  for (size_t i=0; i<send_req.size(); i++) {
211  MPI_Waitall(send_req[i].size(), send_req[i].data(), send_status[i].data());
212  }
213  }
214 
215  //wait for recvs to finish
216  MPI_Waitall(recv_req.size(), recv_req.data(), recv_status.data());
217 
218  if (m_master) {
219  std::cout << "role: " << m_reader->get_role() << " data_store_csv::exchange_data() time: " << get_time() - tm1 << std::endl;
220  }
221 }
