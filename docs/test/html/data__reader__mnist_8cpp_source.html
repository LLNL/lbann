51 bool mnist_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
52  int pixelcount = m_image_width * m_image_height;
53  std::vector<unsigned char>& tmp = m_image_data[data_id];
54 
55  for (int p = 0; p < pixelcount; p++) {
56  X.Set(p, mb_idx, tmp[p+1]);
57  }
58 
59  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
60  augment(pixel_col, m_image_height, m_image_width, 1);
61  normalize(pixel_col, 1);
62  pixel_noise(pixel_col); //add noise to image, disable by default
63  return true;
64 }

74 void load_mnist_data(const std::string imagepath, const std::string labelpath,
75  const int m_first_n, std::vector<std::vector<unsigned char> >& m_image_data) {
76 
77  // read labels
78  FILE *fplbl = fopen(labelpath.c_str(), "rb");
79  if (!fplbl) {
80  throw lbann_exception(
81  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
82  " :: load_mnist_data: failed to open file: " + labelpath);
83  }
84 
85  int magicnum1, numitems1;
86  fread(&magicnum1, 4, 1, fplbl);
87  fread(&numitems1, 4, 1, fplbl);
88  __swapEndianInt((unsigned int&)magicnum1);
89  __swapEndianInt((unsigned int&)numitems1);
90 
91  // read images
92  FILE *fpimg = fopen(imagepath.c_str(), "rb");
93  if (!fpimg) {
94  throw lbann_exception(
95  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
96  " :: load_mnist_data: failed to open file: " + imagepath);
97  }
98 
99  int magicnum2, numitems2, imgwidth, imgheight;
100  fread(&magicnum2, 4, 1, fpimg);
101  fread(&numitems2, 4, 1, fpimg);
102  fread(&imgwidth, 4, 1, fpimg);
103  fread(&imgheight, 4, 1, fpimg);
104  __swapEndianInt((unsigned int&)magicnum2);
105  __swapEndianInt((unsigned int&)numitems2);
106  __swapEndianInt((unsigned int&)imgwidth);
107  __swapEndianInt((unsigned int&)imgheight);
108 
109  if (numitems1 != numitems2) {
110  fclose(fplbl);
111  fclose(fpimg);
112  throw lbann_exception(
113  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
114  " :: load_mnist_data: numitems1 != numitems2");
115  }
116 
117  if (m_first_n > 0) {
118  numitems1 = m_first_n > numitems1 ? numitems1 : m_first_n;
119  }
120 
121  // set to array
122  m_image_data.resize(numitems1);
123  for (int n = 0; n < numitems1; n++) {
124  m_image_data[n].resize(1+(imgwidth * imgheight));
125  fread(&m_image_data[n][0], 1, 1, fplbl);
126  fread(&m_image_data[n][1], imgwidth * imgheight, 1, fpimg);
127  }
128  fclose(fpimg);
129  fclose(fplbl);
130 }

132 void mnist_reader::load() {
133  if (is_master()) {
134  std::cerr << "starting lbann::mnist_reader::load\n";
135  }
136  m_image_data.clear();
137 
138  const std::string FileDir = get_file_dir();
139  const std::string ImageFile = get_data_filename();
140  const std::string LabelFile = get_label_filename();
141 
142  // set filepath
143  const std::string imagepath = FileDir + "/" + ImageFile;
144  const std::string labelpath = FileDir + "/" + LabelFile;
145 
146  if (is_master()) {
147  std::cerr << "read labels!\n";
148  }
149 
150  load_mnist_data(imagepath, labelpath, m_first_n, m_image_data);
151 
152  if (m_first_n > 0) {
153  set_use_percent(1.0);
154  set_absolute_sample_count(0u);
155  }
156 
157  // reset indices
158  m_shuffled_indices.clear();
159  m_shuffled_indices.resize(m_image_data.size());
160  for (size_t n = 0; n < m_shuffled_indices.size(); n++) {
161  m_shuffled_indices[n] = n;
162  }
163  if (is_master()) {
164  std::cerr << "calling select_subset_of_data; m_shuffled_indices.size: " <<
165  m_shuffled_indices.size() << std::endl;
166  }
167  select_subset_of_data();
168 }
