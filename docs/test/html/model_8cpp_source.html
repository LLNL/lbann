221 void model::replace_weights(std::vector<weights*>& new_weights) {
222 
223  // Check that number of weights is valid
224  if (new_weights.size() > m_weights.size()) {
225  std::stringstream err;
226  err << __FILE__ << " " << __LINE__ << " :: "
227  << "attempted to replace weights with an invalid number of weights "
228  << "(expected at most " << m_weights.size() << ", found " << new_weights.size() << ")";
229  throw lbann_exception(err.str());
230  }
231 
232  // Replace weights in list
233  std::vector<weights *> old_weights(m_weights.begin(),
234  m_weights.begin() + new_weights.size());
235  std::unordered_map<weights *,weights *> weights_map;
236  std::unordered_map<Layer *,Layer *> layer_map;
237  for (size_t i = 0; i < new_weights.size(); ++i) {
238  m_weights[i] = weights_map[old_weights[i]] = new_weights[i];
239  }
240  remap_pointers(layer_map, weights_map);
241 
242  // Delete old weights
243  for (const auto& w : old_weights) {
244  delete w;
245  }

312 void model::remap_pointers(const std::unordered_map<Layer *,Layer *>& layer_map,
313  const std::unordered_map<weights *,weights *>& weights_map) {
314 
315  // Fix pointers in objective function
316  if (m_objective_function != nullptr) {
317  auto layer_pointers = m_objective_function->get_layer_pointers();
318  for (auto& layer_pointer : layer_pointers) {
319  if (layer_map.count(layer_pointer) > 0) {
320  layer_pointer = layer_map.at(layer_pointer);
321  }
322  }
323  m_objective_function->set_layer_pointers(layer_pointers);
324  auto weights_pointers = m_objective_function->get_weights_pointers();
325  for (auto& weights_pointer : weights_pointers) {
326  if (weights_map.count(weights_pointer) > 0) {
327  weights_pointer = weights_map.at(weights_pointer);
328  }
329  }
330  m_objective_function->set_weights_pointers(weights_pointers);
331  }
332 
333  // Fix pointers in metrics
334  for (const auto& m : m_metrics) {
335  auto layer_pointers = m->get_layer_pointers();
336  for (auto& layer_pointer : layer_pointers) {
337  if (layer_map.count(layer_pointer) > 0) {
338  layer_pointer = layer_map.at(layer_pointer);
339  }
340  }
341  m->set_layer_pointers(layer_pointers);
342  }
343 
344  // Fix pointers in layers
345  for (const auto& l : m_layers) {
346  auto layer_pointers = l->get_layer_pointers();
347  for (auto& layer_pointer : layer_pointers) {
348  if (layer_map.count(layer_pointer) > 0) {
349  layer_pointer = layer_map.at(layer_pointer);
350  }
351  }
352  l->set_layer_pointers(layer_pointers);
353  auto weights_pointers = l->get_weights();
354  for (auto& weights_pointer : weights_pointers) {
355  if (weights_map.count(weights_pointer) > 0) {
356  weights_pointer = weights_map.at(weights_pointer);
357  }
358  }
359  l->set_weights(weights_pointers);
360  }
361 
362 }

468 void model::setup_weights() {
469 
470  // List of used and unused weights
471  std::unordered_set<weights *> weights_set(m_weights.begin(),
472  m_weights.end());
473  std::set<weights *> unused_weights(m_weights.begin(),
474  m_weights.end());
475 
476  // Find weights used by layers
477  for (const auto& layer : m_layers) {
478  for (const auto& w : layer->get_weights()) {
479  if (weights_set.count(w) == 0) {
480  m_weights.push_back(w);
481  weights_set.insert(w);
482  }
483  unused_weights.erase(w);
484  }
485  }
486 
487  // Find weights used by objective function
488  for (const auto& w : m_objective_function->get_weights_pointers()) {
489  if (weights_set.count(w) == 0) {
490  m_weights.push_back(w);
491  weights_set.insert(w);
492  }
493  unused_weights.erase(w);
494  }
495 
496  // Delete unused weights
497  for (const auto& w : unused_weights) {
498  m_weights.erase(std::remove(m_weights.begin(), m_weights.end(), w),
499  m_weights.end());
500  }
501 
502 }

504 void model::add_connected_layers() {
505 
506  // Initialize breadth-first search queue with layer list
507  std::queue<const Layer *> layer_queue;
508  std::unordered_set<const Layer *> layer_set;
509  for (const auto& layer : m_layers) {
510  layer_queue.push(layer);
511  layer_set.insert(layer);
512  }
513 
514  // Visit nodes in search queue until it is exhausted
515  while (!layer_queue.empty()) {
516  const Layer *layer = layer_queue.front();
517  layer_queue.pop();
518 
519  // Find neighbors of current node
520  std::vector<const Layer *> relatives;
521  for (const auto& parent : layer->get_parent_layers()) {
522  relatives.push_back(parent);
523  }
524  for (const auto& child : layer->get_child_layers()) {
525  relatives.push_back(child);
526  }
527 
528  // Add neighbors to search queue if they aren't in the layer list
529  for (const auto& relative : relatives) {
530  if (layer_set.count(relative) == 0) {
531  m_layers.push_back(const_cast<Layer *>(relative));
532  layer_queue.push(relative);
533  layer_set.insert(relative);
534  }
535  }
536 
537  }

541 void model::add_dummy_layers() {
542  for (size_t i = 0; i < m_layers.size(); ++i) {
543  auto layer = m_layers[i];
544 
545  // Create dummy layers until current layer has enough children
546  std::vector<Layer*> dummy_layers;
547  while (layer->get_num_children() < layer->get_expected_num_child_layers()) {
548  Layer *dummy = nullptr;
549  auto&& cudnn = layer->get_cudnn_manager();
550  switch (layer->get_data_layout()) {
551  case data_layout::DATA_PARALLEL:
552  dummy = new dummy_layer<data_layout::DATA_PARALLEL>(m_comm, cudnn);
553  break;
554  case data_layout::MODEL_PARALLEL:
555  dummy = new dummy_layer<data_layout::MODEL_PARALLEL>(m_comm, cudnn);
556  break;
557  default:
558  std::stringstream err;
559  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
560  throw lbann_exception(err.str());
561  }
562  dummy->set_name(layer->get_name()
563  + "_dummy"
564  + std::to_string(dummy_layers.size()));
565  layer->add_child_layer(dummy);
566  dummy->add_parent_layer(layer);
567  dummy_layers.push_back(dummy);
568  }
569 
570  // Add dummy layers to layer list
571  m_layers.insert(m_layers.begin() + i + 1,
572  dummy_layers.begin(),
573  dummy_layers.end());
574 
575  }
576 }

578 void model::add_split_layers() {
579  for (size_t i = 0; i < m_layers.size(); ++i) {
580  auto layer = m_layers[i];
581 
582  // Add split layer if layer expects one child but has multiple
583  auto& children = layer->get_child_layers();
584  if (layer->get_expected_num_child_layers() == 1
585  && children.size() != 1) {
586 
587  // Create split layer
588  Layer *split;
589  auto&& cudnn = layer->get_cudnn_manager();
590  switch (layer->get_data_layout()) {
591  case data_layout::DATA_PARALLEL:
592  split = new split_layer<data_layout::DATA_PARALLEL>(m_comm, cudnn);
593  break;
594  case data_layout::MODEL_PARALLEL:
595  split = new split_layer<data_layout::MODEL_PARALLEL>(m_comm, cudnn);
596  break;
597  default:
598  std::stringstream err;
599  err << __FILE__ << " " << __LINE__ << " :: " << "invalid data layout";
600  throw lbann_exception(err.str());
601  }
602  split->set_name(layer->get_name() + "_split");
603 
604  // Setup relationships between split layer and child layers
605  for (auto&& const_child : children) {
606  Layer *child = const_cast<Layer*>(const_child);
607  split->add_child_layer(child);
608  auto& child_parents = child->get_parent_layers();
609  std::replace(child_parents.begin(), child_parents.end(),
610  layer, split);
611  }
612 
613  // Setup relationship between current layer and split layer
614  children.clear();
615  layer->add_child_layer(split);
616  split->add_parent_layer(layer);
617 
618  // Add split layer to layer list
619  m_layers.insert(m_layers.begin() + i + 1, split);
620 
621  }
622 
623  }

679 void model::train(int num_epochs) {

694  do_epoch_end_cbs();
695  reset_epoch_statistics(execution_mode::training);
696

698  evaluate(execution_mode::validation);
699  }

744 bool model::train_mini_batch() {
745  reset_mode_and_model(execution_mode::training);
746  do_batch_begin_cbs(execution_mode::training);
747 
748  // Forward prop step
749  clear_gradients();
750  forward_prop(execution_mode::training);
751  // Result is not needed until the end of the mini-batch.
752  m_objective_function->start_evaluation(execution_mode::training,
753  get_current_mini_batch_size());
754  for (const auto& m : m_metrics) {
755  m->evaluate(execution_mode::training,
756  get_current_mini_batch_size());
757  }
758 
759  // Backward prop step
760  clear_error_signals();
761  m_objective_function->differentiate();
762  backward_prop();
763  m_objective_function->compute_weight_regularization();
764 
765  // Finish evaluation.
766  m_objective_function->finish_evaluation(execution_mode::training,
767  get_current_mini_batch_size());
768 
769  // Update step
770  update_weights();
771  const bool finished = update_layers();
772 
773  ++m_current_step;
774  do_batch_end_cbs(execution_mode::training);
775  return finished;

801 void model::backward_prop() {
802  do_model_backward_prop_begin_cbs();
803  for (int l = m_layers.size() - 1; l >= 0; --l) {
804 
805  // Perform backward prop step on current layer
806  Layer *layer = m_layers[l];
807  do_layer_backward_prop_begin_cbs(layer);
808  layer->back_prop();
809  do_layer_backward_prop_end_cbs(layer);
810 
811  // Terminate early if all gradients have been computed
812  bool all_gradients_computed = true;
813  for (auto&& w : m_weights) {
814  auto&& opt = w->get_optimizer();
815  if (opt != nullptr && opt->get_num_gradient_sources() != 0) {
816  all_gradients_computed = false;
817  break;
818  }
819  }
820  if (all_gradients_computed) { break; }
821 
822  }
823  do_model_backward_prop_end_cbs();
824 }
