137 void weights::setup(std::vector<int> matrix_height_dims,
138  std::vector<int> matrix_width_dims,
139  El::Distribution col_dist,
140  El::Distribution row_dist) {
141 
142  if (m_values != nullptr) {
143  // Check that dimensions are unchanged if weights are already
144  // initialized
145  const El::DistData dist_data(*m_values);
146  if (m_matrix_height_dims == matrix_height_dims
147  && m_matrix_width_dims == matrix_width_dims
148  && dist_data.colDist == col_dist
149  && dist_data.rowDist == row_dist) {
150  return;
151  } else {
152  std::stringstream err;
153  err << __FILE__ << " " << __LINE__ << " :: "
154  << "attempted to setup " << m_name << " as a "
155  << get_dims_string(matrix_height_dims, matrix_width_dims) << " "
156  << "weights matrix with "
157  << "col_dist=" << col_dist << ", "
158  << "row_dist=" << row_dist << ", "
159  << "but it is already setup as a "
160  << get_dims_string(m_matrix_height_dims, m_matrix_width_dims) << " "
161  << "matrix with "
162  << "col_dist=" << dist_data.colDist << ", "
163  << "row_dist=" << dist_data.rowDist;
164  throw lbann_exception(err.str());
165  }
166  } else {
167  // Check that tensor dimensions are valid
168  bool dims_are_valid = true;
169  for (const auto& d : matrix_height_dims) {
170  if (d <= 0) { dims_are_valid = false; }
171  }
172  for (const auto& d : matrix_width_dims) {
173  if (d <= 0) { dims_are_valid = false; }
174  }
175  if (!dims_are_valid) {
176  std::stringstream err;
177  err << __FILE__ << " " << __LINE__ << " :: "
178  << "attempted to setup " << m_name << " as a "
179  << get_dims_string(matrix_height_dims, matrix_width_dims) << " "
180  << "weights matrix";
181  throw lbann_exception(err.str());
182  }
183  }
184 
185  // Initialize weights matrix
186  m_matrix_height_dims = matrix_height_dims;
187  m_matrix_width_dims = matrix_width_dims;
188  m_values = m_initializer->construct_matrix(get_matrix_height(),
189  get_matrix_width(),
190  col_dist,
191  row_dist);
192 
193  // Setup GPU objects
194  if (m_cudnn != nullptr) {
195  setup_gpu();
196  }
197 
198  // Setup optimizer
199  if (m_optimizer != nullptr) {
200  m_optimizer->setup(*this);
201  }
202 
203 }

205 void weights::setup_gpu() {
206  #ifndef LBANN_HAS_CUDNN
207  std::stringstream err;
208  err << __FILE__ << " " << __LINE__ << " :: " << "cuDNN not detected";
209  throw lbann_exception(err.str());
210  #else
211 
212  // Check that weights matrix is valid
213  if (m_values == nullptr) {
214  std::stringstream err;
215  err << __FILE__ << " " << __LINE__ << " :: "
216  << "attempted to setup GPU weights matrix "
217  << "before initializing CPU weights matrix";
218  throw lbann_exception(err.str());
219  }
220 
221  // Disable GPU if weights matrix is not STAR,STAR
223  const El::DistData dist_data(*m_values);
224  if (dist_data.colDist != El::STAR || dist_data.rowDist != El::STAR) {
225  m_cudnn = nullptr;
226  return;
227  }
228 
229  // Copy weights matrix to GPU
230  m_cudnn->allocate_on_gpus(m_values_d,
231  m_values->LocalHeight(),
232  m_values->LocalWidth());
233  m_cudnn->broadcast_to_gpus(m_values_d, m_values->LockedMatrix());
234 
235  #endif // LBANN_HAS_CUDNN
236 }

334 void weights::set_value(DataType value, std::vector<int> pos) {
335 
336  // Get tensor dimensions
337  const auto& dims = get_dims();
338 
339 #ifdef LBANN_DEBUG
340  // Check that tensor position is valid
341  bool pos_is_valid = true;
342  if (dims.size() != pos.size()) {
343  pos_is_valid = false;
344  } else {
345  for (size_t i = 0 ; i < dims.size(); ++i) {
346  if (pos[i] < 0 || pos[i] >= dims[i]) { pos_is_valid = false;}
347  }
348  }
349  if (!pos_is_valid) {
350  std::stringstream err;
351  err << __FILE__ << " " << __LINE__ << " :: "
352  << "attempted to set weight value at position (";
353  for (size_t i = 0 ; i < pos.size(); ++i) {
354  err << (i > 0 ? "x" : "") << pos[i];
355  }
356  err << ") in a tensor with dimensions ";
357  for (size_t i = 0 ; i < dims.size(); ++i) {
358  err << (i > 0 ? "x" : "") << dims[i];
359  }
360  throw lbann_exception(err.str());
361  }
362 #endif // LBANN_DEBUG
363 
364  // Get index of weight value and set
365  int index = 0;
366  for (size_t i = 0; i < dims.size(); ++i) {
367  index = index * dims[i] + pos[i];
368  }
369  set_value(value, index);
370 
371 }

373 void weights::set_value(DataType value, int row, int col) {
374 
375 #ifdef LBANN_DEBUG
376  {
377  // Check that matrix entry is valid
378  const auto& height = get_matrix_height();
379  const auto& width = get_matrix_width();
380  if (row < 0 || row >= height || col < 0 || col > width ) {
381  std::stringstream err;
382  err << __FILE__ << " " << __LINE__ << " :: "
383  << "attempted to set weights value at entry "
384  << "(" << row << "," << col << ") "
385  << "in a " << height << "x" << width << " matrix";
386  throw lbann_exception(err.str());
387  }
388  }
389 #endif // LBANN_DEBUG
390 
391  if (m_cudnn == nullptr) {
392  // Set value if it is local
393  if (m_values->IsLocal(row, col)) {
394  const El::Int local_row = m_values->LocalRow(row);
395  const El::Int local_col = m_values->LocalCol(col);
396  m_values->SetLocal(local_row, local_col, value);
397  }
398  } else {
399  // Set value on GPU
400  #ifdef LBANN_HAS_CUDNN
401  Mat cpu_value(1, 1);
402  cpu_value(0, 0) = value;
403  std::vector<DataType*> gpu_value = m_values_d;
404  const int height = get_matrix_height();
405  for (DataType*& pointer : gpu_value) {
406  pointer += row + col * height;
407  }
408  m_cudnn->broadcast_to_gpus(gpu_value, cpu_value);
409  #endif // LBANN_HAS_CUDNN
410  }

472 void weights::write_proto(lbann_data::WeightsData* proto) const {
473 
474  // Set proto properties
475  proto->Clear();
476  proto->set_name(m_name);
477  for (const auto& d : get_dims()) {
478  proto->mutable_shape()->add_dim(d);
479  }
480  proto->set_height(get_matrix_height());
481  proto->set_width(get_matrix_width());
482 
483  // Write weight values to prototext on world master process
484  CircMat values = *m_values;
485  values.SetRoot(0);
486  if (m_comm->am_world_master()) {
487  const auto& local_values = values.LockedMatrix();
488  const El::Int height = local_values.Height();
489  const El::Int width = local_values.Width();
491 
498  for (El::Int i = 0; i < height; ++i) {
499  for (El::Int j = 0; j < width; ++j) {
500  proto->add_data(local_values(i,j));
501  }
502  }
503  }
