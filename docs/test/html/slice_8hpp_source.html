207  void fp_compute_cpu() {
208 
209  // Input tensor
210  const auto& input = get_prev_activations();
211  const int width = input.Width();
212 
213  // Get number of contiguous regions in a tensor slice of width 1
214  const auto& input_dims = this->m_prev_neuron_dims;
215  const int num_regions = std::accumulate(input_dims.begin(),
216  input_dims.begin() + m_slice_axis,
217  1,
218  std::multiplies<int>());
219  const int unit_region_size = std::accumulate(input_dims.begin() + m_slice_axis + 1,
220  input_dims.end(),
221  1,
222  std::multiplies<int>());
223 
224  // Get stride between contiguous regions in input tensor slices
225  const int input_slice_dim = input_dims[m_slice_axis];
226  const int input_region_stride = input_slice_dim * unit_region_size;
227 
228  // Populate output tensors with slices of input tensor
229  for (int i = 0; i < get_num_children(); ++i) {
230  auto& output = get_activations(i);
231 
232  // Get stride between contiguous regions in output tensor slices
233  const int output_slice_dim = m_slice_points[i+1] - m_slice_points[i];
234  const int output_region_stride = output_slice_dim * unit_region_size;
235 
236  // Get position of first contiguous region in input tensor
237  const int input_region_start = m_slice_points[i] * unit_region_size;
238  const int input_region_end = m_slice_points[i+1] * unit_region_size;
239 
240  // Populate current output tensor
241  output.Resize(num_regions * output_region_stride, width);
242  if (num_regions == 1) {
243  El::LockedView(output,
244  input,
245  El::IR(input_region_start, input_region_end),
246  El::ALL);
247  } else {
248  for (int region = 0; region < num_regions; ++region) {
249  El::LockedView(*m_input_region_v,
250  input,
251  El::IR(input_region_start + region * input_region_stride,
252  input_region_end + region * input_region_stride),
253  El::ALL);
254  El::View(*m_output_region_v,
255  output,
256  El::IR(region * output_region_stride,
257  (region+1) * output_region_stride),
258  El::ALL);
259  El::Copy(*m_input_region_v, *m_output_region_v);
260  }
261  }
262  }

266  void bp_compute_cpu() {
267 
268  // Gradient w.r.t. input
269  auto& gradient_wrt_input = get_error_signals();
270 
271  // Get number of contiguous regions in a tensor slice of width 1
272  const auto& input_dims = this->m_prev_neuron_dims;
273  const int num_regions = std::accumulate(input_dims.begin(),
274  input_dims.begin() + m_slice_axis,
275  1,
276  std::multiplies<int>());
277  const int unit_region_size = std::accumulate(input_dims.begin() + m_slice_axis + 1,
278  input_dims.end(),
279  1,
280  std::multiplies<int>());
281 
282  // Get stride between contiguous regions in input tensor slices
283  const int input_slice_dim = input_dims[m_slice_axis];
284  const int input_region_stride = input_slice_dim * unit_region_size;
285 
286  // Populate gradient w.r.t. input with slices of gradient w.r.t. output
287  for (int i = 0; i < get_num_children(); ++i) {
288  const auto& gradient_wrt_output = get_prev_error_signals(i);
289 
290  // Get stride between contiguous regions in output tensor slices
291  const int output_slice_dim = m_slice_points[i+1] - m_slice_points[i];
292  const int output_region_stride = output_slice_dim * unit_region_size;
293 
294  // Get position of first contiguous region in input tensor
295  const int input_region_start = m_slice_points[i] * unit_region_size;
296  const int input_region_end = m_slice_points[i+1] * unit_region_size;
297 
298  // Populate slice of gradient w.r.t. input
299  for (int region = 0; region < num_regions; ++region) {
300  El::View(*m_input_region_v,
301  gradient_wrt_input,
302  El::IR(input_region_start + region * input_region_stride,
303  input_region_end + region * input_region_stride),
304  El::ALL);
305  El::LockedView(*m_output_region_v,
306  gradient_wrt_output,
307  El::IR(region * output_region_stride,
308  (region+1) * output_region_stride),
309  El::ALL);
310  El::Axpy(DataType(1), *m_output_region_v, *m_input_region_v);
311  }
312 
313  }

317  void fp_compute_gpu() {
318  #ifndef LBANN_HAS_CUDNN
319  std::stringstream err;
320  err << __FILE__ << " " << __LINE__ << " :: slice_layer: cuDNN not detected";
321  throw lbann_exception(err.str());
322  #else
323 
324  // Input tensor
325  const auto& input_d = m_prev_activations_d[0];
326 
327  // Get number of contiguous regions in a tensor slice of width 1
328  const auto& input_dims = this->m_prev_neuron_dims;
329  const int num_regions = std::accumulate(input_dims.begin(),
330  input_dims.begin() + m_slice_axis,
331  1,
332  std::multiplies<int>());
333  const int unit_region_size = std::accumulate(input_dims.begin() + m_slice_axis + 1,
334  input_dims.end(),
335  1,
336  std::multiplies<int>());
337 
338  // Get stride between contiguous regions in input tensor slices
339  const int input_slice_dim = input_dims[m_slice_axis];
340  const int input_region_stride = input_slice_dim * unit_region_size;
341 
342  // Populate output tensors with slices of input tensor
343  cudnn::matrix input_region_d(m_cudnn), output_region_d(m_cudnn);
344  for (int i = 0; i < get_num_children(); ++i) {
345  auto& output_d = m_activations_d[i];
346 
347  // Get stride between contiguous regions in output tensor slices
348  const int output_slice_dim = m_slice_points[i+1] - m_slice_points[i];
349  const int output_region_stride = output_slice_dim * unit_region_size;
350 
351  // Get position of first contiguous region in input tensor
352  const int input_region_start = m_slice_points[i] * unit_region_size;
353 
354  // Populate current output tensor
355  if (num_regions == 1) {
356  auto input_ptrs = input_d.get_locked_data();
357  for (auto& ptr : input_ptrs) { ptr += input_region_start; }
358  output_d.locked_attach(input_ptrs,
359  output_d.get_height(),
360  input_d.get_width_per_gpu(),
361  input_d.get_leading_dim());
362  } else {
363  for (int region = 0; region < num_regions; ++region) {
364  auto input_ptrs = input_d.get_locked_data();
365  auto output_ptrs = output_d.get_data();
366  for (auto& ptr : input_ptrs) {
367  ptr += input_region_start + region * input_region_stride;
368  }
369  for (auto& ptr : output_ptrs) {
370  ptr += region * output_region_stride;
371  }
372  input_region_d.locked_attach(input_ptrs,
373  output_region_stride,
374  this->m_mini_batch_size_per_gpu,
375  input_d.get_leading_dim());
376  output_region_d.attach(output_ptrs,
377  output_region_stride,
378  this->m_mini_batch_size_per_gpu,
379  output_d.get_leading_dim());
380  output_region_d.copy(input_region_d);
381  }
382  }
383  }
384 
385  #endif // LBANN_HAS_CUDNN

388  void bp_compute_gpu() {
389  #ifndef LBANN_HAS_CUDNN
390  std::stringstream err;
391  err << __FILE__ << " " << __LINE__ << " :: slice_layer: cuDNN not detected";
392  throw lbann_exception(err.str());
393  #else
394 
395  // Gradient w.r.t. input
396  auto& gradient_wrt_input_d = m_error_signals_d[0];
397 
398  // Get number of contiguous regions in a tensor slice of width 1
399  const auto& input_dims = this->m_prev_neuron_dims;
400  const int num_regions = std::accumulate(input_dims.begin(),
401  input_dims.begin() + m_slice_axis,
402  1,
403  std::multiplies<int>());
404  const int unit_region_size = std::accumulate(input_dims.begin() + m_slice_axis + 1,
405  input_dims.end(),
406  1,
407  std::multiplies<int>());
408 
409  // Get stride between contiguous regions in input tensor slices
410  const int input_slice_dim = input_dims[m_slice_axis];
411  const int input_region_stride = input_slice_dim * unit_region_size;
412 
413  // Populate gradient w.r.t. input with slices of gradient w.r.t. output
414  cudnn::matrix gradient_wrt_input_region_d(m_cudnn);
415  cudnn::matrix gradient_wrt_output_region_d(m_cudnn);
416  for (int i = 0; i < get_num_children(); ++i) {
417  const auto& gradient_wrt_output_d = m_prev_error_signals_d[i];
418 
419  // Get stride between contiguous regions in output tensor slices
420  const int output_slice_dim = m_slice_points[i+1] - m_slice_points[i];
421  const int output_region_stride = output_slice_dim * unit_region_size;
422 
423  // Get position of first contiguous region in input tensor
424  const int input_region_start = m_slice_points[i] * unit_region_size;
425 
426  // Populate slice of gradient w.r.t. input
427  for (int region = 0; region < num_regions; ++region) {
428  auto gradient_wrt_input_ptrs = gradient_wrt_input_d.get_data();
429  auto gradient_wrt_output_ptrs = gradient_wrt_output_d.get_locked_data();
430  for (auto& ptr : gradient_wrt_input_ptrs) {
431  ptr += input_region_start + region * input_region_stride;
432  }
433  for (auto& ptr : gradient_wrt_output_ptrs) {
434  ptr += region * output_region_stride;
435  }
436  gradient_wrt_input_region_d.attach(gradient_wrt_input_ptrs,
437  output_region_stride,
438  this->m_mini_batch_size_per_gpu,
439  gradient_wrt_input_d.get_leading_dim());
440  gradient_wrt_output_region_d.locked_attach(gradient_wrt_output_ptrs,
441  output_region_stride,
442  this->m_mini_batch_size_per_gpu,
443  gradient_wrt_output_d.get_leading_dim());
444  const int num_gpus = m_cudnn->get_num_gpus();
445  for (int gpu = 0; gpu < num_gpus; ++gpu) {
446  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(gpu)));
447  cublas::geam(this->m_cudnn->get_cublas_handle(gpu),
448  CUBLAS_OP_N, CUBLAS_OP_N,
449  gradient_wrt_output_region_d.get_height(),
450  this->m_mini_batch_size_per_gpu,
451  DataType(1),
452  gradient_wrt_output_region_d.get_locked_data(gpu),
453  gradient_wrt_output_region_d.get_leading_dim(),
454  DataType(1),
455  gradient_wrt_input_region_d.get_locked_data(gpu),
456  gradient_wrt_input_region_d.get_leading_dim(),
457  gradient_wrt_input_region_d.get_data(gpu),
458  gradient_wrt_input_region_d.get_leading_dim());
459  }
460  }

462  }
