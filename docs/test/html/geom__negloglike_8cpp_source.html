31 EvalType geom_negloglike::finish_evaluate_compute(
32  const AbsDistMat& predictions, const AbsDistMat& ground_truth) {
33 
34  // Local matrices
35  const Mat& predictions_local = predictions.LockedMatrix();
36  const Mat& ground_truth_local = ground_truth.LockedMatrix();
37 
38  // Matrix parameters
39  const int width = predictions.Width();
40  const int local_height = predictions_local.Height();
41  const int local_width = predictions_local.Width();
42 
43  // Compute sum of terms
44  EvalType sum = EvalType(0);
45 #pragma omp parallel for reduction(+:sum) collapse(2)
46  for (int col = 0; col < local_width; ++col) {
47  for (int row = 0; row < local_height; ++row) {
48  const EvalType true_val = ground_truth_local(row, col);
49  const EvalType pred_val = predictions_local(row, col);
50  sum += (- true_val * std::log(EvalType(1) - pred_val)
51  - std::log(pred_val));
52  }
53  }
54 
55  // Compute mean objective function value across mini-batch
56  return get_comm().allreduce(sum / width, predictions.DistComm());
57 
58 }
