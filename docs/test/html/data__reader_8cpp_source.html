63 int lbann::generic_data_reader::fetch_data(Mat& X) {
64  int nthreads = omp_get_max_threads();
65  if(!position_valid()) {
66  throw lbann_exception(
67  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
68  + " :: generic data reader load error: !position_valid"
69  + " -- current pos = " + std::to_string(m_current_pos)
70  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
71  }
72 
73  if (!m_save_minibatch_indices) {
76  #pragma omp parallel for schedule(static, 1)
77  for (int t = 0; t < nthreads; t++) {
78  preprocess_data_source(omp_get_thread_num());
79  }
80  }
81 
82  int loaded_batch_size = get_loaded_mini_batch_size();
83  const int end_pos = std::min(static_cast<size_t>(m_current_pos+loaded_batch_size),
84  m_shuffled_indices.size());
85  const int mb_size = std::min(
86  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
87  X.Width());
88 
89  if (!m_save_minibatch_indices) {
90  El::Zeros(X, X.Height(), X.Width());
91  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
92  }
93 
94  if (m_save_minibatch_indices) {
95  m_my_minibatch_indices.resize(m_my_minibatch_indices.size() + 1);
96  for (int s = 0; s < mb_size; s++) {
97  int n = m_current_pos + (s * m_sample_stride);
98  m_my_minibatch_indices.back().push_back(n);
99  }
100  }
101 
102  else {
103  #pragma omp parallel for
104  for (int s = 0; s < mb_size; s++) {
105  // Catch exceptions within the OpenMP thread.
106  try {
107  int n = m_current_pos + (s * m_sample_stride);
108  int index = m_shuffled_indices[n];
109  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
110  if (!valid) {
111  throw lbann_exception(
112  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
113  " :: generic data reader load error: datum not valid");
114  }
115  m_indices_fetched_per_mb.Set(s, 0, index);
116  } catch (lbann_exception& e) {
117  lbann_report_exception(e);
118  } catch (std::exception& e) {
119  El::ReportException(e);
120  }
121  }
122 
125  #pragma omp parallel for schedule(static, 1)
126  for (int t = 0; t < nthreads; t++) {
127  postprocess_data_source(omp_get_thread_num());
128  }
129  }
130 
131  return mb_size;
132 }

226 bool generic_data_reader::update(bool is_active_reader) {
227  bool reader_not_done = true; // BVE The sense of this should be fixed
228  m_current_mini_batch_idx++;
229 
230  if(is_active_reader) {
231  m_current_pos = get_next_position();
232 
234  if (!m_save_minibatch_indices) {
235  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Height(), 1);
236  }
237 
238  m_loaded_mini_batch_idx += m_iteration_stride;
239  }
240  if (m_loaded_mini_batch_idx >= m_num_iterations_per_epoch) {
241  reader_not_done = false;
242  }
243  if ((size_t)m_current_pos >= m_shuffled_indices.size()) {
244  reader_not_done = false;
245  }
246  if (m_current_mini_batch_idx == m_num_iterations_per_epoch) {
247  if ((get_rank() < m_num_parallel_readers) && (m_current_pos < (int)m_shuffled_indices.size())) {
248  throw lbann_exception(
249  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
250  + " :: generic data reader update error: the epoch is complete,"
251  + " but not all of the data has been used -- current pos = " + std::to_string(m_current_pos)
252  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices"
253  + " : iteration="
254  + std::to_string(m_current_mini_batch_idx) + "C ["
255  + std::to_string(m_loaded_mini_batch_idx) +"L] of "
256  + std::to_string(m_num_iterations_per_epoch) + "+"
257  + std::to_string(m_iteration_stride) + " : "
258  + " index stride="
259  + std::to_string(m_stride_to_next_mini_batch) + "/"
260  + std::to_string(m_stride_to_last_mini_batch));
261  }
262 
263  if (!m_save_minibatch_indices) {
264  if (m_shuffle) {
265  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
266  get_data_seq_generator());
267  }
268  }
269 
270  set_initial_position();
271 
272  if (!m_save_minibatch_indices) {
273  if (m_data_store) {
274  m_data_store->set_shuffled_indices(&m_shuffled_indices);
275  }
276  }
277  }
278  return reader_not_done;
279 }

328 void generic_data_reader::select_subset_of_data() {
329  m_num_global_indices = m_shuffled_indices.size();
330  shuffle_indices();
331 
332  size_t count = get_absolute_sample_count();
333  double use_percent = get_use_percent();
334  if (count == 0 and use_percent == 0.0) {
335  throw lbann_exception(
336  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
337  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
338  + "and get_absolute_sample_count() are both zero; exactly one "
339  + "must be zero");
340  }
341  if (!(count == 0 or use_percent == 0.0)) {
342  throw lbann_exception(
343  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
344  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
345  "and get_absolute_sample_count() are both non-zero; exactly one "
346  "must be zero");
347  }
348 
349 
350  if (count != 0) {
351  if(count > static_cast<size_t>(get_num_data())) {
352  throw lbann_exception(
353  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
354  " :: generic_data_reader::select_subset_of_data() - absolute_sample_count=" +
355  std::to_string(count) + " is > get_num_data=" +
356  std::to_string(get_num_data()));
357  }
358  m_shuffled_indices.resize(get_absolute_sample_count());
359  }
360 
361  if (use_percent) {
362  m_shuffled_indices.resize(get_use_percent()*get_num_data());
363  }
364 
365  long unused = get_validation_percent()*get_num_data(); //get_num_data() = m_shuffled_indices.size()
366  long use_me = get_num_data() - unused;
367  if (unused > 0) {
368  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
369  m_shuffled_indices.resize(use_me);
370  }
371 
372  if(!m_shuffle) {
373  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
374  std::sort(m_unused_indices.begin(), m_unused_indices.end());
375  }
376 }

394 bool lbann::generic_data_reader::load_from_checkpoint_shared(persist& p, const char *name) {
395  // rank 0 reads the training state file
396  struct packing_header header;
397  if (p.get_rank() == 0) {
398  unpack_scalars(p,&header,name);
399  }
400  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
401  unpack_header(header);
402  if(p.get_rank() ==0){
403  m_shuffled_indices.resize(header.data_size);
404  m_unused_indices.resize(header.unused_data_size);
405  }
406  MPI_Bcast(&m_shuffled_indices[0], header.data_size, MPI_INT, 0, MPI_COMM_WORLD);
407  MPI_Bcast(&m_unused_indices[0], header.unused_data_size, MPI_INT, 0, MPI_COMM_WORLD);
408  return true;
409 }
