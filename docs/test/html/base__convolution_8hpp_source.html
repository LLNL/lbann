217  void setup_data() override {
218  learning_layer::setup_data();
219 
220  // Initialize default weights if none are provided
221  if (this->m_weights.size() > 2) {
222  std::stringstream err;
223  err << __FILE__ << " " << __LINE__ << " :: "
224  << "attempted to setup " << m_name << " with an invalid number of weights";
225  throw lbann_exception(err.str());
226  }
227  this->m_weights.resize(2, nullptr);
228  if (this->m_weights[0] == nullptr) {
229  this->m_weights[0] = new weights(this->m_comm, this->m_cudnn);
230  this->m_weights[0]->set_name(this->m_name + "_kernel");
231  this->m_weights[0]->set_initializer(new he_normal_initializer(this->m_comm));
232  this->m_weights[0]->set_optimizer(m_model->create_optimizer());
233  this->m_model->add_weights(this->m_weights[0]);
234  }
235  if (this->m_weights[1] == nullptr) {
236  this->m_weights[1] = new weights(this->m_comm, this->m_cudnn);
237  this->m_weights[1]->set_name(this->m_name + "_bias");
238  this->m_weights[1]->set_initializer(new constant_initializer(this->m_comm, DataType(0)));
239  this->m_weights[1]->set_optimizer(m_model->create_optimizer());
240  this->m_model->add_weights(this->m_weights[1]);
241  }
242 
243  // Initialize Glorot or He weight initialization
244  auto* cast_initializer
245  = dynamic_cast<fan_in_fan_out_initializer*>(&this->m_weights[0]->get_initializer());
246  if (cast_initializer != nullptr) {
247  cast_initializer->set_fan_in(m_kernel_size / this->m_neuron_dims[0]);
248  cast_initializer->set_fan_out(m_kernel_size / this->m_prev_neuron_dims[0]);
249  }
250 
251  // Initialize bias
252  this->m_weights[1]->setup(this->m_neuron_dims[0]);
253  El::Zeros(m_bias_gradient,
254  this->m_weights[1]->get_matrix_height(),
255  this->m_weights[1]->get_matrix_width());
256 
257  if (m_frozen) {
258  this->m_weights[0]->freeze();
259  this->m_weights[1]->freeze();
260  } else {
261  if (this->m_weights[0]->is_frozen() || this->m_weights[1]->is_frozen()) {
262  throw lbann_exception("base_convolution_layer: layer is not frozen but weights are");
263  }
264  }
265  }

268  void setup_gpu() override {
269  learning_layer::setup_gpu();
270  #ifndef LBANN_HAS_CUDNN
271  throw lbann_exception("base_convolution_layer: cuDNN not detected");
272  #else
273 
274  // Set kernel descriptor
275  CHECK_CUDNN(cudnnCreateFilterDescriptor(&m_kernel_cudnn_desc));
276  CHECK_CUDNN(cudnnSetFilterNdDescriptor(m_kernel_cudnn_desc,
277  cudnn::get_cudnn_data_type(),
278  CUDNN_TENSOR_NCHW,
279  m_kernel_dims.size(),
280  m_kernel_dims.data()));
281 
282  // Set convolution descriptor
283  // Note: upscales are not supported as of cuDNN v5.1
284  CHECK_CUDNN(cudnnCreateConvolutionDescriptor(&m_convolution_cudnn_desc));
285  std::vector<int> upscales(this->m_num_neuron_dims-1, 1);
286  CHECK_CUDNN(cudnnSetConvolutionNdDescriptor(m_convolution_cudnn_desc,
287  m_pads.size(),
288  m_pads.data(),
289  m_strides.data(),
290  upscales.data(),
291  CUDNN_CROSS_CORRELATION,
292  cudnn::get_cudnn_data_type()));
293 
294  // Set bias tensor descriptor
295  std::vector<int> bias_dims(this->m_num_neuron_dims, 1);
296  bias_dims[0] = this->m_neuron_dims[0];
297  cudnn::set_tensor_cudnn_desc(m_bias_cudnn_desc, 1, bias_dims);
298 
299  // Allocate GPU memory
300  m_kernel_gradient_d = cudnn::matrix(m_cudnn,
301  m_kernel_gradient.Height(),
302  m_kernel_gradient.Width());
303  if (m_bias_scaling_factor != DataType(0)) {
304  m_bias_gradient_d = cudnn::matrix(m_cudnn,
305  m_bias_gradient.Height(),
306  m_bias_gradient.Width());
307  }
308 
309  #endif // LBANN_HAS_CUDNN

312  virtual void check_setup() override {
313  learning_layer::check_setup();
314  std::stringstream err;
315 
316  // Check that kernel and bias weights are both initialized
317  if (this->m_weights.size() != 2
318  || this->m_weights[0] == nullptr
319  || this->m_weights[1] == nullptr) {
320  err << __FILE__ << " " << __LINE__ << " :: "
321  << "invalid weights setup in layer " << m_name;
322  throw lbann_exception(err.str());
323  }
324 
325  // Check that kernel data is contiguous
326  const auto& kernel = this->m_weights[0]->get_values();
327  if (kernel.LocalWidth() > 1
328  && kernel.LDim() != kernel.LocalHeight()) {
329  err << __FILE__ << " " << __LINE__ << " :: "
330  << "kernel data in layer " << m_name << " "
331  << "is not contiguous";
332  throw lbann_exception(err.str());
333  }
334 
335  // Check that kernel gradient data is contiguous
336  if (m_kernel_gradient.LocalWidth() > 1
337  && m_kernel_gradient.LDim() != m_kernel_gradient.LocalHeight()) {
338  err << __FILE__ << " " << __LINE__ << " :: "
339  << "kernel gradient data in layer " << m_name << " "
340  << "is not contiguous";
341  throw lbann_exception(err.str());
342  }

349  void apply_convolution_cudnn(bool during_forward_prop) {
350  #ifndef LBANN_HAS_CUDNN
351  throw lbann_exception("base_convolution_layer: cuDNN not detected");
352  #else
353 
354  // Useful constants
355  const DataType one = 1;
356 
357  // GPU data
358  const auto& kernel_d = m_weights[0]->get_values_gpu();
359  const auto& input_d = (during_forward_prop ?
360  m_prev_activations_d[0] :
361  m_prev_error_signals_d[0]);
362  auto& output_d = (during_forward_prop ?
363  m_activations_d[0] :
364  m_error_signals_d[0]);
365  auto&& work_spaces_d = m_cudnn->get_work_spaces();
366 
367  // Convolution parameters
368  DataType mixing_factor;
369  cudnnTensorDescriptor_t input_cudnn_desc, output_cudnn_desc;
370  if (during_forward_prop) {
371  mixing_factor = DataType(0);
372  input_cudnn_desc = this->m_prev_activations_cudnn_desc;
373  output_cudnn_desc = this->m_activations_cudnn_desc;
374  }
375  else {
376  mixing_factor = DataType(1);
377  input_cudnn_desc = this->m_prev_error_signals_cudnn_desc;
378  output_cudnn_desc = this->m_error_signals_cudnn_desc;
379  }
380 
381  // Perform convolution on each GPU
382  const int num_gpus = this->m_cudnn->get_num_gpus();
383  for (int i = 0; i < num_gpus; ++i) {
384 
385  // Determine convolution algorithm
386  const size_t work_space_size = this->m_cudnn->get_work_space_size(i);
387  cudnnConvolutionFwdAlgo_t convolution_cudnn_algorithm
388  = CUDNN_CONVOLUTION_FWD_ALGO_IMPLICIT_GEMM;
389  CHECK_CUDNN(cudnnGetConvolutionForwardAlgorithm(this->m_cudnn->get_handle(i),
390  input_cudnn_desc,
391  m_kernel_cudnn_desc,
392  m_convolution_cudnn_desc,
393  output_cudnn_desc,
394  CUDNN_CONVOLUTION_FWD_SPECIFY_WORKSPACE_LIMIT,
395  work_space_size,
396  &convolution_cudnn_algorithm));
397 
398  // Apply convolution
399  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
400  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
401  this->m_cudnn->get_stream(i)));
402  CHECK_CUDNN(cudnnConvolutionForward(this->m_cudnn->get_handle(i),
403  &one,
404  input_cudnn_desc,
405  input_d.get_locked_data(i),
406  m_kernel_cudnn_desc,
407  kernel_d[i],
408  m_convolution_cudnn_desc,
409  convolution_cudnn_algorithm,
410  work_spaces_d[i],
411  work_space_size,
412  &mixing_factor,
413  output_cudnn_desc,
414  output_d.get_data(i)));
415 
416  }
417 
418  #endif // LBANN_HAS_CUDNN
419  }

519  void compute_gradients_cudnn(bool using_transposed_convolution) {
520  #ifndef LBANN_HAS_CUDNN
521  throw lbann_exception("base_convolution_layer: cuDNN not detected");
522  #else
523 
524  // Useful constants
525  const DataType zero = DataType(0);
526  const DataType one = DataType(1);
527  const int num_gpus = this->m_cudnn->get_num_gpus();
528  const int effective_mini_batch_size = this->m_model->get_effective_mini_batch_size();
529 
530  const auto& input_d = this->m_prev_activations_d[0];
531  const auto& gradient_wrt_output_d = this->m_prev_error_signals_d[0];
532 
533  // Compute bias gradient
534  optimizer* bias_optimizer = m_weights[1]->get_optimizer();
535  if (bias_optimizer != nullptr && m_bias_scaling_factor != DataType(0)) {
536  for (int i = 0; i < num_gpus; ++i) {
537  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
538  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
539  this->m_cudnn->get_stream(i)));
540  CHECK_CUDNN(cudnnConvolutionBackwardBias(this->m_cudnn->get_handle(i),
541  &one,
542  this->m_prev_error_signals_cudnn_desc,
543  gradient_wrt_output_d.get_locked_data(i),
544  &zero,
545  m_bias_cudnn_desc,
546  m_bias_gradient_d.get_data(i)));
547  }
548  const DataType bias_scale = m_bias_scaling_factor / effective_mini_batch_size;
549  bias_optimizer->add_to_gradient_staging(m_bias_gradient_d,
550  bias_scale);
551  }
552 
553  // Compute kernel gradient
554  optimizer* kernel_optimizer = m_weights[0]->get_optimizer();
555  if (kernel_optimizer != nullptr) {
556  auto&& work_spaces_d = this->m_cudnn->get_work_spaces();
557  for (int i = 0; i < num_gpus; ++i) {
558  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
559  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
560  this->m_cudnn->get_stream(i)));
561 
562  // Determine algorithm and compute kernel gradient
563  const size_t work_space_size = this->m_cudnn->get_work_space_size(i);
564  cudnnConvolutionBwdFilterAlgo_t kernel_gradient_cudnn_algorithm
565  = CUDNN_CONVOLUTION_BWD_FILTER_ALGO_0;
566  if (using_transposed_convolution) {
567  CHECK_CUDNN(cudnnGetConvolutionBackwardFilterAlgorithm(this->m_cudnn->get_handle(i),
568  this->m_prev_error_signals_cudnn_desc,
569  this->m_prev_activations_cudnn_desc,
570  m_convolution_cudnn_desc,
571  m_kernel_cudnn_desc,
572  CUDNN_CONVOLUTION_BWD_FILTER_SPECIFY_WORKSPACE_LIMIT,
573  work_space_size,
574  &kernel_gradient_cudnn_algorithm));
575  CHECK_CUDNN(cudnnConvolutionBackwardFilter(this->m_cudnn->get_handle(i),
576  &one,
577  this->m_prev_error_signals_cudnn_desc,
578  gradient_wrt_output_d.get_locked_data(i),
579  this->m_prev_activations_cudnn_desc,

630  get_local_prev_activations() :

633  get_local_activations() :

640  std::vector<int> input_dims, output_dims;
641  if (during_forward_prop) {
642  mixing_factor = DataType(0);

649  output_dims = this->m_prev_neuron_dims;
650  }

654  const int n = output_dims[0];
655  const int k = m_kernel_size / output_dims[0];

659

661  for (El::Int col = 0; col < local_width; ++col) {
662 
663  // Construct im2col matrix from current input column
664  El::LockedView(input_col, local_input, El::ALL, El::IR(col));

750  void apply_bias_cpu() {
751 
752  // Return immediately if there is no bias
753  if (m_bias_scaling_factor == DataType(0)) return;
754 
755  // Local matrices
756  const auto& local_bias = m_weights[1]->get_values().LockedMatrix();
757  auto& local_output = get_local_activations();
758 
759  // Matrix parameters
760  const El::Int local_width = local_output.Width();
761  const El::Int num_output_channels = this->m_neuron_dims[0];
762  const El::Int num_per_output_channel = this->m_num_neurons / num_output_channels;
763 
764  // Apply bias to each output channel
765  #pragma omp parallel for
766  for (El::Int channel = 0; channel < num_output_channels; ++channel) {
767  const El::Int row_start = channel * num_per_output_channel;
768  const El::Int row_end = (channel+1) * num_per_output_channel;
769  const DataType bias_term = m_bias_scaling_factor * local_bias(channel, 0);
770  for (El::Int col = 0; col < local_width; ++col) {
771  for (El::Int row = row_start; row < row_end; ++row) {
772  local_output(row, col) += bias_term;
773  }
774  }
775  }

779  void compute_gradients_im2col(bool using_transposed_convolution) {
780 
781  // Local matrices
782  const auto& local_input = get_local_prev_activations();
783  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
784  auto& local_kernel_gradient = m_kernel_gradient.Matrix();
785  auto& local_bias_gradient = m_bias_gradient.Matrix();
786 
787  // Get convolution parameters
788  const El::Int local_width = local_input.Width();
789  const int num_input_channels = this->m_prev_neuron_dims[0];
790  const int num_output_channels = this->m_neuron_dims[0];
791  const int num_per_output_channel = this->m_num_neurons / num_output_channels;
792  const int effective_mini_batch_size = this->m_model->get_effective_mini_batch_size();
793 
794  // Compute bias gradient
795  // Note: Sum is computed with Kahan summation
796  optimizer* bias_optimizer = this->m_weights[1]->get_optimizer();
797  if (m_bias_scaling_factor != DataType(0) && bias_optimizer != nullptr) {
798  #pragma omp parallel for
799  for (int channel = 0; channel < num_output_channels; ++channel) {
800  const El::Int row_start = channel * num_per_output_channel;
801  const El::Int row_end = (channel+1) * num_per_output_channel;
802  DataType sum = 0;
803  DataType correction = 0;
804  for (El::Int col = 0; col < local_width; ++col) {
805  for (El::Int row = row_start; row < row_end; ++row) {
806  DataType term = local_gradient_wrt_output(row, col);
807  term += correction;
808  const DataType next_sum = sum + term;
809  correction = term - (next_sum - sum);
810  sum = next_sum;
811  }
812  }
813  local_bias_gradient(channel, 0) = m_bias_scaling_factor * sum;
814  }
815  const DataType bias_scale = m_bias_scaling_factor / effective_mini_batch_size;
816  bias_optimizer->add_to_gradient_staging(m_bias_gradient,
817  bias_scale);
818  }
819 
820  // Stop early if kernel is not being optimized
821  optimizer* kernel_optimizer = this->m_weights[0]->get_optimizer();
822  if (kernel_optimizer == nullptr) { return; }
823 
824  // Initialize matrices
825  const int m = (using_transposed_convolution ?
826  m_kernel_size / num_input_channels :
827  m_kernel_size / num_output_channels);
828  const int n = (using_transposed_convolution ?
829  num_input_channels :
830  num_output_channels);
831  const int k = (using_transposed_convolution ?
832  this->m_num_prev_neurons / num_input_channels :
833  this->m_num_neurons / num_output_channels);
834  Mat im2col_matrix(m, k);
835  Mat kernel_gradient_matrix(m, n, local_kernel_gradient.Buffer(), m);
836  El::Zero(kernel_gradient_matrix);
837 
838  // Compute kernel gradient contributions from each data sample
839  for (El::Int col = 0; col < local_width; ++col) {
840  if (using_transposed_convolution) {
841  const Mat input_col(k, n, local_input.LockedBuffer(0,col), k);
842  const Mat gradient_wrt_output_col
843  = El::LockedView(local_gradient_wrt_output, El::ALL, El::IR(col));
844  im2col(gradient_wrt_output_col,
845  im2col_matrix,
846  num_output_channels,
847  this->m_num_neuron_dims - 1,
848  &this->m_neuron_dims[1],
849  m_pads.data(),
850  &m_kernel_dims[2],
851  m_strides.data());
852  El::Gemm(El::NORMAL, El::NORMAL,
853  DataType(1), im2col_matrix, input_col,
854  DataType(1), kernel_gradient_matrix);
855  }
856  else {
857  const Mat input_col
858  = El::LockedView(local_input, El::ALL, El::IR(col));
859  const Mat gradient_wrt_output_col(k, n, local_gradient_wrt_output.LockedBuffer(0,col), k);
860  im2col(input_col,
861  im2col_matrix,
862  num_input_channels,
863  this->m_num_prev_neuron_dims - 1,
864  &this->m_prev_neuron_dims[1],
865  m_pads.data(),
866  &m_kernel_dims[2],
867  m_strides.data());
868  El::Gemm(El::NORMAL, El::NORMAL,
869  DataType(1), im2col_matrix, gradient_wrt_output_col,
870  DataType(1), kernel_gradient_matrix);
871  }
872  }
873 
874  // Scale and accumulate gradients
875  const DataType kernel_scale = DataType(1) / effective_mini_batch_size;
876  kernel_optimizer->add_to_gradient_staging(m_kernel_gradient,
877  kernel_scale);
878 
879  }
