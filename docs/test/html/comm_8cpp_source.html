94 void lbann_comm::split_models(int ppm) {
95  int world_size = El::mpi::Size(get_world_comm());
96  procs_per_model = ppm;
97  if (ppm == 0) {
98  procs_per_model = world_size;
99  }
100  // Check if parameters are valid
101  if (procs_per_model > world_size) {
102  throw lbann_exception(
103  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
104  " :: Not enough processes to create one model; procs_per_model: " +
105  std::to_string(procs_per_model) + " is larger than world_size: " +
106  std::to_string(world_size));
107  }
108  if (world_size % procs_per_model != 0) {
109  throw lbann_exception(
110  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
111  " :: Procs per model does not divide total number of procs; procs_per_model: " +
112  std::to_string(procs_per_model) + " total number of procs (world size): " +
113  std::to_string(world_size));
114  }
115 
116  num_models = world_size / procs_per_model;
117  model_rank = El::mpi::Rank(get_world_comm()) / procs_per_model;
118  rank_in_model = El::mpi::Rank(get_world_comm()) % procs_per_model;
119 
120  // Initialize model and intermodel communicators
121  El::mpi::Split(get_world_comm(), model_rank, rank_in_model, model_comm);
122  El::mpi::Split(get_world_comm(), rank_in_model, model_rank,
123  intermodel_comm);
124 
125  // Initialize Elemental grid
126  if (grid != nullptr) {
127  delete grid;
128  }
129  grid = new Grid(model_comm);
130 }

142 void lbann_comm::allreduce(AbsDistMat& m,
143  const El::mpi::Comm c,
144  El::mpi::Op op,
145  std::type_index t) {
146  const int local_size = m.LocalHeight() * m.LocalWidth();
147  bytes_sent += sizeof(DataType) * local_size;
148 #ifdef LBANN_HAS_ALUMINUM
149  if (m.LocalHeight() != m.LDim()) {
150  throw lbann_exception("Aluminum does not support allreduces on"
151  " non-contiguous matrices");
152  }
153  auto&& comm = get_al_comm(c, t);
154  if (t == std::type_index(typeid(::Al::MPIBackend))) {
155  ::Al::Allreduce<::Al::MPIBackend>(
156  m.Buffer(),
157  local_size,
158  mpi_op_to_al_op(op),
159  *comm);
160  }
162 #ifdef LBANN_HAS_NCCL2
163  if (t == std::type_index(typeid(::Al::NCCLBackend))) {
164  ::Al::Allreduce<::Al::NCCLBackend>(
165  m.Buffer(),
166  local_size,
167  mpi_op_to_al_op(op),
168  *static_cast<::Al::NCCLCommunicator*>(comm));
169  }
170 #endif // LBANN_HAS_NCCL2
171 #else
172  El::AllReduce(m, c, op);
173 #endif
174  bytes_received += sizeof(DataType) * local_size * (El::mpi::Size(c) - 1);
175 }

325 void lbann_comm::intermodel_allreduce(
326  Mat& mat, int max_recv_count,
327  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
328  std::function<int(uint8_t *, Mat&)> recv_transform,
329  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
330  const lbann_comm::allreduce_options opts) {
331  // Determine which algorithm to actually use.
332  lbann_comm::allreduce_algorithm algo = opts.algo;
333  if (algo == allreduce_algorithm::DEFAULT) {
334  algo = get_default_allreduce_algorithm();
335  }
336  const int nprocs = get_num_models();
337  const El::Int small_message_threshold = 64*64;
338  if (algo == allreduce_algorithm::DYNAMIC) {
339  // For small messages and power-of-2 number of processes, use RD.
340  if (!(nprocs & (nprocs - 1)) &&
341  mat.Height() * mat.Width() <= small_message_threshold) {
342  algo = allreduce_algorithm::RECURSIVE_DOUBLING;
343  } else {
344  algo = allreduce_algorithm::PAIRWISE_EXCHANGE_RING;
345  }
346  }
347  switch (algo) {
348  case allreduce_algorithm::RECURSIVE_DOUBLING:
349  recursive_doubling_allreduce_pow2(
350  intermodel_comm, mat, max_recv_count,
351  send_transform, recv_apply_transform, opts);
352  break;
353  case allreduce_algorithm::PAIRWISE_EXCHANGE_RING:
354  pe_ring_allreduce(
355  intermodel_comm, mat, max_recv_count, send_transform,
356  recv_transform, recv_apply_transform, opts);
357  break;
358  case allreduce_algorithm::RING:
359  ring_allreduce(
360  intermodel_comm, mat, max_recv_count, send_transform,
361  recv_transform, recv_apply_transform, opts);
362  break;
363  case allreduce_algorithm::RABENSEIFNER:
364  rabenseifner_allreduce(
365  intermodel_comm, mat, max_recv_count, send_transform,
366  recv_transform, recv_apply_transform, opts);
367  break;
368  case allreduce_algorithm::DEFAULT:
369  case allreduce_algorithm::DYNAMIC:
370  default:
371  throw lbann_exception("intermodel_allreduce: bad algorithm");
372  break;
373  }
374 }

376 void lbann_comm::recursive_doubling_allreduce_pow2(
377  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
378  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
379  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
380  const lbann_comm::allreduce_options opts) {
381  double ar_start = get_time();
382  const int rank = El::mpi::Rank(comm);
383  const unsigned int nprocs = El::mpi::Size(comm);
384  if (nprocs == 1) {
385  return; // Nothing to do.
386  }
387  // This implementation requires a power-of-2 number of processes.
388  if (nprocs & (nprocs - 1)) {
389  throw lbann_exception("lbann_comm: recursive doubling allreduce requires"
390  " a power-of-2 number of participating processes");
391  }
392  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
393  uint8_t *recv_buf = max_recv_buf;
394  unsigned int mask = 1;
395  while (mask < nprocs) {
396  int partner = rank ^ mask; // The rank we exchange with this step.
397  const bool is_local = opts.no_local_trans &&
398  is_rank_node_local(partner, comm);
399  // Transform the data we want to send.
400  double send_trans_start = get_time();
401  int send_size;
402  int recv_size = max_recv_count;
403  uint8_t *send_buf = nullptr;
404  if (is_local) {
405  send_buf = (uint8_t *) mat.Buffer();
406  send_size = sizeof(DataType) * mat.Height() * mat.Width();
407  recv_size = send_size;
408  recv_buf = get_collective_buffer(recv_size);
409  } else {
410  send_buf = send_transform(mat, El::ALL, El::ALL, send_size, false, 0);
411  recv_buf = max_recv_buf;
412  }
413  ar_send_transform_time += get_time() - send_trans_start;
414  bytes_sent += send_size;
415  ar_bytes_sent += send_size;
416  double sendrecv_start = get_time();
417  El::mpi::SendRecv(send_buf, send_size, partner,
418  recv_buf, recv_size, partner, comm);
419  double sendrecv_tot = get_time() - sendrecv_start;
420  ar_send_time += sendrecv_tot;
421  ar_recv_time += sendrecv_tot;
422  // Transform and reduce the received data.
423  double recv_apply_trans_start = get_time();
424  recv_size = recv_apply_transform(recv_buf, mat, is_local);
425  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
426  bytes_received += recv_size;
427  ar_bytes_received += recv_size;
428  mask <<= 1;
429  }
430  ar_time += get_time() - ar_start;
431 }

433 void lbann_comm::pe_ring_allreduce(
434  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
435  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
436  std::function<int(uint8_t *, Mat&)> recv_transform,
437  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
438  const lbann_comm::allreduce_options opts) {
439  double ar_start = get_time();
440  const int rank = El::mpi::Rank(comm);
441  const int nprocs = El::mpi::Size(comm);
442  if (nprocs == 1) {
443  return; // Nothing to do.
444  }
445  // Compute the number of columns each processor sends.
446  // If it doesn't divide evenly, give one extra to the earlier ranks.
447  const El::Int cols_per_proc = mat.Width() / nprocs;
448  const El::Int cols_remainder = mat.Width() % nprocs;
449  // Compute the lengths/ends of the slices.
450  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
451  for (int i = 0; i < cols_remainder; ++i) {
452  slice_lengths[i] += 1;
453  }
454  std::vector<El::Int> slice_ends(nprocs);
455  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
456  slice_ends.begin());
457  std::vector<uint8_t *> max_recv_buffers(opts.max_reduces, nullptr);
458  for (size_t i = 0; i < max_recv_buffers.size(); ++i) {
459  max_recv_buffers[i] = get_collective_buffer(max_recv_count, i);
460  }
461  // Local slice of our accumulated data.
462  auto accum_view = mat(El::ALL, El::IR(slice_ends[rank] - slice_lengths[rank],
463  slice_ends[rank]));
464  // Do a pairwise-exchange reduce-scatter.
465  double rs_start = get_time();
466  for (int outer_step = 1;
467  outer_step < nprocs;
468  outer_step += opts.max_reduces) {
469  const int reduces_this_step = std::min(opts.max_reduces,
470  nprocs - outer_step);
471  std::vector<El::mpi::Request<uint8_t>> send_reqs(reduces_this_step);
472  std::vector<El::mpi::Request<uint8_t>> recv_reqs(reduces_this_step);
473  std::vector<uint8_t *> recv_buffers(max_recv_buffers);
474  int num_local_recvs = 0;
475  std::vector<bool> local_recvs(reduces_this_step, false);
476  for (int step = outer_step; step < outer_step + reduces_this_step; ++step) {
477  const int reduce_idx = step - outer_step;
478  // Compute where we send to/receive from.
479  const int dst = (rank + step) % nprocs;
480  const int src = (rank - step + nprocs) % nprocs;
481  const bool is_send_local = opts.no_local_trans &&
482  is_rank_node_local(dst, comm);
483  const bool is_recv_local = opts.no_local_trans &&
484  is_rank_node_local(src, comm);
485  // Post the receive.
486  double recv_start = get_time();
487  int recv_size = max_recv_count;
488  if (is_recv_local) {
489  recv_size = sizeof(DataType) * accum_view.Height() * accum_view.Width();
490  recv_buffers[reduce_idx] = get_collective_buffer(recv_size,
491  num_local_recvs);
492  ++num_local_recvs;
493  local_recvs[reduce_idx] = is_recv_local;
494  }
495  El::mpi::IRecv(recv_buffers[reduce_idx], recv_size, src, comm,
496  recv_reqs[reduce_idx]);
497  double recv_tot = get_time() - recv_start;
498  ar_recv_time += recv_tot;
499  ar_rs_recv_time += recv_tot;
500  // Transform the data we send. We do not look at the same chunk of data
501  // twice.
502  double send_trans_start = get_time();
503  int send_size;
504  uint8_t *send_buf = nullptr;
505  if (is_send_local) {
506  auto send_view = mat(El::ALL,
507  El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
508  send_buf = (uint8_t *) send_view.Buffer();
509  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
510  } else {
511  send_buf = send_transform(
512  mat, El::ALL, El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]),
513  send_size, true, reduce_idx);
514  }
515  ar_send_transform_time += get_time() - send_trans_start;
516  bytes_sent += send_size;
517  ar_bytes_sent += send_size;
518  ar_rs_bytes_sent += send_size;
519  // Post the send.
520  double send_start = get_time();
521  El::mpi::ISend(send_buf, send_size, dst, comm, send_reqs[reduce_idx]);
522  double send_tot = get_time() - send_start;
523  ar_send_time += send_tot;
524  ar_rs_send_time += send_tot;
525  }
526  // Complete the receives (in any order).
527  // We need to extract the raw MPI_Request because Elemental does not support
528  // MPI_Waitany.
529  std::vector<MPI_Request> raw_reqs(reduces_this_step);
530  for (int i = 0; i < reduces_this_step; ++i) {
531  raw_reqs[i] = recv_reqs[i].backend;
532  }
533  for (int i = 0; i < reduces_this_step; ++i) {
534  int completed_idx;
535  double recv_start = get_time();
536  MPI_Waitany(reduces_this_step, raw_reqs.data(), &completed_idx,
537  MPI_STATUS_IGNORE);
538  double recv_tot = get_time() - recv_start;
539  ar_recv_time += recv_tot;
540  ar_rs_recv_time += recv_tot;
541  double recv_apply_trans_start = get_time();
542  int recv_size = recv_apply_transform(
543  recv_buffers[completed_idx], accum_view, local_recvs[completed_idx]);
544  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
545  bytes_received += recv_size;
546  ar_bytes_received += recv_size;
547  ar_rs_bytes_received += recv_size;
548  }
549  // Complete all the sends.
550  double send_start = get_time();
551  El::mpi::WaitAll(reduces_this_step, send_reqs.data(), MPI_STATUSES_IGNORE);
552  double send_tot = get_time() - send_start;
553  ar_send_time += send_tot;
554  ar_rs_send_time += send_tot;
555  }
556  uint8_t *recv_buf = max_recv_buffers[0]; // Get a regular recv buffer.
557  ar_rs_time += get_time() - rs_start;
558  // Do a ring allgather.
559  double ag_start = get_time();
560  const int src = (rank - 1 + nprocs) % nprocs;
561  const int dst = (rank + 1) % nprocs;
562  // Apply the transform to our locally-accumulated slice of the data.
563  // Since the same data is cycled to every process, we do not do the
564  // no_local_trans here.
565  int send_size;
566  // Do the first step where we forward our local data.
567  {
568  double send_trans_start = get_time();
569  uint8_t *send_buf = send_transform(
570  mat, El::ALL, El::IR(slice_ends[rank] - slice_lengths[rank], slice_ends[rank]),
571  send_size, false, 0);
572  ar_send_transform_time += get_time() - send_trans_start;
573  const int data_src = (rank - 1 + nprocs) % nprocs;
574  bytes_sent += send_size;
575  ar_bytes_sent += send_size;
576  ar_ag_bytes_sent += send_size;
577  auto recv_view = mat(El::ALL,
578  El::IR(slice_ends[data_src] - slice_lengths[data_src],
579  slice_ends[data_src]));
580  // If we can, receive directly into the destination matrix.
581  if (opts.id_recv) {
582  recv_buf = (uint8_t *) recv_view.Buffer();
583  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
584  }
585  double sendrecv_start = get_time();
586  El::mpi::SendRecv(send_buf, send_size, dst,
587  recv_buf, max_recv_count, src, comm);
588  double sendrecv_tot = get_time() - sendrecv_start;
589  ar_send_time += sendrecv_tot;
590  ar_recv_time += sendrecv_tot;
591  ar_ag_send_time += sendrecv_tot;
592  ar_ag_recv_time += sendrecv_tot;
593  double recv_trans_start = get_time();
594  int recv_size = 0;
595  if (opts.id_recv) {
596  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
597  } else {
598  recv_size = recv_transform(recv_buf, recv_view);
599  }
600  ar_recv_transform_time += get_time() - recv_trans_start;
601  bytes_received += recv_size;
602  ar_bytes_received += recv_size;
603  ar_ag_bytes_received += send_size;
604  send_size = recv_size;
605  }
606  // Now do the remaining nprocs - 2 steps.
607  // We always send from recv_buf and receive to recv_buf2, swapping
608  // pointers to avoid copying.
609  uint8_t *recv_buf2 = nullptr;
610  if (!opts.id_recv) {
611  recv_buf2 = get_collective_buffer(max_recv_count, 1);
612  }
613  for (int step = 1; step < nprocs - 1; ++step) {
614  // Compute where the data we get is coming from.
615  const int data_src = (rank - step - 1 + nprocs) % nprocs;
616  auto recv_view = mat(El::ALL,
617  El::IR(slice_ends[data_src] - slice_lengths[data_src],
618  slice_ends[data_src]));
619  if (opts.id_recv) {
620  recv_buf2 = (uint8_t *) recv_view.Buffer();
621  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
622  }
623  bytes_sent += send_size;
624  ar_bytes_sent += send_size;
625  ar_ag_bytes_sent += send_size;
626  double sendrecv_start = get_time();
627  El::mpi::SendRecv(recv_buf, send_size, dst,
628  recv_buf2, max_recv_count, src, comm);
629  double sendrecv_tot = get_time() - sendrecv_start;
630  ar_send_time += sendrecv_tot;
631  ar_recv_time += sendrecv_tot;
632  ar_ag_send_time += sendrecv_tot;
633  ar_ag_recv_time += sendrecv_tot;
634  double recv_trans_start = get_time();
635  int recv_size = 0;
636  if (opts.id_recv) {
637  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
638  } else {
639  recv_size = recv_transform(recv_buf2, recv_view);
640  }
641  ar_recv_transform_time += get_time() - recv_trans_start;
642  bytes_received += recv_size;
643  // Swap the send and receive buffers.
644  std::swap(recv_buf, recv_buf2);
645  send_size = recv_size;
646  ar_bytes_received += recv_size;
647  ar_ag_bytes_received += send_size;
648  }
649  ar_ag_time += get_time() - ag_start;
650  ar_time += get_time() - ar_start;
651 }

653 void lbann_comm::ring_allreduce(
654  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
655  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
656  std::function<int(uint8_t *, Mat&)> recv_transform,
657  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
658  const lbann_comm::allreduce_options opts) {
659  double ar_start = get_time();
660  const int rank = El::mpi::Rank(comm);
661  const int nprocs = El::mpi::Size(comm);
662  if (nprocs == 1) {
663  return; // Nothing to do.
664  }
665  // Compute the number of columns each processor sends.
666  const El::Int cols_per_proc = mat.Width() / nprocs;
667  const El::Int cols_remainder = mat.Width() % nprocs;
668  // Compute the lengths/ends of the slices.
669  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
670  for (int i = 0; i < cols_remainder; ++i) {
671  slice_lengths[i] += 1;
672  }
673  std::vector<El::Int> slice_ends(nprocs);
674  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
675  slice_ends.begin());
676  uint8_t *max_recv_buf = get_collective_buffer(max_recv_count);
677  uint8_t *recv_buf = max_recv_buf;
678  // Compute source/destination in the ring.
679  const int src = (rank - 1 + nprocs) % nprocs;
680  const int dst = (rank + 1) % nprocs;
681  const bool is_send_local = opts.no_local_trans &&
682  is_rank_node_local(dst, comm);
683  const bool is_recv_local = opts.no_local_trans &&
684  is_rank_node_local(src, comm);
685  // Do a ring-based reduce-scatter.
686  // This is like the pairwise-exchange reduce-scatter except instead of
687  // rank i accumulating only slice i, the slices are cycled around and
688  // each node accumulates its portion into the slice when it passes
689  // through. After the nprocs-1 steps slice k will be on rank
690  // (k + nprocs - 1) % nprocs.
691  double rs_start = get_time();
692  for (int step = 0; step < nprocs - 1; ++step) {
693  // Compute the slices to send/recv.
694  const int send_slice = (rank - step + nprocs) % nprocs;
695  const int recv_slice = (rank - step - 1 + nprocs) % nprocs;
696  // Transform the data to send.
697  double send_trans_start = get_time();
698  int send_size;
699  int recv_size = max_recv_count;
700  uint8_t *send_buf = nullptr;
701  if (is_send_local) {
702  auto send_view = mat(El::ALL,
703  El::IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]));
704  send_buf = (uint8_t *) send_view.Buffer();
705  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
706  } else {
707  send_buf = send_transform(mat, El::ALL,
708  El::IR(slice_ends[send_slice] - slice_lengths[send_slice],
709  slice_ends[send_slice]), send_size, false, 0);
710  }
711  auto recv_view = mat(El::ALL,
712  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice], slice_ends[recv_slice]));
713  if (is_recv_local) {
714  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
715  recv_buf = get_collective_buffer(recv_size);
716  } else {
717  recv_buf = max_recv_buf;
718  }
719  ar_send_transform_time += get_time() - send_trans_start;
720  bytes_sent += send_size;
721  ar_bytes_sent += send_size;
722  ar_rs_bytes_sent += send_size;
723  double sendrecv_start = get_time();
724  El::mpi::SendRecv(send_buf, send_size, dst,
725  recv_buf, recv_size, src, comm);
726  double sendrecv_tot = get_time() - sendrecv_start;
727  ar_send_time += sendrecv_tot;
728  ar_recv_time += sendrecv_tot;
729  ar_rs_send_time += sendrecv_tot;
730  ar_rs_recv_time += sendrecv_tot;
731  double recv_apply_trans_start = get_time();
732  recv_size = recv_apply_transform(recv_buf, recv_view, is_recv_local);
733  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
734  bytes_received += recv_size;
735  ar_bytes_received += recv_size;
736  ar_rs_bytes_received += recv_size;
737  }
738  recv_buf = max_recv_buf; // Ensure we're back to the original.
739  ar_rs_time += get_time() - rs_start;
740  // Do a ring allgather, first applying the transform to local data.
741  double ag_start = get_time();
742  int send_size;
743  {
744  const int send_slice = (rank + 1) % nprocs;
745  const int recv_slice = rank;
746  double send_trans_start = get_time();
747  uint8_t *send_buf = send_transform(
748  mat, El::ALL, El::IR(slice_ends[send_slice] - slice_lengths[send_slice],
749  slice_ends[send_slice]), send_size, false, 0);
750  ar_send_transform_time += get_time() - send_trans_start;
751  bytes_sent += send_size;
752  ar_bytes_sent += send_size;
753  ar_ag_bytes_sent += send_size;
754  auto recv_view = mat(El::ALL,
755  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
756  slice_ends[recv_slice]));
757  // If we can, receive directly into the destination matrix.
758  if (opts.id_recv) {
759  recv_buf = (uint8_t *) recv_view.Buffer();
760  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
761  }
762  double sendrecv_start = get_time();
763  El::mpi::SendRecv(send_buf, send_size, dst,
764  recv_buf, max_recv_count, src, comm);
765  double sendrecv_tot = get_time() - sendrecv_start;
766  ar_send_time += sendrecv_tot;
767  ar_recv_time += sendrecv_tot;
768  ar_ag_send_time += sendrecv_tot;
769  ar_ag_recv_time += sendrecv_tot;
770  double recv_trans_start = get_time();
771  int recv_size = 0;
772  if (opts.id_recv) {
773  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
774  } else {
775  recv_size = recv_transform(recv_buf, recv_view);
776  }
777  ar_recv_transform_time += get_time() - recv_trans_start;
778  send_size = recv_size;
779  bytes_received += recv_size;
780  ar_bytes_received += recv_size;
781  ar_ag_bytes_received += recv_size;
782  }
783  uint8_t *recv_buf2 = nullptr;
784  if (!opts.id_recv) {
785  recv_buf2 = get_collective_buffer(max_recv_count, 1);
786  }
787  for (int step = 1; step < nprocs - 1; ++step) {
788  const int recv_slice = (rank - step + nprocs) % nprocs;
789  auto recv_view = mat(El::ALL,
790  El::IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
791  slice_ends[recv_slice]));
792  if (opts.id_recv) {
793  recv_buf2 = (uint8_t *) recv_view.Buffer();
794  max_recv_count = sizeof(DataType) * recv_view.Height() * recv_view.Width();
795  }
796  bytes_sent += send_size;
797  ar_bytes_sent += send_size;
798  ar_ag_bytes_sent += send_size;
799  double sendrecv_start = get_time();
800  El::mpi::SendRecv(recv_buf, send_size, dst,
801  recv_buf2, max_recv_count, src, comm);
802  double sendrecv_tot = get_time() - sendrecv_start;
803  ar_send_time += sendrecv_tot;
804  ar_recv_time += sendrecv_tot;
805  ar_ag_send_time += sendrecv_tot;
806  ar_ag_recv_time += sendrecv_tot;
807  double recv_trans_start = get_time();
808  int recv_size = 0;
809  if (opts.id_recv) {
810  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
811  } else {
812  recv_size = recv_transform(recv_buf2, recv_view);
813  }
814  ar_recv_transform_time += get_time() - recv_trans_start;
815  // Swap the send and receive buffers.
816  std::swap(recv_buf, recv_buf2);
817  send_size = recv_size;
818  bytes_received += recv_size;
819  ar_bytes_received += recv_size;
820  ar_ag_bytes_received += recv_size;
821  }
822  ar_ag_time += get_time() - ag_start;
823  ar_time += get_time() - ar_start;
824 }

826 void lbann_comm::rabenseifner_allreduce(
827  const El::mpi::Comm comm, Mat& mat, int max_recv_count,
828  std::function<uint8_t *(Mat&, El::IR, El::IR, int&, bool, int)> send_transform,
829  std::function<int(uint8_t *, Mat&)> recv_transform,
830  std::function<int(uint8_t *, Mat&, bool)> recv_apply_transform,
831  const lbann_comm::allreduce_options opts) {
832  double ar_start = get_time();
833  const int rank = El::mpi::Rank(comm);
834  const unsigned int nprocs = El::mpi::Size(comm);
835  if (nprocs == 1) {
836  return; // Nothing to do.
837  }
838  // This implementation requires a power-of-2 number of processes.
839  if (nprocs & (nprocs - 1)) {
840  throw lbann_exception("lbann_comm: Rabenseifner allreduce requires"
841  " a power-of-2 number of participating processes");
842  }
843  // Compute the slices on each processor.
844  const El::Int cols_per_proc = mat.Width() / nprocs;
845  const El::Int cols_remainder = mat.Width() % nprocs;
846  // Compute the lengths/ends of the slices.
847  std::vector<El::Int> slice_lengths(nprocs, cols_per_proc);
848  for (int i = 0; i < cols_remainder; ++i) {
849  slice_lengths[i] += 1;
850  }
851  std::vector<El::Int> slice_ends(nprocs);
852  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
853  slice_ends.begin());
854  // Do a recursive-halving reduce-scatter.
855  // In each step here a process sends all the data needed for the other
856  // "half" of the processes. i.e. each process sends half their data in the
857  // first step, a quarter in the second step, etc.
858  double rs_start = get_time();
859  unsigned int partner_mask = nprocs >> 1;
860  unsigned int slice_mask = 1;
861  unsigned int send_idx = 0;
862  unsigned int recv_idx = 0;
863  unsigned int last_idx = nprocs;
864  uint8_t *recv_buf = get_collective_buffer(max_recv_count);
865  while (partner_mask > 0) {
866  int partner = rank ^ partner_mask; // The rank we exchange with this step.
867  const bool is_local = opts.no_local_trans &&
868  is_rank_node_local(partner, comm);
869  // Determine the range of data to send/recv.
870  El::IR send_range, recv_range;
871  if (rank < partner) {
872  send_idx = recv_idx + nprocs / (slice_mask*2);
873  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
874  slice_ends[last_idx-1]);
875  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
876  slice_ends[send_idx-1]);
877  } else {
878  recv_idx = send_idx + nprocs / (slice_mask*2);
879  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
880  slice_ends[recv_idx-1]);
881  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
882  slice_ends[last_idx-1]);
883  }
884  auto recv_view = mat(El::ALL, recv_range);
885  // Transform the data to send.
886  double send_trans_start = get_time();
887  int send_size;
888  int recv_size = max_recv_count;
889  uint8_t *send_buf = nullptr;
890  if (is_local) {
891  auto send_view = mat(El::ALL, send_range);
892  send_buf = (uint8_t *) send_view.Buffer();
893  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
894  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
895  } else {
896  send_buf = send_transform(mat, El::ALL, send_range, send_size, false, 0);
897  }
898  ar_send_transform_time += get_time() - send_trans_start;
899  bytes_sent += send_size;
900  ar_bytes_sent += send_size;
901  ar_rs_bytes_sent += send_size;
902  double sendrecv_start = get_time();
903  El::mpi::SendRecv(send_buf, send_size, partner,
904  recv_buf, recv_size, partner, comm);
905  double sendrecv_tot = get_time() - sendrecv_start;
906  ar_send_time += sendrecv_tot;
907  ar_recv_time += sendrecv_tot;
908  ar_rs_send_time += sendrecv_tot;
909  ar_rs_recv_time += sendrecv_tot;
910  // Transform the received data.
911  double recv_apply_trans_start = get_time();
912  recv_size = recv_apply_transform(recv_buf, recv_view, is_local);
913  ar_recv_apply_transform_time += get_time() - recv_apply_trans_start;
914  bytes_received += recv_size;
915  ar_bytes_received += recv_size;
916  ar_rs_bytes_received += send_size;
917  // Update info for next iteration.
918  // Except last_idx when needed for the allgather.
919  send_idx = recv_idx;
920  partner_mask >>= 1;
921  slice_mask <<= 1;
922  if (partner_mask > 0) {
923  last_idx = recv_idx + nprocs / (slice_mask);
924  }
925  }
926  ar_rs_time += get_time() - rs_start;
927  // Do a recursive-doubling algather.
928  double ag_start = get_time();
929  slice_mask >>= 1;
930  partner_mask = 1;
931  // Now do the remaining steps.
932  while (partner_mask < nprocs) {
933  int partner = rank ^ partner_mask;
934  const bool is_local = opts.no_local_trans &&
935  is_rank_node_local(partner, comm);
936  // Determine range to send/recv.
937  El::IR send_range, recv_range;
938  if (rank < partner) {
939  if (slice_mask != nprocs / 2) {
940  last_idx = last_idx + nprocs / (slice_mask*2);
941  }
942  recv_idx = send_idx + nprocs / (slice_mask*2);
943  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
944  slice_ends[recv_idx-1]);
945  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
946  slice_ends[last_idx-1]);
947  } else {
948  recv_idx = send_idx - nprocs / (slice_mask*2);
949  send_range = El::IR(slice_ends[send_idx] - slice_lengths[send_idx],
950  slice_ends[last_idx-1]);
951  recv_range = El::IR(slice_ends[recv_idx] - slice_lengths[recv_idx],
952  slice_ends[send_idx-1]);
953  }
954  auto recv_view = mat(El::ALL, recv_range);
955  // Transform the data to send.
956  double send_trans_start = get_time();
957  int send_size;
958  int recv_size = max_recv_count;
959  uint8_t *send_buf = nullptr;
960  if (is_local) {
961  auto send_view = mat(El::ALL, send_range);
962  send_buf = (uint8_t *) send_view.Buffer();
963  send_size = sizeof(DataType) * send_view.Height() * send_view.Width();
964  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
965  } else {
966  send_buf = send_transform(mat, El::ALL, send_range, send_size, false, 0);
967  }
968  ar_send_transform_time += get_time() - send_trans_start;
969  if (opts.id_recv || is_local) {
970  recv_buf = (uint8_t *) recv_view.Buffer();
971  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
972  }
973  bytes_sent += send_size;
974  ar_bytes_sent += send_size;
975  ar_ag_bytes_sent += send_size;
976  double sendrecv_start = get_time();
977  El::mpi::SendRecv(send_buf, send_size, partner,
978  recv_buf, recv_size, partner, comm);
979  double sendrecv_tot = get_time() - sendrecv_start;
980  ar_send_time += sendrecv_tot;
981  ar_recv_time += sendrecv_tot;
982  ar_ag_send_time += sendrecv_tot;
983  ar_ag_recv_time += sendrecv_tot;
984  double recv_trans_start = get_time();
985  if (opts.id_recv) {
986  recv_size = sizeof(DataType) * recv_view.Height() * recv_view.Width();
987  } else {
988  recv_size = recv_transform(recv_buf, recv_view);
989  }
990  ar_recv_transform_time += get_time() - recv_trans_start;
991  bytes_received += recv_size;
992  ar_bytes_received += recv_size;
993  ar_ag_bytes_received += send_size;
994  // Update for the next iteration.
995  if (rank > partner) {
996  send_idx = recv_idx;
997  }
998  partner_mask <<= 1;
999  slice_mask >>= 1;
1000  }
1001  ar_ag_time += get_time() - ag_start;
1002  ar_time += get_time() - ar_start;
1003 }

1005 void lbann_comm::setup_node_comm() {
1006 
1007  // Get string specifying compute node
1008  char node_name[MPI_MAX_PROCESSOR_NAME];
1009  int node_name_len;
1010  checkMPI(MPI_Get_processor_name(node_name, &node_name_len));
1011  const std::string node_string(node_name);
1012 
1013  // Hash node names and split MPI processes
1014  int hash = std::hash<std::string>()(node_string);
1015  hash = hash >= 0 ? hash : -hash; // Make sure hash is non-negative
1016  El::mpi::Comm hash_comm;
1017  El::mpi::Split(get_world_comm(), hash,
1018  El::mpi::Rank(get_world_comm()), hash_comm);
1019  const int hash_comm_size = El::mpi::Size(hash_comm);
1020 
1021  // Compare node names and split MPI processes
1022  auto *node_name_list = new char[hash_comm_size*MPI_MAX_PROCESSOR_NAME];
1023  checkMPI(MPI_Allgather(node_name, MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
1024  node_name_list, MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
1025  hash_comm.comm));
1026  int node_num = El::mpi::Rank(hash_comm);
1027  for(int i=0; i<hash_comm_size; ++i) {
1028  const std::string other_node_string(node_name_list + i*MPI_MAX_PROCESSOR_NAME);
