46 /* creates directory given in dir (absolute path),
47  * rank 0 creates directory, all other procs get result via bcast,
48  * returns 1 if dir was created, 0 otherwise */
49 int lbann::makedir(const char *dir) {
50  // get our rank
51  int rank;
52  MPI_Comm_rank(comm, &rank);
53 
54  // have rank 0 create directory
55  int mkdir_rc;
56  if (rank == 0) {
57  mkdir_rc = mkdir(dir, mode_dir);
58  if (mkdir_rc != 0) {
59  if (errno == EEXIST) {
60  // not an error if the directory already exists
61  mkdir_rc = 0;
62  } else {
63  fprintf(stderr, "ERROR: Failed to create directory `%s' (%d: %s) @ %s:%d\n",
64  dir, errno, strerror(errno), __FILE__, __LINE__
65  );
66  fflush(stderr);
67  }
68  }
69  }
70 
71  // bcast whether directory was created or not
72  MPI_Bcast(&mkdir_rc, 1, MPI_INT, 0, comm);
73 
74  // return 1 if dir was created successfully
75  int ret = (mkdir_rc == 0);
76  return ret;
77 }
