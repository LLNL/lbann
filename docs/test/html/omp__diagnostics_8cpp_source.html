88 /*
89  * Get the affinity.
90  */
91 int get_affinity(uint8_t *cpus, uint8_t *count)
92 {
93  int i, rc;
94  cpu_set_t resmask;
95  int pus = get_num_pus();
96 
97 
98  CPU_ZERO(&resmask);
99  if ( (rc = sched_getaffinity(0, sizeof(resmask), &resmask)) < 0 ) {
100  perror("sched_getaffinity");
101  return rc;
102  }
103 
104  *count = 0;
105  for (i=0; i<pus; i++)
106  if ( CPU_ISSET(i, &resmask) ) {
107  cpus[*count] = i;
108  (*count)++;
109  }
110 
111  return 0;
112 }

123

179 void print_affinity_subset(int rank, int np, char *host)
180 {
181  MPI_Comm sh_comm;
182  int sh_rank, sh_np;
183 
184 
185  // Create communicators
186  MPI_CHECK( MPI_Comm_split_type(MPI_COMM_WORLD, MPI_COMM_TYPE_SHARED,
187  rank, MPI_INFO_NULL, &sh_comm) );
188  MPI_CHECK( MPI_Comm_rank(sh_comm, &sh_rank) );
189  MPI_CHECK( MPI_Comm_size(sh_comm, &sh_np) );
190 
191  if (rank < 2*sh_np)
192  print_affinity(rank, np, host);
193 
194  MPI_CHECK( MPI_Comm_free(&sh_comm) );
195 }
