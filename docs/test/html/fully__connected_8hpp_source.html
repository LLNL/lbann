184  void setup_data() override {
185  learning_layer::setup_data();
186 
187  // Initialize default weights if none are provided
188  if (this->m_weights.size() > 2) {
189  std::stringstream err;
190  err << __FILE__ << " " << __LINE__ << " :: "
191  << "attempted to setup " << m_name << " with an invalid number of weights";
192  throw lbann_exception(err.str());
193  }
194  this->m_weights.resize(2, nullptr);
195  if (this->m_weights[0] == nullptr) {
196  this->m_weights[0] = new weights(this->m_comm, this->m_cudnn);
197  this->m_weights[0]->set_name(this->m_name + "_linearity_weights");
198  this->m_weights[0]->set_initializer(new he_normal_initializer(this->m_comm));
199  this->m_weights[0]->set_optimizer(m_model->create_optimizer());
200  this->m_model->add_weights(this->m_weights[0]);
201  }
202  if (this->m_weights[1] == nullptr) {
203  this->m_weights[1] = new weights(this->m_comm, this->m_cudnn);
204  this->m_weights[1]->set_name(this->m_name + "_bias_weights");
205  this->m_weights[1]->set_optimizer(m_model->create_optimizer());
206  this->m_model->add_weights(this->m_weights[1]);
207  }
208 
209  // Initialize Glorot or He weight initialization
210  auto* cast_initializer
211  = dynamic_cast<fan_in_fan_out_initializer*>(&this->m_weights[0]->get_initializer());
212  if (cast_initializer != nullptr) {
213  cast_initializer->set_fan_in(this->m_num_prev_neurons);
214  cast_initializer->set_fan_out(this->m_num_neurons);
215  }
216 
217  // Setup weights
218  // Note: linearity matrix is duplicated across processes unless
219  // the data layout is model-parallel.
220  El::Distribution col_dist = El::STAR;
221  El::Distribution row_dist = El::STAR;
222  if (get_data_layout() == data_layout::MODEL_PARALLEL) {
223  col_dist = El::MC;
224  row_dist = El::MR;
225  }
226  this->m_weights[0]->setup(this->m_num_neurons,
227  this->m_num_prev_neurons,
228  col_dist, row_dist);
229  this->m_weights[1]->setup(this->m_num_neurons,
230  1,
231  get_activations().DistData().colDist,
232  El::STAR);
233 
234  // Setup weight gradients
235  El::Zeros(*this->m_linearity_gradient,
236  this->m_weights[0]->get_matrix_height(),
237  this->m_weights[0]->get_matrix_width());
238  El::Zeros(*this->m_bias_gradient,
239  this->m_weights[1]->get_matrix_height(),
240  this->m_weights[1]->get_matrix_width());
241 
242  if (m_frozen) {
243  this->m_weights[0]->freeze();
244  this->m_weights[1]->freeze();
245  } else {
246  if (this->m_weights[0]->is_frozen() || this->m_weights[1]->is_frozen()) {
247  throw lbann_exception("fully_connected_layer: layer is not frozen but weights are");
248  }
249  }
250  }

292  void fp_compute_cuda() {
293 #ifndef LBANN_HAS_CUDNN
294  throw lbann_exception("fully_connected: CUDA not detected");
295 #else
296 
297  // GPU matrices
298  const auto& linearity_d = m_weights[0]->get_values_gpu();
299  const auto& input_d = this->m_prev_activations_d[0];
300  auto& output_d = this->m_activations_d[0];
301 
302  // Matrix parameters
303  const int input_size = get_num_prev_neurons();
304  const int output_size = get_num_neurons();
305  const int mini_batch_size = m_mini_batch_size_per_gpu;
306  const int num_gpus = this->m_cudnn->get_num_gpus();
307  const int input_ldim = input_d.get_leading_dim();
308  const int output_ldim = output_d.get_leading_dim();
309 
310  // Stop early if possible
311  if (mini_batch_size == 0) { return; }
312 
313  // Apply linearity
314  for (int i=0; i<num_gpus; ++i) {
315  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
316  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
317  CUBLAS_OP_N, CUBLAS_OP_N,
318  output_size, mini_batch_size, input_size,
319  DataType(1),
320  linearity_d[i], output_size,
321  input_d.get_locked_data(i), input_ldim,
322  DataType(0),
323  output_d.get_data(i), output_ldim);
324  }
325 
326  // Apply bias if needed
327  if(m_bias_scaling_factor != DataType(0)) {
328  const auto& bias_d = m_weights[1]->get_values_gpu();
329 
330  // Initialize work space with ones
331  cudnn::matrix ones_d(this->m_cudnn);
332  ones_d.attach_to_work_spaces(mini_batch_size);
333  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), mini_batch_size);
334 
335  // Apply bias with outer product
336  for (int i = 0; i < num_gpus; ++i) {
337  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
338  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
339  CUBLAS_OP_N, CUBLAS_OP_T,
340  output_size, mini_batch_size, 1,
341  DataType(1),
342  bias_d[i], output_size,
343  ones_d.get_data(i), mini_batch_size,
344  DataType(1),
345  output_d.get_data(i), output_ldim);
346  }
347 
348  }
349 #endif // LBANN_HAS_CUDNN
350  }

353  void bp_compute_cuda() {
354 #ifndef LBANN_HAS_CUDNN
355  throw lbann_exception("fully_connected: CUDA not detected");
356 #else
357 
358  // GPU matrices
359  const auto& linearity_d = m_weights[0]->get_values_gpu();
360  const auto& input_d = this->m_prev_activations_d[0];
361  const auto& gradient_wrt_output_d = this->m_prev_error_signals_d[0];
362  auto& gradient_wrt_input_d = this->m_error_signals_d[0];
363 
364  // Matrix parameters
365  const int input_size = get_num_prev_neurons();
366  const int output_size = get_num_neurons();
367  const int mini_batch_size = m_mini_batch_size_per_gpu;
368  const int num_gpus = this->m_cudnn->get_num_gpus();
369  const int input_ldim = input_d.get_leading_dim();
370  const int gradient_wrt_output_ldim = gradient_wrt_output_d.get_leading_dim();
371  const int gradient_wrt_input_ldim = gradient_wrt_input_d.get_leading_dim();
372 
373  // Compute gradient w.r.t. bias if needed
374  optimizer* bias_optimizer = this->m_weights[1]->get_optimizer();
375  if (m_bias_scaling_factor != DataType(0)
376  && bias_optimizer != nullptr) {
377 
378  // Initialize work space with ones
379  cudnn::matrix ones_d(this->m_cudnn);
380  ones_d.attach_to_work_spaces(mini_batch_size);
381  m_cudnn->set_on_gpus(ones_d.get_data(), DataType(1), mini_batch_size);
382 
383  // Obtain gradient with a sum over rows
384  for (int i = 0; i < num_gpus; ++i) {
385  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
386  cublas::gemv(this->m_cudnn->get_cublas_handle(i),
387  CUBLAS_OP_N,
388  output_size, mini_batch_size,
389  DataType(1),
390  gradient_wrt_output_d.get_locked_data(i), gradient_wrt_output_ldim,
391  ones_d.get_data(i), 1,
392  DataType(0),
393  m_bias_gradient_d.get_data(i), 1);
394  }
395  bias_optimizer->add_to_gradient_staging(
396  m_bias_gradient_d,
397  m_bias_scaling_factor / this->m_model->get_effective_mini_batch_size());
398  }
399 
400  // Compute gradient w.r.t. linearity if needed
401  optimizer* linearity_optimizer = this->m_weights[0]->get_optimizer();
402  if (linearity_optimizer != nullptr) {
403  for (int i = 0; i < num_gpus; ++i) {
404  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
405  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
406  CUBLAS_OP_N, CUBLAS_OP_T,
407  output_size, input_size, mini_batch_size,
408  DataType(1),
409  gradient_wrt_output_d.get_locked_data(i), gradient_wrt_output_ldim,
410  input_d.get_locked_data(i), input_ldim,
411  DataType(0),
412  m_linearity_gradient_d.get_data(i), output_size);
413  }
414  linearity_optimizer->add_to_gradient_staging(
415  m_linearity_gradient_d,
416  DataType(1) / this->m_model->get_effective_mini_batch_size());
417  }
418 
419  // Compute gradient w.r.t. input
420  if (mini_batch_size != 0) {
421  for (int i = 0; i < num_gpus; ++i) {
422  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
423  cublas::gemm(this->m_cudnn->get_cublas_handle(i),
424  CUBLAS_OP_T, CUBLAS_OP_N,
425  input_size, mini_batch_size, output_size,
426  DataType(1),
427  linearity_d[i], output_size,
428  gradient_wrt_output_d.get_locked_data(i), gradient_wrt_output_ldim,
429  DataType(1),
430  gradient_wrt_input_d.get_data(i), gradient_wrt_input_ldim);
431  }
432  }
433 
434 #endif // LBANN_HAS_CUDNN
435  }
