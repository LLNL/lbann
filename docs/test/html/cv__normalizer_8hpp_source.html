209 template<class InputIterator, class OutputIterator>
210 inline OutputIterator cv_normalizer::scale(
211  InputIterator first, InputIterator last, OutputIterator result,
212  const std::vector<channel_trans_t> trans) {
213  const size_t NCh = trans.size();
214  bool trivial_alpha = true;
215  bool trivial_beta = true;
216 
217  for (size_t ch=0u; ch < NCh; ++ch) {
218  trivial_alpha = trivial_alpha && (trans[ch].first == 1.0);
219  trivial_beta = trivial_beta && (trans[ch].second == 0.0);
220  }
221 
222  if (trivial_alpha && trivial_beta) {
223  if ((typeid(*first) == typeid(*result)) &&
224  (reinterpret_cast<const void *>(&(*first)) ==
225  reinterpret_cast<const void *>(&(*result))))
226  // This way, it works both for iterator and for pointer
227  {
228  std::advance(result, std::distance(first,last));
229  return result;
230  } else {
231  return std::copy(first, last, result);
232  }
233  }
234 
235  using T = typename std::iterator_traits<OutputIterator>::value_type;
236 
237  // At this point NCh should not be zero because both alpha and beta are not trivial.
238  if (NCh == 1) {
239  const ComputeType a = trans[0].first;
240  const ComputeType b = trans[0].second;
241 
242  while (first != last) {
243  *result = cv::saturate_cast<T>(a * (*first) + b);
244  ++result;
245  ++first;
246  }
247  } else {
248  size_t ch = 0u;
249 
250  while (first != last) {
251  *result = cv::saturate_cast<T>(trans[ch].first * (*first) + trans[ch].second);
252  ++result;
253  ++first;
254  ++ch;
255  ch = (ch % NCh);
256  }
257  }
258  return result;
259 }

272 template<typename Tsrc, typename Tdst>
273 inline bool cv_normalizer::scale_with_known_type(cv::Mat& image,
274  const std::vector<channel_trans_t>& trans) {
275  const auto Width = static_cast<unsigned int>(image.cols);
276  const auto Height = static_cast<unsigned int>(image.rows);
277  const auto NCh = static_cast<unsigned int>(image.channels());
278  if ((trans.size() > 0u) && (trans.size() != NCh)) {
279  return false;
280  }
281 
282 
283  // overwrite the storage of the source image if the source and the result have
284  // the same data type. Otherwise, create a new image for the result. The result
285  // will replace the image referenced by the input.
286  if (std::is_same<Tsrc, Tdst>::value) {
287  if (image.isContinuous()) {
288  scale(reinterpret_cast<const Tsrc *>(image.datastart),
289  reinterpret_cast<const Tsrc *>(image.dataend),
290  reinterpret_cast<Tsrc *>(image.data), trans);
291  } else {
292  // TODO: Should we make this to copy to a new continuous block instead of
293  // updating the values in-place?
294  const unsigned int stride = Width*NCh;
295  for (unsigned int i = 0u; i < Height; ++i) {
296  auto *optr = reinterpret_cast<Tsrc *>(image.ptr<Tsrc>(i));
297  const Tsrc *iptr = optr;
298  scale(iptr, iptr+stride, optr, trans);
299  }
300  }
301  } else {
302  cv::Mat image_out = cv::Mat(Height, Width, CV_MAKETYPE(cv::DataType<Tdst>::depth, NCh));
303 
304  if (image.isContinuous()) {
305  scale(reinterpret_cast<const Tsrc *>(image.datastart),
306  reinterpret_cast<const Tsrc *>(image.dataend),
307  reinterpret_cast<Tdst *>(image_out.data), trans);
308  } else {
309  const unsigned int stride = Width*NCh;
310  auto *ptr_out = reinterpret_cast<Tdst *>(image_out.data);
311  for (unsigned int i = 0u; i < Height; ++i, ptr_out += stride) {
312  const Tsrc *ptr = reinterpret_cast<Tsrc *>(image.ptr<Tsrc>(i));
313  scale(ptr, ptr+stride, ptr_out, trans);
314  }
315  }
316  image = image_out;
317  }
318  return true;
319 }

326 template<typename T>
327 inline bool cv_normalizer::compute_mean_stddev_with_known_type(const cv::Mat& image,
328  std::vector<ComputeType>& mean, std::vector<ComputeType>& stddev, cv::InputArray mask) {
329  mean.clear();
330  stddev.clear();
331  if (image.empty()) {
332  return false;
333  }
334 
335  const int NCh = image.channels();
336  const int num_pixels = image.rows * image.cols;
337  ComputeType sum[NCh];
338  ComputeType sqsum[NCh];
339  ComputeType shift[NCh];
340 
341  for (int ch = 0; ch < NCh; ++ch) {
342  sum[ch] = 0.0;
343  sqsum[ch] = 0.0;
344  const auto *ptr = reinterpret_cast<const T *>(image.datastart);
345  shift[ch] = static_cast<ComputeType>(*(ptr+ch));
346  }
347 
348  mean.resize(NCh);
349  stddev.resize(NCh);
350 
351  if (image.isContinuous()) {
352  const auto *ptr = reinterpret_cast<const T *>(image.datastart);
353  const auto *const ptrend = reinterpret_cast<const T *>(image.dataend);
354 
355  int ch = 0;
356  do {
357  const ComputeType diff = (*ptr - shift[ch]);
358  sum[ch] += diff;
359  sqsum[ch] += diff*diff;
360  ++ch;
361  ch = ch % NCh;
362  } while ((++ptr) != ptrend);
363 
364  for (int c = 0; c < NCh; ++c) {
365  const ComputeType shifted_mean = sum[c] / num_pixels;
366  mean[c] = shifted_mean + shift[c];
367  stddev[c] = sqrt(std::max(sqsum[c]/num_pixels - shifted_mean * shifted_mean, ComputeType(0)));
368  }
369  } else {
370  const int stride = image.cols*NCh;
371  const int Height = image.rows;
372 
373  for (int i = 0; i < Height; ++i) {
374  const auto *ptr = reinterpret_cast<const T *>(image.ptr<const T>(i));
375  const T *const ptrend = ptr + stride;
376 
377  int ch = 0;
378  do {
379  const ComputeType diff = (*ptr - shift[ch]);
380  sum[ch] += diff;
381  sqsum[ch] += diff*diff;
382  ++ch;
383  ch = ch % NCh;
384  } while ((++ptr) != ptrend);
385  }
386 
387  for (int ch = 0; ch < NCh; ++ch) {
388  const ComputeType shifted_mean = sum[ch] / num_pixels;
389  mean[ch] = shifted_mean + shift[ch];
390  stddev[ch] = sqrt(std::max(sqsum[ch]/num_pixels - shifted_mean*shifted_mean, ComputeType(0)));
391  }
392  }
393  return true;
394 }
