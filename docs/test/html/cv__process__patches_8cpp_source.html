80 bool cv_process_patches::preprocess(cv::Mat& image, std::vector<cv::Mat>& patches) {
81  bool ok = true;
82  patches.clear();
83 
84  ok = cv_process::preprocess(image, 0u, m_when_to_extract);
85  ok = ok && m_pd.extract_patches(image, patches);
86 
87  for (size_t i=0u; ok && (i < patches.size()); ++i) {
88  ok = cv_process::preprocess(patches[i], m_when_to_extract);
89  }
90 
91  return ok;
92 }
