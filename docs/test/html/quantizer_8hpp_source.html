328  inline int get_adaptive_quantization_threads(El::Int width) {
329  int num_threads = omp_get_max_threads();
330  if (width <= 64) {
331  num_threads = 2;
332  } else if (width <= 128) {
333  num_threads = 8;
334  } else if (width <= 256) {
335  num_threads = 12;
336  } else if (width <= 1024) {
337  num_threads = 24;
338  }
339  return std::min(omp_get_max_threads(), num_threads);
340  }
341 
351  inline int get_adaptive_quantization_copy_threads(El::Int width) {
352  int num_threads = get_adaptive_quantization_threads(width);
353  if (width >= 16384) {
354  num_threads /= 2;
355  }
356  return num_threads;
357  }
