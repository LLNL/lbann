99 void csv_reader::load() {
100  setup_ifstreams();
101  std::ifstream& ifs = *m_ifstreams[0];
102  // TODO: Only one (or a subset) of ranks should read this, then distribute
103  // the results to avoid every rank reading the same file.
104  std::string line;
105  // Parse the header to determine how many columns there are.
106  // Skip rows if needed.
107  skip_rows(ifs, m_skip_rows);
108  std::streampos header_start = ifs.tellg();
109  // TODO: Skip comment lines.
110  if (std::getline(ifs, line)) {
111  m_num_cols = std::count(line.begin(), line.end(), m_separator) + 1;
112  if (m_skip_cols >= m_num_cols) {
113  throw lbann_exception(
114  "csv_reader: asked to skip more columns than are present");
115  }
116  if (!m_disable_labels) {
117  if (m_label_col < 0) {
118  // Last column becomes the label column.
119  m_label_col = m_num_cols - 1;
120  }
121  if (m_label_col >= m_num_cols) {
122  throw lbann_exception(
123  "csv_reader: label column" + std::to_string(m_label_col) +
124  " is not present");
125  }
126  }
127  if (!m_disable_responses) {
128  if (m_response_col < 0) {
129  // Last column becomes the response column.
130  m_response_col = m_num_cols - 1;
131  }
132  if (m_response_col >= m_num_cols) {
133  throw lbann_exception(
134  "csv_reader: response column" + std::to_string(m_response_col) +
135  " is not present");
136  }
137  }
138  } else {
139  throw lbann_exception(
140  "csv_reader: failed to read header in " + get_data_filename());
141  }
142  if (ifs.eof()) {
143  throw lbann_exception(
144  "csv_reader: reached EOF after reading header");
145  }
146  // If there was no header, skip back to the beginning.
147  if (!m_has_header) {
148  ifs.clear();
149  ifs.seekg(header_start, std::ios::beg);
150  }
151  // Construct an index mapping each line (sample) to its offset.
152  // TODO: Skip comment lines.
153  // Used to count the number of label classes.
154  std::unordered_set<int> label_classes;
155  m_index.push_back(ifs.tellg());
156  int line_num = 0;
157  while (std::getline(ifs, line)) {
158  ++line_num;
159  if (line_num == 100000) break;
160  // Verify the line has the right number of columns.
161  if (std::count(line.begin(), line.end(), m_separator) + 1 != m_num_cols) {
162  throw lbann_exception(
163  "csv_reader: line " + std::to_string(line_num) +
164  " does not have right number of entries");
165  }
166  m_index.push_back(ifs.tellg());
167  // Extract the label.
168  if (!m_disable_labels) {
169  size_t cur_pos = 0;
170  for (int col = 0; col < m_num_cols; ++col) {
171  size_t end_pos = line.find_first_of(m_separator, cur_pos);
172  if (col == m_label_col) {
173  int label = m_label_transform(line.substr(cur_pos, end_pos - cur_pos));
174  label_classes.insert(label);
175  m_labels.push_back(label);
176  break;
177  }
178  cur_pos = end_pos + 1;
179  }
180  }
181  // Possibly extract the response.
182  if (!m_disable_responses) {
183  size_t cur_pos = 0;
184  for (int col = 0; col < m_num_cols; ++col) {
185  size_t end_pos = line.find_first_of(m_separator, cur_pos);
186  if (col == m_response_col) {
187  DataType response = m_response_transform(
188  line.substr(cur_pos, end_pos - cur_pos));
189  m_responses.push_back(response);
190  break;
191  }
192  cur_pos = end_pos + 1;
193  }
194  }
195  }
196  if (!ifs.eof()) {
197  //If we didn't get to EOF, something went wrong.
198  //throw lbann_exception(
199  //"csv_reader: did not reach EOF");
200  }
201  if (!m_disable_labels) {
202  // Do some simple validation checks on the classes.
203  // Ensure the elements begin with 0, and there are no gaps.
204  auto minmax = std::minmax_element(label_classes.begin(), label_classes.end());
205  if (*minmax.first != 0) {
206  throw lbann_exception(
207  "csv_reader: classes are not indexed from 0");
208  }
209  if (*minmax.second != (int) label_classes.size() - 1) {
210  throw lbann_exception(
211  "csv_reader: label classes are not contiguous");
212  }
213  m_num_labels = label_classes.size();
214  }
215  m_num_samples = m_index.size() - 1;
216  // End of file offset.
217  m_index.push_back(ifs.tellg());
218  // Seek back to the beginning.
219  ifs.clear();
220  ifs.seekg(0, std::ios::beg);
221  // Reset indices.
222  m_shuffled_indices.resize(m_num_samples);
223  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
224  select_subset_of_data();
225 }

254 std::vector<DataType> csv_reader::fetch_line(int data_id) {
255  std::string line = fetch_raw_line(data_id);
256  std::vector<DataType> parsed_line;
257  // Note: load already verified that every line is properly formatted.
258  size_t cur_pos = 0; // Current *start* of a column.
259  for (int col = 0; col < m_num_cols; ++col) {
260  // Note for last column, this returns std::npos, which substr handles.
261  size_t end_pos = line.find_first_of(m_separator, cur_pos);
262  // Skip columns if needed.
263  if (col < m_skip_cols) {
264  cur_pos = end_pos + 1;
265  continue;
266  }
267  // Note: This results in a copy, switch to string_view when we can.
268  std::string str_val = line.substr(cur_pos, end_pos - cur_pos);
269  cur_pos = end_pos + 1;
270  DataType val;
271  if (m_col_transforms.count(col)) {
272  val = m_col_transforms[col](str_val);
273  } else {
274  // No easy way to parameterize based on DataType, so always use double.
275  val = std::stod(str_val);
276  }
277  parsed_line.push_back(val);
278  }
279  return parsed_line;
280 }

282 std::vector<DataType> csv_reader::fetch_line_label_response(
283  int data_id) {
284  std::string line = fetch_raw_line(data_id);
285  std::vector<DataType> parsed_line;
286  // Note: load already verified that every line is properly formatted.
287  size_t cur_pos = 0; // Current *start* of a column.
288  for (int col = 0; col < m_num_cols; ++col) {
289  // Note for last column, this returns std::npos, which substr handles.
290  size_t end_pos = line.find_first_of(m_separator, cur_pos);
291  // Skip the label, response, and any columns if needed.
292  if ((!m_disable_labels && col == m_label_col) ||
293  (!m_disable_responses && col == m_response_col) ||
294  col < m_skip_cols) {
295  cur_pos = end_pos + 1;
296  continue;
297  }
298  // Note: This results in a copy, switch to string_view when we can.
299  std::string str_val = line.substr(cur_pos, end_pos - cur_pos);
300  cur_pos = end_pos + 1;
301  DataType val;
302  if (m_col_transforms.count(col)) {
303  val = m_col_transforms[col](str_val);
304  } else {
305  // No easy way to parameterize based on DataType, so always use double.
306  try {
307  val = std::stod(str_val);
308  } catch (std::invalid_argument& e) {
309  throw lbann_exception(
310  "csv_reader: could not convert '" + str_val + "'");
311  }
312  }
313  parsed_line.push_back(val);
314  }
315  return parsed_line;
316 }

318 std::string csv_reader::fetch_raw_line(int data_id) {
319  std::ifstream& ifs = *m_ifstreams[omp_get_thread_num()];
320  // Seek to the start of this datum's line.
321  ifs.seekg(m_index[data_id], std::ios::beg);
322  // Compute the length of the line to read, excluding newline.
323  std::streamsize cnt = m_index[data_id+1] - m_index[data_id] - 1;
324  // Read directly into a string buffer.
325  std::string line;
326  line.resize(cnt);
327  if (!ifs.read(&line[0], cnt)) {
328  throw lbann_exception(
329  "csv_reader: error on reading " + std::to_string(data_id));
330  }
331  return line;
332 }
