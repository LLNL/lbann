56 sgd& sgd::operator=(const sgd& other) {

61  // Copy velocity matrix

63  && m_velocity->DistData() == other.m_velocity->DistData()) {
64  El::Copy(*other.m_velocity, *m_velocity);

66  else {
67  if (m_velocity != nullptr) { delete m_velocity; }
68  m_velocity = other.m_velocity;
69  if (m_velocity != nullptr) { m_velocity = m_velocity->Copy(); }
70  }
71 
72  // Copy GPU data
73  #ifdef LBANN_HAS_CUDNN

75  const int height = other.m_weights->get_matrix_height();
76  const int width = other.m_weights->get_matrix_width();
77  m_cudnn->deallocate_on_gpus(m_velocity_d);

121 void sgd::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
122 
123  // SGD without momentum is just an Axpy
124  if (m_momentum == DataType(0)) {
125  El::Axpy(-m_learning_rate, gradient, values);
126  return;
127  }
128 
129  // Get local matrix data
130  const int local_height = values.LocalHeight();
131  const int local_width = values.LocalWidth();
132  DataType* __restrict__ values_buffer = values.Buffer();
133  const int values_ldim = values.LDim();
134  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
135  const int gradient_ldim = gradient.LDim();
136  DataType* __restrict__ velocity_buffer = m_velocity->Buffer();
137  const int velocity_ldim = m_velocity->LDim();
138 
139  // Check if matrix data is contiguous
140  if (values_ldim != local_height
141  || gradient_ldim != local_height
142  || velocity_ldim != local_height) {
143  // (Nesterov) momentum SGD for non-contiguous data
144  #pragma omp parallel for collapse(2)
145  for (int j=0; j<local_width; ++j) {
146  for (int i=0; i<local_height; ++i) {
147  const DataType g = gradient_buffer[i+j*gradient_ldim];
148  DataType& v = velocity_buffer[i+j*velocity_ldim];
149  DataType& x = values_buffer[i+j*values_ldim];
150  v = m_momentum * v + g;
151  x -= (m_nesterov ?
152  m_learning_rate * (m_momentum * v + g) :
153  m_learning_rate * v);
154  }
155  }
156  } else {
157  if (m_nesterov) {
158  // Nesterov's accelerated gradient descent for contiguous data
159  #pragma omp parallel for
160  for (int i=0; i<local_height*local_width; ++i) {
161  DataType& x = values_buffer[i];
162  const DataType g = gradient_buffer[i];
163  DataType& v = velocity_buffer[i];
164  v = m_momentum * v + g;
165  x -= m_learning_rate * (m_momentum * v + g);
166  }
167  } else {
168  // Momentum SGD for contiguous data
169 #pragma omp parallel for
170  for (int i=0; i<local_height*local_width; ++i) {
171  DataType& x = values_buffer[i];
172  const DataType g = gradient_buffer[i];
173  DataType& v = velocity_buffer[i];
174  v = m_momentum * v + g;
175  x -= m_learning_rate * v;
176  }
177  }
178  }
