60  void setup_pointers() override {
61  // Check that pooling layer is valid
62  if(m_pooling_layer == nullptr) {
63  throw lbann_exception("unpooling_layer: no paired pooling layer");
64  }

73  void setup_dims() override {
74 
75  // Initialize previous neuron tensor dimensions
76  transform_layer::setup_dims();
77 
78  // Check that previous neuron tensor is valid
79  if(this->m_num_prev_neurons != m_pooling_layer->m_num_neurons
80  || this->m_num_prev_neuron_dims != m_pooling_layer->m_num_neuron_dims
81  || this->m_prev_neuron_dims != m_pooling_layer->m_neuron_dims) {
82  throw lbann_exception("unpooling_layer: previous neuron tensor must match neuron tensor of corresponding pooling layer");

135  void fp_compute_im2col() {
136 
137  // Get local matrices
138  const Mat& prev_activations_local = get_local_prev_activations();
139  Mat& activations_local = get_local_activations();
140 
141  // Get parameters
142  const int local_width = prev_activations_local.Width();
143  const int num_channels = this->m_prev_neuron_dims[0];
144  const int num_per_input_channel = this->m_num_prev_neurons / num_channels;
145  const int pool_size = m_pooling_layer->m_pool_size;
146 
147  // Initialize im2col matrix
148  Mat im2col_mat(pool_size * num_channels, num_per_input_channel);
149 
150  // Iterate through data samples
151  for(int sample = 0; sample < local_width; ++sample) {
152 
153  // Clear im2col matrix
154  El::Zero(im2col_mat);
155 
156  // Populate im2col matrix
157  const DataType *prev_activations_buffer
158  = prev_activations_local.LockedBuffer(0, sample);
159  const int *indices_buffer
160  = &m_pooling_layer->m_max_pool_indices[sample * this->m_num_prev_neurons];
161  #pragma omp parallel for
162  for(int channel = 0; channel < num_channels; ++channel) {
163  for(int j = 0; j < num_per_input_channel; ++j) {
164  const int input_index = j + channel * num_per_input_channel;
165  const int max_index = indices_buffer[input_index];
166  DataType *im2col_buffer
167  = im2col_mat.Buffer(channel * pool_size, j);
168  im2col_buffer[max_index]
169  = prev_activations_buffer[input_index];
170  }
171  }
172 
173  // Convert im2col matrix to output matrix
174  Mat output_mat = El::View(activations_local, El::ALL, El::IR(sample));
175  col2im(im2col_mat,
176  output_mat,
177  num_channels,
178  this->m_num_neuron_dims - 1,
179  this->m_neuron_dims.data() + 1,
180  m_pooling_layer->m_pads.data(),
181  m_pooling_layer->m_pool_dims.data(),
182  m_pooling_layer->m_strides.data(),
183  static_cast<const DataType&(*)(const DataType&,const DataType&)>(&std::max<DataType>));
184 
185  }
186 
187  }

190  void bp_compute_im2col() {
191 
192  // Get local matrices
193  const Mat& prev_error_signal_local = get_local_prev_error_signals();
194  Mat& error_signal_local = get_local_error_signals();
195 
196  // Get parameters
197  const int local_width = prev_error_signal_local.Width();
198  const int num_channels = this->m_prev_neuron_dims[0];
199  const int num_per_output_channel = this->m_num_prev_neurons / num_channels;
200  const int pool_size = m_pooling_layer->m_pool_size;
201 
202  // Initialize im2col matrix
203  Mat im2col_mat(pool_size * num_channels, num_per_output_channel);
204 
205  // Iterate through data samples
206  for(int sample = 0; sample < local_width; ++sample) {
207 
208  // Construct im2col matrix from input
209  const Mat input_mat = El::LockedView(prev_error_signal_local,
210  El::ALL, El::IR(sample));
211  im2col(input_mat,
212  im2col_mat,
213  num_channels,
214  this->m_num_neuron_dims - 1,
215  this->m_neuron_dims.data() + 1,
216  m_pooling_layer->m_pads.data(),
217  m_pooling_layer->m_pool_dims.data(),
218  m_pooling_layer->m_strides.data());
219 
220  // Propagate error signal based on pooling layer
221  DataType *output_buffer = error_signal_local.Buffer(0, sample);
222  const int *indices_buffer
223  = &m_pooling_layer->m_max_pool_indices[sample * this->m_num_prev_neurons];
224  #pragma omp parallel for
225  for(int channel = 0; channel < num_channels; ++channel) {
226  for(int j = 0; j < num_per_output_channel; ++j) {
227  const int output_index = j + channel * num_per_output_channel;
228  const int max_index = indices_buffer[output_index];
229  DataType *im2col_buffer
230  = im2col_mat.Buffer(channel * pool_size, j);
231  output_buffer[output_index] = im2col_buffer[max_index];
232  }
233  }
234 
235  }
236 
237  }
