107 void image_data_reader::load() {
108  //const std::string imageDir = get_file_dir();
109  const std::string imageListFile = get_data_filename();
110 
111  m_image_list.clear();
112 
113  // load image list
114  FILE *fplist = fopen(imageListFile.c_str(), "rt");
115  if (!fplist) {
116  throw lbann_exception(
117  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
118  " :: failed to open: " + imageListFile);
119  }
120 
121  while (!feof(fplist)) {
122  char imagepath[512];
123  label_t imagelabel;
124  if (fscanf(fplist, "%s%d", imagepath, &imagelabel) <= 1) {
125  break;
126  }
127  m_image_list.emplace_back(imagepath, imagelabel);
128  }
129  fclose(fplist);
130 
131  // reset indices
132  m_shuffled_indices.clear();
133  m_shuffled_indices.resize(m_image_list.size());
134  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
135 
136  select_subset_of_data();
137 }
