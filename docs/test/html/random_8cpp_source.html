106 void init_random(int seed, lbann_comm *comm) {
107  if (seed != -1) {
108  // Seed every OpenMP thread, if present.
109  // Note: Threadprivate OMP variables don't work with dynamic threads.
110 #ifdef _OPENMP
111  #pragma omp parallel
112  {
113  get_generator().seed((seed << 8) | omp_get_thread_num());
114  get_fast_generator().seed((seed << 8) | omp_get_thread_num());
115  }
116 #else
117  get_generator().seed(seed);
118  get_fast_generator().seed(seed);
119 #endif
120 #ifdef LBANN_SET_EL_RNG
121  if (comm != nullptr) {
122  El::Generator().seed(seed ^ comm->get_rank_in_model());
123  } else {
124  El::Generator().seed(seed ^ El::mpi::Rank(El::mpi::COMM_WORLD));
125  }
126 #endif
127  } else {
128  // Seed with a random value.
129  std::random_device rd;
130  unsigned rand_val = rd();
131 #ifdef _OPENMP
132  #pragma omp parallel
133  {
134  get_generator().seed((rand_val << 8) | omp_get_thread_num());
135  get_fast_generator().seed((rand_val << 8) | omp_get_thread_num());
136  }
137 #else
138  get_generator().seed(rand_val);
139  get_fast_generator().seed(rand_val);
140 #endif
141 #ifdef LBANN_SET_EL_RNG
142  El::Generator().seed(rand_val);
143 #endif
144  }
145 }

192 void gaussian_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType mean,
193  DataType stddev) {
194  El::Zeros(mat, m, n);
195  if (mat.Grid().Rank() == 0) {
196  mat.Reserve(n * m);
197  auto& gen = get_generator();
198  std::normal_distribution<DataType> dist(mean, stddev);
199  for (El::Int col = 0; col < n; ++col) {
200  for (El::Int row = 0; row < m; ++row) {
201  mat.QueueUpdate(row, col, dist(gen));
202  }
203  }
204  }
205  mat.ProcessQueues();
206 }

208 void bernoulli_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, double p) {
209  El::Zeros(mat, m, n);
210  if (mat.Grid().Rank() == 0) {
211  mat.Reserve(m * n);
212  auto& gen = get_generator();
213  std::bernoulli_distribution dist(p);
214  for (El::Int col = 0; col < n; ++col) {
215  for (El::Int row = 0; row < m; ++row) {
216  mat.QueueUpdate(row, col, dist(gen) ? 1.0f : 0.0f);
217  }
218  }
219  }
220  mat.ProcessQueues();
221 }

223 void uniform_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType center,
224  DataType radius) {
225  El::Zeros(mat, m, n);
226  if (mat.Grid().Rank() == 0) {
227  mat.Reserve(n * m);
228  auto& gen = get_generator();
229  std::uniform_real_distribution<DataType> dist(center - radius,
230  center + radius);
231  for (El::Int col = 0; col < n; ++col) {
232  for (El::Int row = 0; row < m; ++row) {
233  mat.QueueUpdate(row, col, dist(gen));
234  }
235  }
236  }
237  mat.ProcessQueues();
238 }
