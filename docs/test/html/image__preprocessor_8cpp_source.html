55 void lbann_image_preprocessor::augment(Mat& pixels, unsigned imheight,
56  unsigned imwidth,
57  unsigned num_channels) {
58  bool do_transform = m_horizontal_flip || m_vertical_flip ||
59  m_rotation_range || m_horizontal_shift || m_vertical_shift ||
60  m_shear_range;
61  if (do_transform) {
62  cv::Mat sqpixels = cv_pixels(pixels, imheight, imwidth, num_channels);
63  rng_gen& gen = get_generator();
64  std::uniform_int_distribution<int> bool_dist(0, 1);
65  // Flips.
66  bool horiz_flip = bool_dist(gen) && m_horizontal_flip;
67  bool vert_flip = bool_dist(gen) && m_vertical_flip;
68  if (horiz_flip || vert_flip) {
69  if (horiz_flip && !vert_flip) {
70  flip(sqpixels, 1);
71  } else if (!horiz_flip && vert_flip) {
72  flip(sqpixels, 0);
73  } else {
74  flip(sqpixels, -1);
75  }
76  }
77  // Translations.
78  float x_trans = 0.0f;
79  float y_trans = 0.0f;
80  if (m_horizontal_shift) {
81  std::uniform_real_distribution<float> dist(-m_horizontal_shift,
82  m_horizontal_shift);
83  x_trans = dist(gen) * imwidth;
84  }
85  if (m_vertical_shift) {
86  std::uniform_real_distribution<float> dist(-m_vertical_shift,
87  m_vertical_shift);
88  y_trans = dist(gen) * imheight;
89  }
90  Mat trans_mat;
91  El::Diagonal(trans_mat, std::vector<DataType>({1.0f, 1.0f, 1.0f}));
92  trans_mat(0, 2) = x_trans;
93  trans_mat(1, 2) = y_trans;
94  // Shearing.
95  float shear = 0.0f;
96  if (m_shear_range) {
97  std::uniform_real_distribution<float> dist(-m_shear_range,
98  m_shear_range);
99  shear = dist(gen);
100  }
101  Mat shear_mat;
102  El::Zeros(shear_mat, 3, 3);
103  shear_mat(0, 0) = 1.0f;
104  shear_mat(2, 2) = 1.0f;
105  shear_mat(0, 1) = -std::sin(shear);
106  shear_mat(1, 1) = std::cos(shear);
107  // Rotation.
108  float rotate = 0.0f;
109  if (m_rotation_range) {
110  std::uniform_real_distribution<float> dist(-m_rotation_range,
111  m_rotation_range);
112  rotate = ::pi / 180.0f * dist(gen);
113  }
114  Mat rot_mat;
115  El::Zeros(rot_mat, 3, 3);
116  rot_mat(2, 2) = 1.0f;
117  rot_mat(0, 0) = std::cos(rotate);
118  rot_mat(0, 1) = -std::sin(rotate);
119  rot_mat(1, 0) = std::sin(rotate);
120  rot_mat(1, 1) = std::cos(rotate);
121  // Compute the final transformation.
122  Mat affine_mat_tmp(3, 3);
123  Mat affine_mat(3, 3);
124  El::Gemm(El::NORMAL, El::NORMAL, (DataType) 1.0, trans_mat, shear_mat,
125  (DataType) 0.0, affine_mat_tmp);
126  El::Gemm(El::NORMAL, El::NORMAL, (DataType) 1.0, affine_mat_tmp, rot_mat,
127  (DataType) 0.0, affine_mat);
128  affine_trans(sqpixels, affine_mat);
129  col_pixels(sqpixels, pixels, num_channels);
130  }

133 void lbann_image_preprocessor::normalize(Mat& pixels, unsigned num_channels) {
134  if (m_z_score || (m_mean_subtraction && m_unit_variance)) {
135  z_score(pixels, num_channels);

149 void lbann_image_preprocessor::mean_subtraction(Mat& pixels,
150  unsigned num_channels) {
151  const unsigned height = pixels.Height();
152  const unsigned height_per_channel = height / num_channels;
153  for (unsigned channel = 0; channel < num_channels; ++channel) {
154  const unsigned channel_start = channel*height_per_channel;
155  const unsigned channel_end = (channel+1)*height_per_channel;
156  Mat pixels_channel = El::View(pixels, El::IR(channel_start, channel_end), El::ALL);
157  DataType mean, stdev;
158  entrywise_mean_and_stdev(pixels_channel, mean, stdev);
159  for (unsigned i = 0; i < height_per_channel; ++i) {
160  DataType& pixels_entry = pixels_channel(i, 0);
161  pixels_entry -= mean;
162  }
163  }
164 }

166 void lbann_image_preprocessor::unit_variance(
167  Mat& pixels, unsigned num_channels) {
168 
169  // Get image parameters
170  const unsigned height = pixels.Height();
171  const unsigned height_per_channel = height / num_channels;
172 
173  // Scale each channel separately
174  for (unsigned channel = 0; channel < num_channels; ++channel) {
175  const unsigned channel_start = channel*height_per_channel;
176  const unsigned channel_end = (channel+1)*height_per_channel;
177  Mat pixels_channel = El::View(pixels, El::IR(channel_start, channel_end), El::ALL);
178  DataType mean, stdev;
179  entrywise_mean_and_stdev(pixels_channel, mean, stdev);
180  if(stdev > DataType(1e-7)*std::abs(mean)) {
181  const DataType inv_stdev = 1 / stdev;
182  for (unsigned i = 0; i < height_per_channel; ++i) {
183  DataType& pixels_entry = pixels_channel(i, 0);
184  pixels_entry = (pixels_entry - mean) * inv_stdev + mean;
185  }
186  }
187  }

200 {
201  if(m_noise_factor){
202  Mat X_noise;
203  El::Gaussian(X_noise, pixels.Height(), pixels.Width(), DataType(0), DataType(1));
204  El::Axpy(m_noise_factor,X_noise,pixels);
205  //@todo - clip to min and max of input entry
206  auto clip = [](const DataType& z) {
207  return std::max(DataType(0), std::min(z,DataType(1)));

213 void lbann_image_preprocessor::z_score(Mat& pixels,
214  unsigned num_channels) {
215 
216  // Get image parameters
217  const unsigned height = pixels.Height();
218  const unsigned height_per_channel = height / num_channels;
219 
220  // Shift and scale each channel separately
221  for (unsigned channel = 0; channel < num_channels; ++channel) {
222  const unsigned channel_start = channel*height_per_channel;
223  const unsigned channel_end = (channel+1)*height_per_channel;
224  Mat pixels_channel = El::View(pixels, El::IR(channel_start, channel_end), El::ALL);
225  DataType mean, stdev;
226  entrywise_mean_and_stdev(pixels_channel, mean, stdev);
227  if(stdev > DataType(1e-7)*std::abs(mean)) {
228  const DataType inv_stdev = 1 / stdev;
229  for (unsigned i = 0; i < height_per_channel; ++i) {
230  DataType& pixels_entry = pixels_channel(i, 0);
231  pixels_entry = (pixels_entry - mean) * inv_stdev;
232  }
233  } else {
234  Zero(pixels_channel);
235  }
236  }

240 cv::Mat lbann_image_preprocessor::cv_pixels(const Mat& pixels,
241  unsigned imheight,
242  unsigned imwidth,
243  unsigned num_channels) {
244  if (num_channels == 1) {
245  cv::Mat m(imheight, imwidth, CV_32FC1);
246  for (unsigned y = 0; y < imheight; ++y) {
247  for (unsigned x = 0; x < imwidth; ++x) {
248  m.at<float>(y, x) = pixels(y * imwidth + x, 0);
249  }
250  }
251  return m;
252  } else if (num_channels == 3) {
253  cv::Mat m(imheight, imwidth, CV_32FC3);
254  for (unsigned y = 0; y < imheight; ++y) {
255  for (unsigned x = 0; x < imwidth; ++x) {
256  cv::Vec3f pixel;
257  unsigned offset = y * imwidth + x;
258  pixel[0] = pixels(offset, 0);
259  pixel[1] = pixels(offset + imheight*imwidth, 0);
260  pixel[2] = pixels(offset + 2*imheight*imwidth, 0);
261  m.at<cv::Vec3f>(y, x) = pixel;
262  }
263  }
264  return m;
265  } else {
266  throw lbann_exception(std::string{} + __FILE__ + " " +
267  std::to_string(__LINE__) +
268  "Only support 1 and 3 channels");
269  }

272 void lbann_image_preprocessor::col_pixels(const cv::Mat& sqpixels, Mat& pixels,
273  unsigned num_channels) {
274  unsigned imheight = sqpixels.rows;
275  unsigned imwidth = sqpixels.cols;
276  if (num_channels == 1) {
277  for (unsigned y = 0; y < imheight; ++y) {
278  for (unsigned x = 0; x < imwidth; ++x) {
279  pixels(y * imwidth + x, 0) = sqpixels.at<float>(y, x);
280  }
281  }
282  } else if (num_channels == 3) {
283  for (unsigned y = 0; y < imheight; ++y) {
284  for (unsigned x = 0; x < imwidth; ++x) {
285  cv::Vec3f pixel = sqpixels.at<cv::Vec3f>(y, x);
286  unsigned offset = y * imwidth + x;
287  pixels(offset, 0) = pixel[0];
288  pixels(offset + imheight*imwidth, 0) = pixel[1];
289  pixels(offset + 2*imheight*imwidth, 0) = pixel[2];
290  }
291  }
292  } else {
293  throw lbann_exception(std::string{} + __FILE__ + " " +
294  std::to_string(__LINE__) +
295  "Only support 1 and 3 channels");
296  }

305 void lbann_image_preprocessor::affine_trans(cv::Mat& sqpixels,
306  const Mat& trans) {
307  cv::Mat sqpixels_copy = sqpixels.clone();
308  // Construct the OpenCV transformation matrix.
309  cv::Mat cv_trans(2, 3, CV_32FC1);
310  cv_trans.at<float>(0, 0) = trans(0, 0);
311  cv_trans.at<float>(0, 1) = trans(0, 1);
312  cv_trans.at<float>(0, 2) = trans(0, 2);
313  cv_trans.at<float>(1, 0) = trans(1, 0);
314  cv_trans.at<float>(1, 1) = trans(1, 1);
315  cv_trans.at<float>(1, 2) = trans(1, 2);
316  cv::warpAffine(sqpixels_copy, sqpixels, cv_trans, sqpixels.size(),
317  cv::INTER_LINEAR, cv::BORDER_REPLICATE);
318 }

320 void lbann_image_preprocessor::internal_save_image(
321  Mat& pixels, const std::string filename, unsigned imheight, unsigned imwidth,
322  unsigned num_channels, bool do_scale) {
323  cv::Mat sqpixels = cv_pixels(pixels, imheight, imwidth, num_channels);
324  cv::Mat converted_pixels;
325  int dst_type = 0;
326  if (num_channels == 1) {
327  dst_type = CV_8UC1;
328  } else if (num_channels == 3) {
329  dst_type = CV_8UC3;
330  } // cv_pixels ensures no other case happens.
331  sqpixels.convertTo(converted_pixels, dst_type, do_scale ? 255.0f : 1.0f);
332  cv::imwrite(filename, converted_pixels);
333 }
