64 int lbann::DataGenerator::fetch_data(Mat& X)
65 {
66  if(!DataReader::position_valid()) {
67  stringstream err;
68  err << __FILE__<<" "<<__LINE__<< " :: Data Generator load error: !position_valid";
69  throw lbann_exception(err.str());
70  }
71 
72  int current_batch_size = getBatchSize();
73 
74  int n = 0;
75  for (n = CurrentPos; n < CurrentPos + current_batch_size; n++) {
76  if (n >= (int)ShuffledIndices.size())
77  break;
78 
79  int k = n - CurrentPos;
80  int index = ShuffledIndices[n];
81 
82  for (int p = 0; p < get_linearized_data_size(); p++) {
83  X.Set(p, k, m_data.GetLocal(p, index));
84  }
85  }
86 
87  return (n - CurrentPos);
88 }
