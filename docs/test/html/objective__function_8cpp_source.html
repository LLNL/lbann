84 EvalType objective_function::finish_evaluation(execution_mode mode,
85  int mini_batch_size) {
86  const auto start_time = get_time();
87  EvalType value = EvalType(0);
88  for (const auto& term : m_terms) {
89  value += term->finish_evaluation();
90  }
91  m_statistics[mode].add_value(mini_batch_size * value,
92  mini_batch_size);
93  m_evaluation_time += get_time() - start_time;
94  return value;

141 void objective_function::set_layer_pointers(std::vector<Layer*> layers) {
142  auto it = layers.begin();
143  for (objective_function_term *term : m_terms) {
144  const size_t num_layers = term->get_layer_pointers().size();
145  std::vector<Layer*> term_layers(it, it + num_layers);
146  term->set_layer_pointers(term_layers);
147  it += num_layers;
148  }
149  if (it != layers.end()) {
150  std::stringstream err;
151  err << __FILE__ << " " << __LINE__ << " :: "
152  << "attempted to set an invalid number of layer pointers "
153  << "(expected " << it - layers.begin() << ", "
154  << "found " << layers.size() << ")";
155  throw lbann_exception(err.str());
156  }
157 }
