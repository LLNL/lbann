40 template <data_layout T_layout>
41 class selu_dropout : public regularizer_layer {
42  public:
44  selu_dropout(lbann_comm *comm,
45  float keep_prob=0.95f,
46  DataType alpha = DataType(1.6732632423543772848170429916717),
47  DataType scale = DataType(1.0507009873554804934193349852946)) :
48  regularizer_layer(comm),
49  m_keep_prob(keep_prob),
50  m_mask(nullptr) {
51 #ifdef LBANN_PROCDET_DROPOUT
52  throw lbann_exception("selu_dropout: deterministic dropout not supported");
53 #endif
54  // Compute alpha' and the affine transform.
55  m_alpha_prime = -scale*alpha;
56  m_a = keep_prob +
57  m_alpha_prime*m_alpha_prime*keep_prob*(DataType(1) - keep_prob);
58  m_a = DataType(1) / std::sqrt(m_a);
59  m_b = -m_a * m_alpha_prime*(DataType(1) - keep_prob);
60  }
61 
62  selu_dropout(const selu_dropout& other) :
63  regularizer_layer(other),
64  m_alpha_prime(other.m_alpha_prime),
65  m_a(other.m_a),
66  m_b(other.m_b),
67  m_keep_prob(other.m_keep_prob),
68  m_mask(other.m_mask) {
69  if (m_mask != nullptr) { m_mask = m_mask->Copy(); }
70  }
71 
72  selu_dropout& operator=(const selu_dropout& other) {
73  regularizer_layer::operator=(other);
74  m_alpha_prime = other.m_alpha_prime;
75  m_a = other.m_a;
76  m_b = other.m_b;

78  if (m_mask != nullptr) { delete m_mask; }

80  if (m_mask != nullptr) { m_mask = m_mask->Copy(); }
81  return *this;
82  }

100  protected:
102  void fp_compute() override {
103  if (this->m_model->get_execution_mode() != execution_mode::training ||
104  m_keep_prob < 0.0f) {
105  // Do nothing if dropout is disabled
106  El::Copy(get_prev_activations(), get_activations());
107  } else {
108 
109  AbsDistMat *input_acts = &get_prev_activations();
110  const El::Int height = input_acts->Height();
111  const El::Int width = input_acts->Width();
112  const El::Int local_height = input_acts->LocalHeight();
113  const El::Int local_width = input_acts->LocalWidth();
114 
115  Mat& local_input_acts = input_acts->Matrix();
116  Mat& local_output_acts = get_local_activations();
117  Mat& local_mask = m_mask->Matrix();
118 
119  // Construct and apply mask and the affine transform.
120  // TODO: Optimize.
121  El::Bernoulli(*m_mask, height, width, m_keep_prob);
122  for (El::Int col = 0; col < local_width; ++col) {
123  for (El::Int row = 0; row < local_height; ++row) {
124  local_output_acts(row, col) = m_a *
125  (local_input_acts(row, col)*local_mask(row, col) +
126  m_alpha_prime*(1 - local_mask(row, col))) + m_b;
127  }
128  }
129 
130  }
131  }
132 
134  void bp_compute() override {
135  if (this->m_model->get_execution_mode() != execution_mode::training
136  || m_keep_prob < 0.0f) {
137  El::LockedView(get_error_signals(), get_prev_error_signals());
138  } else {
139 
140  Mat& local_prev_error_signal = get_local_prev_error_signals();
141  Mat& local_error_signal = get_local_error_signals();
142  Mat& local_mask = m_mask->Matrix();
143  const El::Int local_height = local_prev_error_signal.Height();
144  const El::Int local_width = local_prev_error_signal.Width();
145  // Reweight with the affine scale factor and the dropout mask.
146  for (El::Int col = 0; col < local_width; ++col) {
147  for (El::Int row = 0; row < local_height; ++row) {
148  local_error_signal(row, col) =
149  m_a * local_prev_error_signal(row, col) * local_mask(row, col);
150  }
151  }
152 
153  }
154  }
155 
156  private:
158  DataType m_alpha_prime;
160  DataType m_a;

164  float m_keep_prob;
166  AbsDistMat *m_mask;
167 };
