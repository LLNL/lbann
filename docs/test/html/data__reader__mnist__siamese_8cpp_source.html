90 int data_reader_mnist_siamese::fetch_data(Mat& X) {
91  int nthreads = omp_get_max_threads();
92  if(!position_valid()) {
93  throw lbann_exception(
94  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
95  + " :: " + get_type() + " load error: !position_valid"
96  + " -- current pos = " + std::to_string(m_current_pos)
97  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
98  }
99 
102  #pragma omp parallel for schedule(static, 1)
103  for (int t = 0; t < nthreads; t++) {
104  preprocess_data_source(omp_get_thread_num());
105  }
106 
107  int loaded_batch_size = get_loaded_mini_batch_size();
108  const int end_pos = std::min(static_cast<size_t>(m_current_pos+loaded_batch_size),
109  m_shuffled_indices.size());
110  const int mb_size = std::min(
111  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
112  X.Width());
113 
114  El::Zeros(X, X.Height(), X.Width());
115  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
116 
117  #pragma omp parallel for
118  for (int s = 0; s < mb_size; s++) {
119  // Catch exceptions within the OpenMP thread.
120  try {
121  int n = m_current_pos + (s * m_sample_stride);
122  sample_t index = std::make_pair(m_shuffled_indices[n], m_shuffled_indices2[n]);
123  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
124  if (!valid) {
125  throw lbann_exception(
126  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
127  " :: generic data reader load error: datum not valid");
128  }
129  El::Int index_coded = m_shuffled_indices[n] + m_shuffled_indices2[n]*(std::numeric_limits<label_t>::max()+1);
130  m_indices_fetched_per_mb.Set(s, 0, index_coded);
131  } catch (lbann_exception& e) {
132  lbann_report_exception(e);
133  } catch (std::exception& e) {
134  El::ReportException(e);
135  }
136  }
137 
140  #pragma omp parallel for schedule(static, 1)
141  for (int t = 0; t < nthreads; t++) {
142  postprocess_data_source(omp_get_thread_num());
143  }
144 
145  return mb_size;
146 }

198 bool data_reader_mnist_siamese::fetch_datum(Mat& X, std::pair<int, int> data_id, int mb_idx, int tid) {
199 
200  std::vector<::Mat> X_v = create_datum_views(X, mb_idx);
201 
202  using raw_data_t = std::vector<unsigned char>;
203  using local_sample_t = std::array<raw_data_t*, 2>;
204  local_sample_t sample;
205  sample[0] = &m_image_data[data_id.first];
206  sample[1] = &m_image_data[data_id.second];
207 
208  for(size_t i=0u; i < sample.size(); ++i) {
209  int width=0, height=0, img_type=0;
210  bool ret = true;
211 
212 #if 1
213  // Construct a zero copying view to a portion of a preloaded data buffer
214  // This has nothing to do with the image type but only to create view on a block of bytes
215  using InputBuf_T = lbann::cv_image_type<uint8_t>;
216  const cv::Mat image_buf(1, sample[i]->size()-1, InputBuf_T::T(1), &((*sample[i])[1]));
217 #else
218  raw_data_t image_buf(sample[i]->begin()+1, sample[i]->end()); // make copy of the raw data
219 #endif
220  ret = lbann::image_utils::import_image(image_buf, width, height, img_type, *(m_pps[tid]), X_v[i]);
221 
222  if(!ret) {
223  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
224  + get_type() + ": image_utils::import_image failed to load");
225  }
226  if((width * height * CV_MAT_CN(img_type)) != m_image_linearized_size) {
227  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " "
228  + get_type() + ": mismatch data size -- either width, height or channel - "
229  + " [w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
230  + "x" + std::to_string(CV_MAT_CN(img_type)) + "] != " + std::to_string(m_image_linearized_size));
231  }
232  }
233  return true;
234 }

265 void data_reader_mnist_siamese::load() {
266  if (is_master()) {
267  std::cerr << "starting lbann::" << get_type() << "::load\n";
268  }
269  m_image_data.clear();
270 
271  const std::string FileDir = get_file_dir();
272  const std::string ImageFile = get_data_filename();
273  const std::string LabelFile = get_label_filename();
274 
275  // set filepath
276  const std::string imagepath = FileDir + "/" + ImageFile;
277  const std::string labelpath = FileDir + "/" + LabelFile;
278 
279  if (is_master()) {
280  std::cerr << "read labels!\n";
281  }
282 
283  load_mnist_data(imagepath, labelpath, m_first_n, m_image_data);
284 
285  if (m_first_n > 0) {
286  set_use_percent(1.0);
287  set_absolute_sample_count(0u);
288  }
289 
290  // reset indices
291  m_shuffled_indices.clear();
292  m_shuffled_indices.resize(m_image_data.size());
293  for (size_t n = 0; n < m_shuffled_indices.size(); n++) {
294  m_shuffled_indices[n] = n;
295  }
296  if (is_master()) {
297  std::cerr << "calling select_subset_of_data; m_shuffled_indices.size: " <<
298  m_shuffled_indices.size() << std::endl;
299  }
300  select_subset_of_data();
301 }

304 void data_reader_mnist_siamese::shuffle_indices() {
305  if (m_shuffled_indices2.size() != m_shuffled_indices.size()) {
306  m_shuffled_indices2 = m_shuffled_indices;
307  if (!m_shuffle) {
308  std::shuffle(m_shuffled_indices2.begin(), m_shuffled_indices2.end(),
309  get_data_seq_generator());
310  }
311  }
312  if (m_shuffle) {
313  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
314  get_data_seq_generator());
315  std::shuffle(m_shuffled_indices2.begin(), m_shuffled_indices2.end(),
316  get_data_seq_generator());
317  }
