47 void data_store_pilot2_molecular::setup() {
48  double tm1 = get_time();
49  std::stringstream err;
50  m_owner = (int)m_reader->get_compound_rank() == (int)m_rank;
51  m_owner_rank = m_reader->get_compound_rank();
52 
53  if (m_owner) std::cerr << "starting data_store_pilot2_molecular::setup() for role: "
54  << m_reader->get_role() << "; owning processor: " << m_owner_rank << std::endl;
55  if (m_owner) std::cerr << "calling generic_data_store::setup()\n";
56  generic_data_store::setup();
57 
58  if (! m_in_memory) {
59  err << __FILE__ << " " << __LINE__ << " :: "
60  << "not yet implemented";
61  throw lbann_exception(err.str());
62  }
63 
64  else {
65  //sanity check
66  pilot2_molecular_reader *reader = dynamic_cast<pilot2_molecular_reader*>(m_reader);
67  if (reader == nullptr) {
68  err << __FILE__ << " " << __LINE__ << " :: "
69  << "dynamic_cast<data_reader_pilot2_molecular*>(m_reader) failed";
70  throw lbann_exception(err.str());
71  }
72  m_pilot2_reader = reader;
73 
74  // get list of indices used in calls to generic_data_reader::fetch_data
75  // for this processor
76  get_minibatch_index_vector();
77 
78  // get list of indices used in calls to generic_data_reader::fetch_data
79  // for all processors
80  if (m_master) std::cerr << "calling exchange_mb_indices\n";
81  exchange_mb_indices();
82 
83  // allocate storage for the data that will be passed to the data reader's
84  // fetch_datum method.
85  m_data_buffer.resize(omp_get_max_threads());
86  int num_features = m_pilot2_reader->get_num_features();
87  int num_neighbors = m_pilot2_reader->get_num_neighbors();
88  for (size_t j=0; j<m_data_buffer.size(); j++) {
89  m_data_buffer[j].resize(num_features * (num_neighbors+1));
90  }
91 
92  if (m_owner) {
93  std::cerr << "calling construct_data_store()\n";
94  construct_data_store();
95  }
96 
97  if (m_owner) std::cerr << "calling build_nabor_map()\n";
98  build_nabor_map();
99 
100  if (m_owner) std::cerr << "calling exchange_data()\n";
101  exchange_data();
102  }
103 
104  if (m_owner) {
105  std::cerr << "data_store_pilot2_molecular::setup time: " << get_time() - tm1 << "\n";
106  }

131 void data_store_pilot2_molecular::fill_in_data(
132  const int data_id,
133  const int num_samples_per_frame,
134  const int num_features,
135  double *features) {
136  const int frame = m_pilot2_reader->get_frame(data_id);
137  const int frame_offset = frame * num_features * num_samples_per_frame;
138  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
139  double *data = features + frame_offset + intra_frame_data_id * num_features;
140  if (m_data.find(data_id) != m_data.end()) {
141  std::stringstream err;
142  err << __FILE__ << " :: " << __LINE__ << " :: "
143  << " duplicate data_id: " << data_id;
144  throw lbann_exception(err.str());
145  }
146  m_data[data_id].resize(num_features);
147  for (int i=0; i<num_features; i++) {
148  m_data[data_id][i] = m_pilot2_reader->scale_data<double>(i, data[i]);
149  }
150 }

155 void data_store_pilot2_molecular::build_nabor_map() {
156  //bcast neighbor data
157  int sz;
158  if (m_owner) sz= m_pilot2_reader->get_neighbors_data_size();
159  MPI_Bcast(&sz, 1, MPI_INT, m_owner_rank, m_mpi_comm);
160  double *neighbors_8;
161  std::vector<double> work;
162  if (m_owner) {
163  neighbors_8 = m_pilot2_reader->get_neighbors_8();
164  } else {
165  work.resize(sz);
166  neighbors_8 = work.data();
167  }
168  MPI_Bcast(neighbors_8, sz, MPI_DOUBLE, m_owner_rank, m_mpi_comm);
169 
170  //fill in the nabors map
171  for (auto data_id : (*m_shuffled_indices)) {
172  int frame = m_pilot2_reader->get_frame(data_id);
173  int max_neighborhood = m_pilot2_reader->get_max_neighborhood();
174  int num_samples_per_frame = m_pilot2_reader->get_num_samples_per_frame();
175  const int neighbor_frame_offset = frame * num_samples_per_frame * (2 * max_neighborhood);
176  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
177  int num_neighbors = m_pilot2_reader->get_num_neighbors();
178  m_neighbors[data_id].reserve(num_neighbors);
179  double *neighbor_data = neighbors_8 + neighbor_frame_offset + intra_frame_data_id * (2 * max_neighborhood);
180  for (int i=1; i<num_neighbors + 1; ++i) {
181  int neighbor_id = neighbor_data[i];
182  m_neighbors[data_id].push_back(neighbor_id);
183  }
184  }
185 }

187 void data_store_pilot2_molecular::get_data_buf(int data_id, int tid, std::vector<double> *&buf) {
188  std::stringstream err;
189  std::vector<double> &v = m_data_buffer[tid];
190  std::fill(v.begin(), v.end(), 0.0);
191  if (m_neighbors.find(data_id) == m_neighbors.end()) {
192  err << __FILE__ << " " << __LINE__ << " :: "
193  << data_id << " not found in m_neighbors (primary molecule)";
194  throw lbann_exception(err.str());
195  }
196  if (m_my_molecules.find(data_id) == m_my_molecules.end()) {
197  err << __FILE__ << " " << __LINE__ << " :: "
198  << data_id << " not found in m_my_molecules";
199  throw lbann_exception(err.str());
200  }
201 
202  //fill in data for the primary molecule
203  size_t jj = 0;
204  std::vector<double> &d1 = m_my_molecules[data_id];
205  for (size_t j=0; j<d1.size(); j++) {
206  v[jj++] = d1[j];
207  }
208 
209  //fill in data for the primary molecule's neighbor
210  std::vector<int> &nabors = m_neighbors[data_id];
211  int num_features = m_pilot2_reader->get_num_features();
212  for (size_t h=0; h<nabors.size(); h++) {
213  if (nabors[h] != -1) {
214  if (m_my_molecules.find(data_id) == m_my_molecules.end()) {
215  err << __FILE__ << " " << __LINE__ << " :: "
216  << nabors[h] << " not found in m_my_molecules (neighbor)";
217  throw lbann_exception(err.str());
218  }
219  std::vector<double> &d2 = m_my_molecules[nabors[h]];
220  for (size_t i=0; i<d2.size(); i++) {
221  v[jj++] = d2[i];
222  }
223  } else {
224  jj += num_features;
225  }
226  }
227 
228  buf = &v;
229 }

231 void data_store_pilot2_molecular::get_required_molecules(std::unordered_set<int> &required_molecules, int p) {
232  required_molecules.clear();
233  std::vector<int> &v = m_all_minibatch_indices[p];
234  for (auto t : v) {
235  int data_id = (*m_shuffled_indices)[t];
236  required_molecules.insert(data_id);
237  if (m_neighbors.find(data_id) == m_neighbors.end()) {
238  std::stringstream err;
239  err << __FILE__ << " :: " << __LINE__ << " :: "
240  << " m_neighbors.find(" << data_id << " failed";
241  throw lbann_exception(err.str());
242  }
243  for (auto t2 : m_neighbors[data_id]) {
244  if (t2 != -1) {
245  required_molecules.insert(t2);
246  }
247  }
248  }

251 void data_store_pilot2_molecular::exchange_data() {
252  double tm1 = get_time();
253  std::stringstream err;
254 
255  //get set of molecules required for the next epoch for myself
256  std::unordered_set<int> required_molecules;
257  get_required_molecules(required_molecules, m_rank);
258 
259  //start receives for my required molecules
260  m_my_molecules.clear();
261  int num_features = m_pilot2_reader->get_num_features();
262  std::vector<MPI_Request> recv_req(required_molecules.size());
263  std::vector<MPI_Status> recv_status(required_molecules.size());
264  size_t jj = 0;
265  for (auto t : required_molecules) {
266  m_my_molecules[t].resize(num_features);
267  MPI_Irecv(m_my_molecules[t].data(), num_features, MPI_DOUBLE, m_owner_rank, t, m_mpi_comm, &(recv_req[jj++]));
268  }
269 
270  //owner starts sends
271  std::vector<std::vector<MPI_Request>> send_req;
272  std::vector<std::vector<MPI_Status>> send_status;
273  if (m_owner) {
274  send_req.resize(m_np);
275  send_status.resize(m_np);
276  for (int p = 0; p<m_np; p++) {
277  jj = 0;
278  get_required_molecules(required_molecules, p);
279  send_req[p].resize(required_molecules.size());
280  send_status[p].resize(required_molecules.size());
281  for (auto t : required_molecules) {
282  MPI_Isend(m_data[t].data(), num_features, MPI_DOUBLE, p, t, m_mpi_comm, &(send_req[p][jj++]));
283  }
284  }
285  }
286 
287  //wait for sends to finish
288  if (m_owner) {
289  for (size_t i=0; i<send_req.size(); i++) {
290  MPI_Waitall(send_req[i].size(), send_req[i].data(), send_status[i].data());
291  }
292  }
293 
294  //wait for recvs to finish
295  MPI_Waitall(recv_req.size(), recv_req.data(), recv_status.data());
296 
297  if (m_owner) {
298  std::cout << "role: " << m_reader->get_role() << " data_store_pilot2_molecular::exchange_data() time: " << get_time() - tm1 << std::endl;
299  }
300 }
