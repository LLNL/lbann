197 bool cv_process::preprocess(cv::Mat& image, unsigned int tr_start, unsigned int tr_end) {
198  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
199 
200  bool ok = true;
201 
202  if (tr_end == 0u) return true;
203  if (tr_start == 0u) {
204  if (to_flip())
205  cv::flip(image, image, how_to_flip());
206  } else if ((tr_start >= m_transforms.size()) || (tr_start >= tr_end)) {
207  return true;
208  }
209 
210  // While many transforms can update pixel values in place, some require new
211  // memory locations to write new values. In addition, at the end of a pre-
212  // processing pipeline, the values in an OpenCV matrix is copied into an
213  // Elemental matrix. Normalization typically is the last transform in a
214  // preprocessing pipeline. It is also simple enough (e.g., applying a linear
215  // function to existing values) that we can merge it with copying from one memory
216  // to another. Therefore, unless there is another preprocessing operation to be
217  // done after normalization, in which case we prefer in-place updating,
218  // we implicitly apply it during copying between memory locations to avoid
219  // redundant memory access overheads. For this reason, we treat normalization
220  // differently from other transforms. However, if a subtractor is used as a
221  // normalizer, it is treated as an ordinary transform.
222 
223  const bool lazy_normalization = to_fuse_normalizer_with_copy();
224  const unsigned int n_immediate_transforms
225  = std::min((lazy_normalization?
226  m_normalizer_idx : static_cast<unsigned int>(m_transforms.size())),
227  tr_end);
228 
229  for (size_t i = tr_start; i < n_immediate_transforms; ++i) {
230  if (m_transforms[i]->determine_transform(image)) {
231  ok = m_transforms[i]->apply(image);
232  }
233  }
234 
235  if (lazy_normalization) {
236  m_transforms[m_normalizer_idx]->determine_transform(image);
237  }
238 
239  return ok;
240 }

246 bool cv_process::postprocess(cv::Mat& image) {
247  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
248 
249  bool ok = true;
250 
251  const bool lazy_normalization = to_fuse_normalizer_with_copy();
252  const unsigned int n_immediate_transforms
253  = (lazy_normalization? m_normalizer_idx : m_transforms.size());
254 
255  // If normalizer is the last transform in the preprocessing pipeline, it will
256  // be the first in the postprocessing. In addition, it has implicitly been
257  // inversed during copying from El::Mat to cv::Mat before calling postprocess(image)
258 
259  for (size_t i = n_immediate_transforms; i > 0; --i) {
260  if (m_transforms[i-1]->determine_inverse_transform()) {
261  ok = m_transforms[i-1]->apply(image);
262  _LBANN_MILD_EXCEPTION(!ok, "inverse transform " << i-1 << " has failed!", false);
263  }
264  }
265 
266  if (to_flip()) {
267  cv::flip(image, image, how_to_flip());
268  }
269 
270  return ok;
271 }
