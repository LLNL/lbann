156  void fp_compute() override {
157  execution_mode mode = this->m_model->get_execution_mode();
158  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(m_paired_input_layer->get_data_reader(), mode);
159 
160  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
161  update_num_samples_processed(num_samples_in_batch);
162  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
163  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
165  update_num_samples_processed(num_samples_in_batch);
166  }
167 
168  int curr_mini_batch_size = this->m_model->get_current_mini_batch_size();
169  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode) && num_samples_in_batch != curr_mini_batch_size) {
170  throw lbann_exception("lbann_target_layer_distributed_minibatch: number of labels ("
171  + std::to_string(num_samples_in_batch) + ") does not match the current mini-batch size ("
172  + std::to_string(curr_mini_batch_size) + ")."
173  );
174  }
176  io_buffer->distribute_from_local_matrix(*m_ground_truth, m_paired_input_layer->get_data_reader(), mode);
177  }else {
178  std::stringstream err;
179  err << __FILE__ << " " << __LINE__ << " :: "
180  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
181  throw lbann_exception(err.str());
182  }
183  }
