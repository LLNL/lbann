36  EvalType sum_of_squares(const Mat& mat) {
37  const El::Int height = mat.Height();
38  const El::Int width = mat.Width();
39  const El::Int ldim = mat.LDim();
40  const auto& __restrict__ buf = mat.LockedBuffer();
41  EvalType sqsum = EvalType(0);
42  if (ldim == height) {
43  // Parallelize single loop if data is contiguous
44  const El::Int size = height*width;
45  #pragma omp parallel for reduction(+:sqsum)
46  for (El::Int i = 0; i < size; ++i) {
47  const EvalType val = buf[i];
48  sqsum += val * val;
49  }
50  } else {
51  // Parallelize double loop if data is not contiguous
52  #pragma omp parallel for reduction(+:sqsum) collapse(2)
53  for (El::Int j = 0; j < width; ++j) {
54  for (El::Int i = 0; i < height; ++i) {
55  const EvalType val = buf[i + j*ldim];
56  sqsum += val * val;
57  }
58  }
59  }
60  return sqsum;
61  }

100 void l2_weight_regularization::start_evaluation() {
101  if (m_scale_factor == EvalType(0)) { return; }
102 
103  // Reset terms for each weights
104  const int num_weights = m_weights.size();
105  std::fill(m_sqsums.begin(), m_sqsums.end(), EvalType(0));
106 
107 #ifdef LBANN_HAS_CUDNN
108  // Compute terms for GPU weights
109  if (m_cudnn != nullptr) {
110  const auto& num_gpus = m_cudnn->get_num_gpus();
111  const auto& weights_per_gpu = (num_weights + num_gpus - 1) / num_gpus;
112  cudnn::matrix sqsums_d(m_cudnn);
113  sqsums_d.attach_to_work_spaces(1, weights_per_gpu);
114  sqsums_d.zero();
115  for (int gpu = 0; gpu < num_gpus; ++gpu) {
116  CHECK_CUDA(cudaSetDevice(m_cudnn->get_gpu(gpu)));
117  auto&& handle = m_cudnn->get_cublas_handle(gpu);
118  CHECK_CUDA(cublasSetPointerMode(handle, CUBLAS_POINTER_MODE_DEVICE));
119  const auto& weights_start = std::min(gpu * weights_per_gpu, num_weights);
120  const auto& weights_end = std::min((gpu+1) * weights_per_gpu, num_weights);
121  for (int i = weights_start; i < weights_end; ++i) {
122  const auto& w = m_weights[i];
123  if (w->get_cudnn_manager() != nullptr) {
124  const auto& values_d = w->get_values_gpu()[gpu];
125  auto&& sqsum_d = sqsums_d.get_data(gpu) + i - weights_start;
126  cublas::dot(handle, w->get_size(), values_d, 1, values_d, 1, sqsum_d);
127  }
128  }
129  CHECK_CUDA(cublasSetPointerMode(handle, CUBLAS_POINTER_MODE_HOST));
130  }
131  Mat sqsums(1, num_weights);
132  m_cudnn->gather_from_gpus(sqsums, sqsums_d.get_locked_data(), weights_per_gpu);
133  m_cudnn->synchronize();
134  for (int i = 0; i < num_weights; ++i) {
135  m_sqsums[i] = EvalType(sqsums(0, i));
136  }
137  }
138 #endif // LBANN_HAS_CUDNN
139 
140  // Compute terms for CPU weights
141  std::fill(m_allreduce_started.begin(), m_allreduce_started.end(), false);
142  for (int i = 0; i < num_weights; ++i) {
143  const auto& w = m_weights[i];
144  if (w->get_cudnn_manager() == nullptr) {
145  const auto& values = w->get_values();
146  m_sqsums[i] = sum_of_squares(values.LockedMatrix());
147  get_comm().nb_allreduce(&(m_sqsums[i]), 1, values.DistComm(),
148  m_allreduce_reqs[i], El::mpi::SUM);
149  m_allreduce_started[i] = true;
150  }
151  }
152 
153 }

155 EvalType l2_weight_regularization::finish_evaluation() {
156  if (m_scale_factor == EvalType(0)) { return EvalType(0); }
157  for (size_t i = 0; i < m_weights.size(); ++i) {
158  if (m_allreduce_started[i]) {
159  get_comm().wait(m_allreduce_reqs[i]);
160  }
161  }
162  const auto& sqsum = std::accumulate(m_sqsums.begin(),
163  m_sqsums.end(),
164  EvalType(0));
165  return m_scale_factor * sqsum / 2;
166 }

168 void l2_weight_regularization::compute_weight_regularization() {
169  if (m_scale_factor == EvalType(0)) { return; }
170 
171 #ifdef LBANN_HAS_CUDNN
172  // Compute gradient of L2 regularization term for GPU weights
173  for (auto&& w : m_weights) {
174  if (w->get_cudnn_manager() != nullptr) {
175  cudnn::matrix values_d(m_cudnn);
176  values_d.locked_attach(w->get_values_gpu(), w->get_size());
177  w->get_optimizer()->add_to_gradient(values_d, m_scale_factor);
178  }
179  }
180 #endif // LBANN_HAS_CUDNN
181 
182  // Compute gradient of L2 regularization term for CPU weights
183  for (auto&& w : m_weights) {
184  if (w->get_cudnn_manager() == nullptr) {
185  w->get_optimizer()->add_to_gradient(w->get_values(), m_scale_factor);
186  }
187  }
188 
189 }
